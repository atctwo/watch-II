154 uint8_t TwoWire::endTransmission(bool sendStop) // Assumes Wire.beginTransaction(), Wire.write()
155 {
156  if(transmitting == 1) {
157  // txlength is howmany bytes in txbuffer have been use
158  last_error = writeTransmission(txAddress, &txBuffer[txQueued], txLength - txQueued, sendStop);
159  if(last_error == I2C_ERROR_CONTINUE){
160  txQueued = txLength;
161  } else if( last_error == I2C_ERROR_OK){
162  rxIndex = 0;
163  rxLength = rxQueued;
164  rxQueued = 0;
165  txQueued = 0; // the SendStop=true will restart all Queueing
166  }
167  } else {
168  last_error = I2C_ERROR_NO_BEGIN;
169  flush();
170  }
171  txIndex = 0;
172  txLength = 0;
173  transmitting = 0;
174  return (last_error == I2C_ERROR_CONTINUE)?I2C_ERROR_OK:last_error; // Don't return Continue for compatibility.
175 }

179 uint8_t TwoWire::requestFrom(uint16_t address, uint8_t size, bool sendStop)
180 {
181  //use internal Wire rxBuffer, multiple requestFrom()'s may be pending, try to share rxBuffer
182  uint32_t cnt = rxQueued; // currently queued reads, next available position in rxBuffer
183  if(cnt < (I2C_BUFFER_LENGTH-1) && (size + cnt) <= I2C_BUFFER_LENGTH) { // any room left in rxBuffer
184  rxQueued += size;
185  } else { // no room to receive more!
186  log_e("rxBuff overflow %d", cnt + size);
187  cnt = 0;
188  last_error = I2C_ERROR_MEMORY;
189  flush();
190  return cnt;
191  }
192 
193  last_error = readTransmission(address, &rxBuffer[cnt], size, sendStop, &cnt);
194  rxIndex = 0;
195 
196  rxLength = cnt;
197 
198  if( last_error != I2C_ERROR_CONTINUE){ // not a buffered ReSTART operation
199  // so this operation actually moved data, queuing is done.
200  rxQueued = 0;
201  txQueued = 0; // the SendStop=true will restart all Queueing or error condition
202  }
203 
204  if(last_error != I2C_ERROR_OK){ // ReSTART on read does not return any data
205  cnt = 0;
206  }
207 
208  return cnt;
209 }

211 size_t TwoWire::write(uint8_t data)
212 {
213  if(transmitting) {
214  if(txLength >= I2C_BUFFER_LENGTH) {
215  last_error = I2C_ERROR_MEMORY;
216  return 0;
217  }
218  txBuffer[txIndex] = data;
219  ++txIndex;
220  txLength = txIndex;
221  return 1;
222  }
223  last_error = I2C_ERROR_NO_BEGIN; // no begin, not transmitting
224  return 0;
225 }
