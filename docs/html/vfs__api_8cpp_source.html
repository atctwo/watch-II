19 FileImplPtr VFSImpl::open(const char* path, const char* mode)
20 {
21  if(!_mountpoint) {
22  log_e("File system is not mounted");
23  return FileImplPtr();
24  }
25 
26  if(!path || path[0] != '/') {
27  log_e("%s does not start with /", path);
28  return FileImplPtr();
29  }
30 
31  char * temp = (char *)malloc(strlen(path)+strlen(_mountpoint)+2);
32  if(!temp) {
33  log_e("malloc failed");
34  return FileImplPtr();
35  }
36 
37  sprintf(temp,"%s%s", _mountpoint, path);
38 
39  struct stat st;
40  //file lound
41  if(!stat(temp, &st)) {
42  free(temp);
43  if (S_ISREG(st.st_mode) || S_ISDIR(st.st_mode)) {
44  return std::make_shared<VFSFileImpl>(this, path, mode);
45  }
46  log_e("%s has wrong mode 0x%08X", path, st.st_mode);
47  return FileImplPtr();
48  }
49 
50  //file not found but mode permits creation
51  if(mode && mode[0] != 'r') {
52  free(temp);
53  return std::make_shared<VFSFileImpl>(this, path, mode);
54  }
55 
56  //try to open this as directory (might be mount point)
57  DIR * d = opendir(temp);
58  if(d) {
59  closedir(d);
60  free(temp);
61  return std::make_shared<VFSFileImpl>(this, path, mode);
62  }
63 
64  log_e("%s does not exist", temp);
65  free(temp);
66  return FileImplPtr();
67 }

211 VFSFileImpl::VFSFileImpl(VFSImpl* fs, const char* path, const char* mode)
212  : _fs(fs)
213  , _f(NULL)
214  , _d(NULL)
215  , _path(NULL)
216  , _isDirectory(false)
217  , _written(false)
218 {
219  char * temp = (char *)malloc(strlen(path)+strlen(_fs->_mountpoint)+1);
220  if(!temp) {
221  return;
222  }
223  sprintf(temp,"%s%s", _fs->_mountpoint, path);
224 
225  _path = strdup(path);
226  if(!_path) {
227  log_e("strdup(%s) failed", path);
228  free(temp);
229  return;
230  }
231 
232  if(!stat(temp, &_stat)) {
233  //file found
234  if (S_ISREG(_stat.st_mode)) {
235  _isDirectory = false;
236  _f = fopen(temp, mode);
237  if(!_f) {
238  log_e("fopen(%s) failed", temp);
239  }
240  } else if(S_ISDIR(_stat.st_mode)) {
241  _isDirectory = true;
242  _d = opendir(temp);
243  if(!_d) {
244  log_e("opendir(%s) failed", temp);
245  }
246  } else {
247  log_e("Unknown type 0x%08X for file %s", ((_stat.st_mode)&_IFMT), temp);
248  }
249  } else {
250  //file not found
251  if(!mode || mode[0] == 'r') {
252  //try to open as directory
253  _d = opendir(temp);
254  if(_d) {
255  _isDirectory = true;
256  } else {
257  _isDirectory = false;
258  //log_w("stat(%s) failed", temp);
259  }
260  } else {
261  //lets create this new file
262  _isDirectory = false;
263  _f = fopen(temp, mode);
264  if(!_f) {
265  log_e("fopen(%s) failed", temp);
266  }
267  }
268  }
269  free(temp);
270 }
