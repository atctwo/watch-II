66 BLEUUID::BLEUUID(std::string value) {
67  m_valueSet = true;
68  if (value.length() == 4) {
69  m_uuid.len = ESP_UUID_LEN_16;
70  m_uuid.uuid.uuid16 = 0;
71  for(int i=0;i<value.length();){
72  uint8_t MSB = value.c_str()[i];
73  uint8_t LSB = value.c_str()[i+1];
74 
75  if(MSB > '9') MSB -= 7;
76  if(LSB > '9') LSB -= 7;
77  m_uuid.uuid.uuid16 += (((MSB&0x0F) <<4) | (LSB & 0x0F))<<(2-i)*4;
78  i+=2;
79  }
80  }

91  i+=2;
