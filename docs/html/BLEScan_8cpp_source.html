41 void BLEScan::handleGAPEvent(
42  esp_gap_ble_cb_event_t event,
43  esp_ble_gap_cb_param_t* param) {
44 
45  switch(event) {
46 
47  // ---------------------------
48  // scan_rst:
49  // esp_gap_search_evt_t search_evt
50  // esp_bd_addr_t bda
51  // esp_bt_dev_type_t dev_type
52  // esp_ble_addr_type_t ble_addr_type
53  // esp_ble_evt_type_t ble_evt_type
54  // int rssi
55  // uint8_t ble_adv[ESP_BLE_ADV_DATA_LEN_MAX]
56  // int flag
57  // int num_resps
58  // uint8_t adv_data_len
59  // uint8_t scan_rsp_len
60  case ESP_GAP_BLE_SCAN_RESULT_EVT: {
61 
62  switch(param->scan_rst.search_evt) {
63  //
64  // ESP_GAP_SEARCH_INQ_CMPL_EVT
65  //
66  // Event that indicates that the duration allowed for the search has completed or that we have been
67  // asked to stop.
68  case ESP_GAP_SEARCH_INQ_CMPL_EVT: {
69  log_w("ESP_GAP_SEARCH_INQ_CMPL_EVT");
70  m_stopped = true;
71  m_semaphoreScanEnd.give();
72  if (m_scanCompleteCB != nullptr) {
73  m_scanCompleteCB(m_scanResults);
74  }
75  break;
76  } // ESP_GAP_SEARCH_INQ_CMPL_EVT
77 
78  //
79  // ESP_GAP_SEARCH_INQ_RES_EVT
80  //
81  // Result that has arrived back from a Scan inquiry.
82  case ESP_GAP_SEARCH_INQ_RES_EVT: {
83  if (m_stopped) { // If we are not scanning, nothing to do with the extra results.
84  break;
85  }
86 
87 // Examine our list of previously scanned addresses and, if we found this one already,
88 // ignore it.
89  BLEAddress advertisedAddress(param->scan_rst.bda);
90  bool found = false;
91 
92  if (m_scanResults.m_vectorAdvertisedDevices.count(advertisedAddress.toString()) != 0) {
93  found = true;
94  }
95 
96  if (found && !m_wantDuplicates) { // If we found a previous entry AND we don't want duplicates, then we are done.
97  log_d("Ignoring %s, already seen it.", advertisedAddress.toString().c_str());
98  vTaskDelay(1); // <--- allow to switch task in case we scan infinity and dont have new devices to report, or we are blocked here
99  break;
100  }
101 
102  // We now construct a model of the advertised device that we have just found for the first
103  // time.
104  // ESP_LOG_BUFFER_HEXDUMP((uint8_t*)param->scan_rst.ble_adv, param->scan_rst.adv_data_len + param->scan_rst.scan_rsp_len, ESP_LOG_DEBUG);
105  // log_w("bytes length: %d + %d, addr type: %d", param->scan_rst.adv_data_len, param->scan_rst.scan_rsp_len, param->scan_rst.ble_addr_type);
106  BLEAdvertisedDevice *advertisedDevice = new BLEAdvertisedDevice();
107  advertisedDevice->setAddress(advertisedAddress);
108  advertisedDevice->setRSSI(param->scan_rst.rssi);
109  advertisedDevice->setAdFlag(param->scan_rst.flag);
110  advertisedDevice->parseAdvertisement((uint8_t*)param->scan_rst.ble_adv, param->scan_rst.adv_data_len + param->scan_rst.scan_rsp_len);
111  advertisedDevice->setScan(this);
112  advertisedDevice->setAddressType(param->scan_rst.ble_addr_type);
113 
114  if (!found) { // If we have previously seen this device, don't record it again.
115  m_scanResults.m_vectorAdvertisedDevices.insert(std::pair<std::string, BLEAdvertisedDevice*>(advertisedAddress.toString(), advertisedDevice));
116  }
117 
118  if (m_pAdvertisedDeviceCallbacks) {
119  m_pAdvertisedDeviceCallbacks->onResult(*advertisedDevice);
120  }
121  if(found)
122  delete advertisedDevice;
123 
124  break;
125  } // ESP_GAP_SEARCH_INQ_RES_EVT
126 
127  default: {
128  break;
129  }
130  } // switch - search_evt
131 
132 
133  break;
134  } // ESP_GAP_BLE_SCAN_RESULT_EVT
135 
136  default: {
137  break;
138  } // default
139  } // End switch
140 } // gapEventHandler

194 bool BLEScan::start(uint32_t duration, void (*scanCompleteCB)(BLEScanResults), bool is_continue) {
195  log_v(">> start(duration=%d)", duration);
196 
197  m_semaphoreScanEnd.take(std::string("start"));
198  m_scanCompleteCB = scanCompleteCB; // Save the callback to be invoked when the scan completes.
199 
200  // if we are connecting to devices that are advertising even after being connected, multiconnecting peripherals
201  // then we should not clear map or we will connect the same device few times
202  if(!is_continue) {
203  for(auto _dev : m_scanResults.m_vectorAdvertisedDevices){
204  delete _dev.second;
205  }
206  m_scanResults.m_vectorAdvertisedDevices.clear();
207  }
208 
209  esp_err_t errRc = ::esp_ble_gap_set_scan_params(&m_scan_params);
210 
211  if (errRc != ESP_OK) {
212  log_e("esp_ble_gap_set_scan_params: err: %d, text: %s", errRc, GeneralUtils::errorToString(errRc));
213  m_semaphoreScanEnd.give();
214  return false;
215  }
216 
217  errRc = ::esp_ble_gap_start_scanning(duration);
218 
219  if (errRc != ESP_OK) {
220  log_e("esp_ble_gap_start_scanning: err: %d, text: %s", errRc, GeneralUtils::errorToString(errRc));
221  m_semaphoreScanEnd.give();
222  return false;
223  }
224 
225  m_stopped = false;
226 
227  log_v("<< start()");
228  return true;
229 } // start
