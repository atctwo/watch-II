38 bool SDMMCFS::begin(const char * mountpoint, bool mode1bit)
39 {
40  if(_card) {
41  return true;
42  }
43  //mount
44  sdmmc_slot_config_t slot_config = SDMMC_SLOT_CONFIG_DEFAULT();
45  sdmmc_host_t host = {
46  .flags = SDMMC_HOST_FLAG_4BIT,
47  .slot = SDMMC_HOST_SLOT_1,
48  .max_freq_khz = SDMMC_FREQ_DEFAULT,
49  .io_voltage = 3.3f,
50  .init = &sdmmc_host_init,
51  .set_bus_width = &sdmmc_host_set_bus_width,
52  .get_bus_width = &sdmmc_host_get_slot_width,
53  .set_bus_ddr_mode = &sdmmc_host_set_bus_ddr_mode,
54  .set_card_clk = &sdmmc_host_set_card_clk,
55  .do_transaction = &sdmmc_host_do_transaction,
56  .deinit = &sdmmc_host_deinit,
57  .io_int_enable = &sdmmc_host_io_int_enable,
58  .io_int_wait = &sdmmc_host_io_int_wait,
59  .command_timeout_ms = 0
60  };
61  host.max_freq_khz = SDMMC_FREQ_HIGHSPEED;
62 #ifdef BOARD_HAS_1BIT_SDMMC
63  mode1bit = true;
64 #endif
65  if(mode1bit) {
66  host.flags = SDMMC_HOST_FLAG_1BIT; //use 1-line SD mode
67  slot_config.width = 1;
68  }

70  esp_vfs_fat_sdmmc_mount_config_t mount_config = {
71  .format_if_mount_failed = false,
72  .max_files = 5,
73  .allocation_unit_size = 0
74  };
75 
76  esp_err_t ret = esp_vfs_fat_sdmmc_mount(mountpoint, &host, &slot_config, &mount_config, &_card);
77  if (ret != ESP_OK) {
78  if (ret == ESP_FAIL) {
79  log_e("Failed to mount filesystem. If you want the card to be formatted, set format_if_mount_failed = true.");
80  } else if (ret == ESP_ERR_INVALID_STATE) {
81  _impl->mountpoint(mountpoint);
82  log_w("SD Already mounted");
83  return true;
84  } else {
85  log_e("Failed to initialize the card (%d). Make sure SD card lines have pull-up resistors in place.", ret);
86  }
87  _card = NULL;
88  return false;
89  }
90  _impl->mountpoint(mountpoint);
91  return true;
92 }
