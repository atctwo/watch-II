32 TRANSPORT_HANDLE IoTHubTransport_Create(IOTHUB_CLIENT_TRANSPORT_PROVIDER protocol, const char* iotHubName, const char* iotHubSuffix)
33 {
34  TRANSPORT_HANDLE_DATA *result;
35 
36  if (protocol == NULL || iotHubName == NULL || iotHubSuffix == NULL)
37  {
38  /*Codes_SRS_IOTHUBTRANSPORT_17_002: [ If protocol is NULL, this function shall return NULL. ]*/
39  /*Codes_SRS_IOTHUBTRANSPORT_17_003: [ If iotHubName is NULL, this function shall return NULL. ]*/
40  /*Codes_SRS_IOTHUBTRANSPORT_17_004: [ If iotHubSuffix is NULL, this function shall return NULL. ]*/
41  LogError("Invalid NULL argument, protocol [%p], name [%p], suffix [%p].", protocol, iotHubName, iotHubSuffix);
42  result = NULL;
43  }
44  else
45  {
46  /*Codes_SRS_IOTHUBTRANSPORT_17_032: [ IoTHubTransport_Create shall allocate memory for the transport data. ]*/
47  result = (TRANSPORT_HANDLE_DATA*)malloc(sizeof(TRANSPORT_HANDLE_DATA));
48  if (result == NULL)
49  {
50  /*Codes_SRS_IOTHUBTRANSPORT_17_040: [ If memory allocation fails, IoTHubTransport_Create shall return NULL. ]*/
51  LogError("Transport handle was not allocated.");
52  }
53  else
54  {
55  TRANSPORT_PROVIDER * transportProtocol = (TRANSPORT_PROVIDER*)(protocol());
56  IOTHUB_CLIENT_CONFIG upperConfig;
57  upperConfig.deviceId = NULL;
58  upperConfig.deviceKey = NULL;
59  upperConfig.iotHubName = iotHubName;
60  upperConfig.iotHubSuffix = iotHubSuffix;
61  upperConfig.protocol = protocol;

63 
64  IOTHUBTRANSPORT_CONFIG transportLLConfig;

70  result->transportLLHandle = transportProtocol->IoTHubTransport_Create(&transportLLConfig);
71  if (result->transportLLHandle == NULL)
72  {
73  /*Codes_SRS_IOTHUBTRANSPORT_17_006: [ If the creation of the transport fails, IoTHubTransport_Create shall return NULL. ]*/
74  LogError("Lower Layer transport not created.");
75  free(result);
76  result = NULL;
77  }
78  else
79  {
80  /*Codes_SRS_IOTHUBTRANSPORT_17_007: [ IoTHubTransport_Create shall create the transport lock by Calling Lock_Init. ]*/
81  result->lockHandle = Lock_Init();
82  if (result->lockHandle == NULL)
83  {
84  /*Codes_SRS_IOTHUBTRANSPORT_17_008: [ If the lock creation fails, IoTHubTransport_Create shall return NULL. ]*/
85  LogError("transport Lock not created.");
86  transportProtocol->IoTHubTransport_Destroy(result->transportLLHandle);
87  free(result);
88  result = NULL;
89  }
90  else if ((result->clientsLockHandle = Lock_Init()) == NULL)
91  {
92  LogError("clients Lock not created.");
93  Lock_Deinit(result->lockHandle);
94  transportProtocol->IoTHubTransport_Destroy(result->transportLLHandle);
95  free(result);
96  result = NULL;
97  }
98  else
99  {
100  /*Codes_SRS_IOTHUBTRANSPORT_17_038: [ IoTHubTransport_Create shall call VECTOR_Create to make a list of IOTHUB_CLIENT_HANDLE using this transport. ]*/
101  result->clients = VECTOR_create(sizeof(IOTHUB_CLIENT_HANDLE));
102  if (result->clients == NULL)
103  {
104  /*Codes_SRS_IOTHUBTRANSPORT_17_039: [ If the Vector creation fails, IoTHubTransport_Create shall return NULL. ]*/
105  /*Codes_SRS_IOTHUBTRANSPORT_17_009: [ IoTHubTransport_Create shall clean up any resources it creates if the function does not succeed. ]*/
106  LogError("clients list not created.");
107  Lock_Deinit(result->clientsLockHandle);
108  Lock_Deinit(result->lockHandle);
109  transportProtocol->IoTHubTransport_Destroy(result->transportLLHandle);
110  free(result);
111  result = NULL;
112  }
113  else
114  {
115  /*Codes_SRS_IOTHUBTRANSPORT_17_001: [ IoTHubTransport_Create shall return a non-NULL handle on success.]*/
116  result->stopThread = 1;
117  result->clientDoWork = NULL;
118  result->workerThreadHandle = NULL; /* create thread when work needs to be done */
119  result->IoTHubTransport_GetHostname = transportProtocol->IoTHubTransport_GetHostname;
120  result->IoTHubTransport_SetOption = transportProtocol->IoTHubTransport_SetOption;
121  result->IoTHubTransport_Create = transportProtocol->IoTHubTransport_Create;

139 static void multiplexed_client_do_work(TRANSPORT_HANDLE_DATA* transportData)
140 {
141  if (Lock(transportData->clientsLockHandle) != LOCK_OK)
142  {
143  LogError("failed to lock for multiplexed_client_do_work");
144  }
145  else
146  {
147  size_t numberOfClients;
148  size_t iterator;
149 
150  numberOfClients = VECTOR_size(transportData->clients);
151  for (iterator = 0; iterator < numberOfClients; iterator++)
152  {
153  IOTHUB_CLIENT_HANDLE* clientHandle = (IOTHUB_CLIENT_HANDLE*)VECTOR_element(transportData->clients, iterator);
154 
155  if (clientHandle != NULL)

158  }
159  }
160 
161  if (Unlock(transportData->clientsLockHandle) != LOCK_OK)

164  }

168 static int transport_worker_thread(void* threadArgument)
169 {
170  TRANSPORT_HANDLE_DATA* transportData = (TRANSPORT_HANDLE_DATA*)threadArgument;
171 
172  while (1)
173  {
174  /*Codes_SRS_IOTHUBTRANSPORT_17_030: [ All calls to lower layer transport DoWork shall be protected by the lock created in IoTHubTransport_Create. ]*/
175  if (Lock(transportData->lockHandle) == LOCK_OK)
176  {
177  /*Codes_SRS_IOTHUBTRANSPORT_17_031: [ If acquiring the lock fails, lower layer transport DoWork shall not be called. ]*/
178  if (transportData->stopThread)
179  {
180  /*Codes_SRS_IOTHUBTRANSPORT_17_028: [ The thread shall exit when IoTHubTransport_EndWorkerThread has been called for each clientHandle which invoked IoTHubTransport_StartWorkerThread. ]*/
181  (void)Unlock(transportData->lockHandle);
182  break;
183  }
184  else
185  {
186  (transportData->IoTHubTransport_DoWork)(transportData->transportLLHandle, NULL);
187 
188  (void)Unlock(transportData->lockHandle);
189  }
190  }
191 
192  multiplexed_client_do_work(transportData);
193 
194  /*Codes_SRS_IOTHUBTRANSPORT_17_029: [ The thread shall call lower layer transport DoWork every 1 ms. ]*/
195  ThreadAPI_Sleep(1);
196  }
197 
198  ThreadAPI_Exit(0);
199  return 0;
200 }

210 static IOTHUB_CLIENT_RESULT start_worker_if_needed(TRANSPORT_HANDLE_DATA * transportData, IOTHUB_CLIENT_HANDLE clientHandle)
211 {
212  IOTHUB_CLIENT_RESULT result;
213  if (transportData->workerThreadHandle == NULL)
214  {
215  /*Codes_SRS_IOTHUBTRANSPORT_17_018: [ If the worker thread does not exist, IoTHubTransport_StartWorkerThread shall start the thread using ThreadAPI_Create. ]*/
216  transportData->stopThread = 0;
217  if (ThreadAPI_Create(&transportData->workerThreadHandle, transport_worker_thread, transportData) != THREADAPI_OK)
218  {
219  transportData->workerThreadHandle = NULL;
220  }
221  }
222  if (transportData->workerThreadHandle != NULL)
223  {
224  if (Lock(transportData->clientsLockHandle) != LOCK_OK)
225  {
226  LogError("failed to lock for start_worker_if_needed");
227  result = IOTHUB_CLIENT_ERROR;
228  }
229  else
230  {
231  /*Codes_SRS_IOTHUBTRANSPORT_17_020: [ IoTHubTransport_StartWorkerThread shall search for IoTHubClient clientHandle in the list of IoTHubClient handles. ]*/
232  bool addToList = ((VECTOR_size(transportData->clients) == 0) || (VECTOR_find_if(transportData->clients, find_by_handle, clientHandle) == NULL));
233  if (addToList)
234  {
235  /*Codes_SRS_IOTHUBTRANSPORT_17_021: [ If handle is not found, then clientHandle shall be added to the list. ]*/
236  if (VECTOR_push_back(transportData->clients, &clientHandle, 1) != 0)
237  {
238  LogError("Failed adding device to list (VECTOR_push_back failed)");
239  /*Codes_SRS_IOTHUBTRANSPORT_17_042: [ If Adding to the client list fails, IoTHubTransport_StartWorkerThread shall return IOTHUB_CLIENT_ERROR. ]*/
240  result = IOTHUB_CLIENT_ERROR;
241  }
242  else
243  {
244  result = IOTHUB_CLIENT_OK;
245  }
246  }
247  else
248  {
249  result = IOTHUB_CLIENT_OK;
250  }
251 
252  if (Unlock(transportData->clientsLockHandle) != LOCK_OK)
253  {
254  LogError("failed to unlock on start_worker_if_needed");
255  }
256  }
257  }
258  else
259  {
260  result = IOTHUB_CLIENT_ERROR;
261  }
262  return result;
263 }

288 static bool signal_end_worker_thread(TRANSPORT_HANDLE_DATA * transportData, IOTHUB_CLIENT_HANDLE clientHandle)
289 {
290  bool okToJoin;
291 
292  if (Lock(transportData->clientsLockHandle) != LOCK_OK)
293  {
294  LogError("failed to lock for signal_end_worker_thread");
295  okToJoin = false;
296  }
297  else
298  {
299  void* element = VECTOR_find_if(transportData->clients, find_by_handle, clientHandle);
300  if (element != NULL)
301  {
302  /*Codes_SRS_IOTHUBTRANSPORT_17_026: [ IoTHubTransport_EndWorkerThread shall remove clientHandlehandle from handle list. ]*/
303  VECTOR_erase(transportData->clients, element, 1);
304  }
305  /*Codes_SRS_IOTHUBTRANSPORT_17_025: [ If the worker thread does not exist, then IoTHubTransport_EndWorkerThread shall return. ]*/
306  if (transportData->workerThreadHandle != NULL)
307  {
308  if (VECTOR_size(transportData->clients) == 0)
309  {
310  stop_worker_thread(transportData);
311  okToJoin = true;
312  }
313  else
314  {
315  okToJoin = false;
316  }
317  }
318  else
319  {
320  okToJoin = false;
321  }
322 
323  if (Unlock(transportData->clientsLockHandle) != LOCK_OK)
324  {
325  LogError("failed to unlock on signal_end_worker_thread");
326  }
327  }
328  return okToJoin;

331 void IoTHubTransport_Destroy(TRANSPORT_HANDLE transportHandle)
332 {
333  /*Codes_SRS_IOTHUBTRANSPORT_17_011: [ IoTHubTransport_Destroy shall do nothing if transportHandle is NULL. ]*/
334  if (transportHandle != NULL)
335  {
336  TRANSPORT_HANDLE_DATA * transportData = (TRANSPORT_HANDLE_DATA*)transportHandle;
337  /*Codes_SRS_IOTHUBTRANSPORT_17_033: [ IoTHubTransport_Destroy shall lock the transport lock. ]*/

348  wait_worker_thread(transportData);

350  Lock_Deinit(transportData->lockHandle);
351  (transportData->IoTHubTransport_Destroy)(transportData->transportLLHandle);
352  VECTOR_destroy(transportData->clients);
353  Lock_Deinit(transportData->clientsLockHandle);
354  free(transportHandle);
355  }
