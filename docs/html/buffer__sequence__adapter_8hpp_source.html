157  template <typename Iterator>
158  void init(Iterator begin, Iterator end)
159  {
160  Iterator iter = begin;
161  for (; iter != end && count_ < max_buffers; ++iter, ++count_)
162  {
163  Buffer buffer(*iter);
164  init_native_buffer(buffers_[count_], buffer);
165  total_buffer_size_ += buffer.size();
166  }
167  }

169  template <typename Iterator>
170  static bool all_empty(Iterator begin, Iterator end)
171  {
172  Iterator iter = begin;
173  std::size_t i = 0;
174  for (; iter != end && i < max_buffers; ++iter, ++i)
175  if (Buffer(*iter).size() > 0)
176  return false;
177  return true;
178  }

187  buffer.data();
188  }
189  }
190 
191  template <typename Iterator>
192  static Buffer first(Iterator begin, Iterator end)
193  {
194  Iterator iter = begin;
195  for (; iter != end; ++iter)
196  {
197  Buffer buffer(*iter);
198  if (buffer.size() != 0)
199  return buffer;
200  }
201  return Buffer();
202  }
