115 wl_status_t WiFiSTAClass::begin(const char* ssid, const char *passphrase, int32_t channel, const uint8_t* bssid, bool connect)
116 {
117 
118  if(!WiFi.enableSTA(true)) {
119  log_e("STA enable failed!");
120  return WL_CONNECT_FAILED;
121  }
122 
123  if(!ssid || *ssid == 0x00 || strlen(ssid) > 31) {
124  log_e("SSID too long or missing!");
125  return WL_CONNECT_FAILED;
126  }
127 
128  if(passphrase && strlen(passphrase) > 64) {
129  log_e("passphrase too long!");
130  return WL_CONNECT_FAILED;
131  }
132 
133  wifi_config_t conf;
134  memset(&conf, 0, sizeof(wifi_config_t));
135  strcpy(reinterpret_cast<char*>(conf.sta.ssid), ssid);
136 
137  if(passphrase) {
138  if (strlen(passphrase) == 64){ // it's not a passphrase, is the PSK
139  memcpy(reinterpret_cast<char*>(conf.sta.password), passphrase, 64);
140  } else {
141  strcpy(reinterpret_cast<char*>(conf.sta.password), passphrase);
142  }
143  }
144 
145  if(bssid) {
146  conf.sta.bssid_set = 1;
147  memcpy((void *) &conf.sta.bssid[0], (void *) bssid, 6);
148  }
149 
150  if(channel > 0 && channel <= 13) {
151  conf.sta.channel = channel;
152  }
153 
154  wifi_config_t current_conf;
155  esp_wifi_get_config(WIFI_IF_STA, &current_conf);
156  if(!sta_config_equal(current_conf, conf)) {
157  if(esp_wifi_disconnect()){
158  log_e("disconnect failed!");
159  return WL_CONNECT_FAILED;
160  }
161 
162  esp_wifi_set_config(WIFI_IF_STA, &conf);
163  } else if(status() == WL_CONNECTED){
164  return WL_CONNECTED;
165  } else {
166  esp_wifi_set_config(WIFI_IF_STA, &conf);
167  }
168 
169  if(!_useStaticIp) {
170  if(tcpip_adapter_dhcpc_start(TCPIP_ADAPTER_IF_STA) == ESP_ERR_TCPIP_ADAPTER_DHCPC_START_FAILED){
171  log_e("dhcp client start failed!");
172  return WL_CONNECT_FAILED;
173  }
174  } else {
175  tcpip_adapter_dhcpc_stop(TCPIP_ADAPTER_IF_STA);
176  }
177 
178  if(connect && esp_wifi_connect()) {
179  log_e("connect failed!");
180  return WL_CONNECT_FAILED;
181  }
182 
183  return status();
184 }

277 bool WiFiSTAClass::config(IPAddress local_ip, IPAddress gateway, IPAddress subnet, IPAddress dns1, IPAddress dns2)
278 {
279  esp_err_t err = ESP_OK;
280 
281  if(!WiFi.enableSTA(true)) {
282  return false;
283  }
284 
285  tcpip_adapter_ip_info_t info;
286 
287  if(local_ip != (uint32_t)0x00000000){
288  info.ip.addr = static_cast<uint32_t>(local_ip);
289  info.gw.addr = static_cast<uint32_t>(gateway);
290  info.netmask.addr = static_cast<uint32_t>(subnet);
291  } else {
292  info.ip.addr = 0;
293  info.gw.addr = 0;
294  info.netmask.addr = 0;
295  }
296 
297  err = tcpip_adapter_dhcpc_stop(TCPIP_ADAPTER_IF_STA);
298  if(err != ESP_OK && err != ESP_ERR_TCPIP_ADAPTER_DHCP_ALREADY_STOPPED){
299  log_e("DHCP could not be stopped! Error: %d", err);
300  return false;
301  }
302 
303  err = tcpip_adapter_set_ip_info(TCPIP_ADAPTER_IF_STA, &info);
304  if(err != ERR_OK){
305  log_e("STA IP could not be configured! Error: %d", err);
306  return false;
307  }
308 
309  if(info.ip.addr){
310  _useStaticIp = true;
311  } else {
312  err = tcpip_adapter_dhcpc_start(TCPIP_ADAPTER_IF_STA);
313  if(err == ESP_ERR_TCPIP_ADAPTER_DHCPC_START_FAILED){
314  log_e("dhcp client start failed!");
315  return false;
316  }
317  _useStaticIp = false;
318  }
319 
320  ip_addr_t d;
321  d.type = IPADDR_TYPE_V4;
322 
323  if(dns1 != (uint32_t)0x00000000) {
324  // Set DNS1-Server
325  d.u_addr.ip4.addr = static_cast<uint32_t>(dns1);
326  dns_setserver(0, &d);
327  }
328 
329  if(dns2 != (uint32_t)0x00000000) {
330  // Set DNS2-Server
331  d.u_addr.ip4.addr = static_cast<uint32_t>(dns2);
332  dns_setserver(1, &d);
333  }
334 
335  return true;
336 }

736 void WiFiSTAClass::_smartConfigCallback(uint32_t st, void* result) {
737  smartconfig_status_t status = (smartconfig_status_t) st;
738  log_d("Status: %s", sc_status_strings[st % 5]);
739  if (status == SC_STATUS_GETTING_SSID_PSWD) {
740 #if ARDUHAL_LOG_LEVEL >= ARDUHAL_LOG_LEVEL_DEBUG
741  smartconfig_type_t * type = (smartconfig_type_t *)result;
742  log_d("Type: %s", sc_type_strings[*type % 3]);
743 #endif
744  } else if (status == SC_STATUS_LINK) {
745  wifi_sta_config_t *sta_conf = reinterpret_cast<wifi_sta_config_t *>(result);
746  log_d("SSID: %s", (char *)(sta_conf->ssid));
747  sta_conf->bssid_set = 0;
748  esp_wifi_set_config(WIFI_IF_STA, (wifi_config_t *)sta_conf);
749  esp_wifi_connect();
750  _smartConfigDone = true;

753 #if ARDUHAL_LOG_LEVEL >= ARDUHAL_LOG_LEVEL_DEBUG
754  ip4_addr_t * ip = (ip4_addr_t *)result;
755  log_d("Sender IP: " IPSTR, IP2STR(ip));
