320 // Compare a name from certificate and domain name, return true if they match
321 static bool matchName(const std::string& name, const std::string& domainName)
322 {
323  size_t wildcardPos = name.find('*');
324  if (wildcardPos == std::string::npos) {
325  // Not a wildcard, expect an exact match
326  return name == domainName;
327  }
328 
329  size_t firstDotPos = name.find('.');
330  if (wildcardPos > firstDotPos) {
331  // Wildcard is not part of leftmost component of domain name
332  // Do not attempt to match (rfc6125 6.4.3.1)
333  return false;
334  }
335  if (wildcardPos != 0 || firstDotPos != 1) {
336  // Matching of wildcards such as baz*.example.com and b*z.example.com
337  // is optional. Maybe implement this in the future?
338  return false;
339  }
340  size_t domainNameFirstDotPos = domainName.find('.');
341  if (domainNameFirstDotPos == std::string::npos) {
342  return false;
343  }
344  return domainName.substr(domainNameFirstDotPos) == name.substr(firstDotPos);
345 }

347 // Verifies certificate provided by the peer to match specified SHA256 fingerprint
348 bool verify_ssl_fingerprint(sslclient_context *ssl_client, const char* fp, const char* domain_name)
349 {
350  // Convert hex string to byte array
351  uint8_t fingerprint_local[32];
352  int len = strlen(fp);
353  int pos = 0;
354  for (size_t i = 0; i < sizeof(fingerprint_local); ++i) {
355  while (pos < len && ((fp[pos] == ' ') || (fp[pos] == ':'))) {
356  ++pos;
357  }
358  if (pos > len - 2) {
359  log_d("pos:%d len:%d fingerprint too short", pos, len);
360  return false;
361  }
362  uint8_t high, low;
363  if (!parseHexNibble(fp[pos], &high) || !parseHexNibble(fp[pos+1], &low)) {
364  log_d("pos:%d len:%d invalid hex sequence: %c%c", pos, len, fp[pos], fp[pos+1]);
365  return false;
366  }
367  pos += 2;
368  fingerprint_local[i] = low | (high << 4);
369  }
370 
371  // Get certificate provided by the peer
372  const mbedtls_x509_crt* crt = mbedtls_ssl_get_peer_cert(&ssl_client->ssl_ctx);
373 
374  if (!crt)
375  {
376  log_d("could not fetch peer certificate");
377  return false;
378  }
379 
380  // Calculate certificate's SHA256 fingerprint
381  uint8_t fingerprint_remote[32];
382  mbedtls_sha256_context sha256_ctx;
383  mbedtls_sha256_init(&sha256_ctx);
384  mbedtls_sha256_starts(&sha256_ctx, false);
385  mbedtls_sha256_update(&sha256_ctx, crt->raw.p, crt->raw.len);
386  mbedtls_sha256_finish(&sha256_ctx, fingerprint_remote);
387 
388  // Check if fingerprints match
389  if (memcmp(fingerprint_local, fingerprint_remote, 32))
390  {
391  log_d("fingerprint doesn't match");
392  return false;
393  }
394 
395  // Additionally check if certificate has domain name if provided
396  if (domain_name)
397  return verify_ssl_dn(ssl_client, domain_name);
398  else
399  return true;
400 }

402 // Checks if peer certificate has specified domain in CN or SANs
403 bool verify_ssl_dn(sslclient_context *ssl_client, const char* domain_name)
404 {
405  log_d("domain name: '%s'", (domain_name)?domain_name:"(null)");
406  std::string domain_name_str(domain_name);
407  std::transform(domain_name_str.begin(), domain_name_str.end(), domain_name_str.begin(), ::tolower);
408 
409  // Get certificate provided by the peer
410  const mbedtls_x509_crt* crt = mbedtls_ssl_get_peer_cert(&ssl_client->ssl_ctx);
411 
412  // Check for domain name in SANs
413  const mbedtls_x509_sequence* san = &crt->subject_alt_names;
414  while (san != nullptr)
415  {
416  std::string san_str((const char*)san->buf.p, san->buf.len);
417  std::transform(san_str.begin(), san_str.end(), san_str.begin(), ::tolower);
418 
419  if (matchName(san_str, domain_name_str))
420  return true;
421 
422  log_d("SAN '%s': no match", san_str.c_str());
423 
424  // Fetch next SAN
425  san = san->next;
426  }
427 
428  // Check for domain name in CN
429  const mbedtls_asn1_named_data* common_name = &crt->subject;
430  while (common_name != nullptr)
431  {
432  // While iterating through DN objects, check for CN object
433  if (!MBEDTLS_OID_CMP(MBEDTLS_OID_AT_CN, &common_name->oid))
434  {
435  std::string common_name_str((const char*)common_name->val.p, common_name->val.len);
436 
437  if (matchName(common_name_str, domain_name_str))
438  return true;
439 
440  log_d("CN '%s': not match", common_name_str.c_str());
441  }
442 
443  // Fetch next DN object
444  common_name = common_name->next;
445  }
446 
447  return false;
