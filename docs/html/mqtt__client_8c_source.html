572 static void recvCompleteCallback(void* context, CONTROL_PACKET_TYPE packet, int flags, BUFFER_HANDLE headerData)
573 {
574  MQTT_CLIENT* mqtt_client = (MQTT_CLIENT*)context;
575  if ((mqtt_client != NULL && headerData != NULL) || packet == PINGRESP_TYPE)
576  {
577  size_t len = BUFFER_length(headerData);
578  uint8_t* iterator = BUFFER_u_char(headerData);
579 
580  logIncomingRawTrace(mqtt_client, packet, (uint8_t)flags, iterator, len);
581 
582  if ((iterator != NULL && len > 0) || packet == PINGRESP_TYPE)
583  {
584  switch (packet)
585  {
586  case CONNACK_TYPE:
587  {
588  if (mqtt_client->fnOperationCallback != NULL)
589  {
590  STRING_HANDLE trace_log = NULL;
591 
592  /*Codes_SRS_MQTT_CLIENT_07_028: [If the actionResult parameter is of type CONNECT_ACK then the msgInfo value shall be a CONNECT_ACK structure.]*/
593  CONNECT_ACK connack = { 0 };
594  connack.isSessionPresent = (byteutil_readByte(&iterator) == 0x1) ? true : false;
595  connack.returnCode = byteutil_readByte(&iterator);
596 
597  if (mqtt_client->logTrace)
598  {
599  trace_log = STRING_construct_sprintf("CONNACK | SESSION_PRESENT: %s | RETURN_CODE: 0x%x", connack.isSessionPresent ? TRUE_CONST : FALSE_CONST, connack.returnCode);
600  log_incoming_trace(mqtt_client, trace_log);
601  STRING_delete(trace_log);
602  }
603  mqtt_client->fnOperationCallback(mqtt_client, MQTT_CLIENT_ON_CONNACK, (void*)&connack, mqtt_client->ctx);
604 
605  if (connack.returnCode == CONNECTION_ACCEPTED)
606  {
607  mqtt_client->clientConnected = true;
608  }
609  }
610  else
611  {
612  LOG(AZ_LOG_ERROR, LOG_LINE, "fnOperationCallback NULL");
613  }
614  break;
615  }
616  case PUBLISH_TYPE:
617  {
618  if (mqtt_client->fnMessageRecv != NULL)
619  {
620  STRING_HANDLE trace_log = NULL;
621 
622  bool isDuplicateMsg = (flags & DUPLICATE_FLAG_MASK) ? true : false;
623  bool isRetainMsg = (flags & RETAIN_FLAG_MASK) ? true : false;
624  QOS_VALUE qosValue = (flags == 0) ? DELIVER_AT_MOST_ONCE : (flags & QOS_LEAST_ONCE_FLAG_MASK) ? DELIVER_AT_LEAST_ONCE : DELIVER_EXACTLY_ONCE;
625 
626  if (mqtt_client->logTrace)
627  {
628  trace_log = STRING_construct_sprintf("PUBLISH | IS_DUP: %s | RETAIN: %d | QOS: %s", isDuplicateMsg ? TRUE_CONST : FALSE_CONST,
629  isRetainMsg ? 1 : 0, ENUM_TO_STRING(QOS_VALUE, qosValue) );
630  }
631 
632  uint8_t* initialPos = iterator;

642  }
643  }

651  length = len - (iterator - initialPos);
652  if (qosValue != DELIVER_AT_MOST_ONCE)

654  packetId = byteutil_read_uint16(&iterator, length);
655  if (mqtt_client->logTrace)

657  STRING_sprintf(trace_log, " | PACKET_ID: %"PRIu16, packetId);
658  }
659  }

661

665  LOG(AZ_LOG_ERROR, LOG_LINE, "failure in mqttmessage_create");

668  STRING_delete(trace_log);
669  }
670  }

677  set_error_callback(mqtt_client, MQTT_CLIENT_MEMORY_ERROR);
678  if (trace_log != NULL) {
679  STRING_delete(trace_log);
680  }
681  }
682  else
683  {
684  if (mqtt_client->logTrace)
685  {
686  STRING_sprintf(trace_log, " | PAYLOAD_LEN: %lu", length);

688  STRING_delete(trace_log);
689  }
690 
691  mqtt_client->fnMessageRecv(msgHandle, mqtt_client->ctx);
692 
693  BUFFER_HANDLE pubRel = NULL;
694  if (qosValue == DELIVER_EXACTLY_ONCE)
695  {

699  LOG(AZ_LOG_ERROR, LOG_LINE, "Failed to allocate publish receive message.");
700  set_error_callback(mqtt_client, MQTT_CLIENT_MEMORY_ERROR);
701  }
702  }
703  else if (qosValue == DELIVER_AT_LEAST_ONCE)
704  {
705  pubRel = mqtt_codec_publishAck(packetId);
706  if (pubRel == NULL)
707  {
708  LOG(AZ_LOG_ERROR, LOG_LINE, "Failed to allocate publish ack message.");
709  set_error_callback(mqtt_client, MQTT_CLIENT_MEMORY_ERROR);
710  }
711  }
712  if (pubRel != NULL)
713  {
714  size_t size = BUFFER_length(pubRel);
715  (void)sendPacketItem(mqtt_client, BUFFER_u_char(pubRel), size);
716  BUFFER_delete(pubRel);
717  }
718  }
719  mqttmessage_destroy(msgHandle);
720  }
721  free(topicName);
722  }
723  }
724  break;
725  }
726  case PUBACK_TYPE:
727  case PUBREC_TYPE:
728  case PUBREL_TYPE:
729  case PUBCOMP_TYPE:

737  (packet == PUBREC_TYPE) ? MQTT_CLIENT_ON_PUBLISH_RECV :

749  STRING_delete(trace_log);
750  }

754  if (packet == PUBREC_TYPE)

777  }
778  }
779  break;
780  }
781  case SUBACK_TYPE:

783  if (mqtt_client->fnOperationCallback)
784  {
785  STRING_HANDLE trace_log = NULL;
786

793

811  }

816  STRING_delete(trace_log);
817  }
818  mqtt_client->fnOperationCallback(mqtt_client, MQTT_CLIENT_ON_SUBSCRIBE_ACK, (void*)&suback, mqtt_client->ctx);
819  free(suback.qosReturn);
820  }
821  else

824  set_error_callback(mqtt_client, MQTT_CLIENT_MEMORY_ERROR);
825  }
826  }
827  break;
828  }
829  case UNSUBACK_TYPE:
830  {
831  if (mqtt_client->fnOperationCallback)
832  {
833  STRING_HANDLE trace_log = NULL;
834 
835  /*Codes_SRS_MQTT_CLIENT_07_031: [If the actionResult parameter is of type UNSUBACK_TYPE then the msgInfo value shall be a UNSUBSCRIBE_ACK structure.]*/

843  STRING_delete(trace_log);
844  }
845  mqtt_client->fnOperationCallback(mqtt_client, MQTT_CLIENT_ON_UNSUBSCRIBE_ACK, (void*)&unsuback, mqtt_client->ctx);
846  }
847  break;
848  }
849  case PINGRESP_TYPE:
850  mqtt_client->timeSincePing = 0;
851  if (mqtt_client->logTrace)
852  {

855  STRING_delete(trace_log);

865  }
866  }
867  }
