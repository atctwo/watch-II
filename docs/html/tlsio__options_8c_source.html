13 // Initialize the TLSIO_OPTIONS struct
14 void tlsio_options_initialize(TLSIO_OPTIONS* options, int supported_options)
15 {
16  // Using static function rules, so 'options' is not checked for NULL
17  //
18  // The supported_options value does not need validation because undefined bits are
19  // ignored, while any valid missing bits result in an "option not supported" error
20  // that will show up in unit testing.
21  options->supported_options = supported_options;
22  options->trusted_certs = NULL;
23  options->x509_type = TLSIO_OPTIONS_x509_TYPE_UNSPECIFIED;
24  options->x509_cert = NULL;
25  options->x509_key = NULL;
26 }

28 static int set_and_validate_x509_type(TLSIO_OPTIONS* options, TLSIO_OPTIONS_x509_TYPE x509_type)
29 {
30  int result;
31  if ((options->supported_options & x509_type) == 0)
32  {
33  // This case also rejects the nonsensical TLSIO_OPTIONS_x509_TYPE_UNSPECIFIED
34  LogError("Unsupported x509 type: %d", x509_type);
35  result = __FAILURE__;
36  }
37  else if (options->x509_type == TLSIO_OPTIONS_x509_TYPE_UNSPECIFIED)
38  {
39  // Initial type setting
40  options->x509_type = x509_type;
41  result = 0;
42  }
43  else if (options->x509_type != x509_type)
44  {
45  LogError("Supplied x509 type conflicts with previously set x509");
46  result = __FAILURE__;
47  }
48  else
49  {
50  // The types match okay
51  result = 0;
52  }
53 
54  return result;
55 }

81 TLSIO_OPTIONS_RESULT tlsio_options_destroy_option(const char* name, const void* value)
82 {
83  TLSIO_OPTIONS_RESULT result;
84  if (name == NULL || value == NULL)
85  {
86  LogError("NULL parameter: name: %p, value: %p", name, value);
87  result = TLSIO_OPTIONS_RESULT_ERROR;

102 {

104 
105  if (name == NULL || value == NULL || out_value == NULL)
106  {

131 TLSIO_OPTIONS_RESULT tlsio_options_set(TLSIO_OPTIONS* options,
132  const char* optionName, const void* value)
133 {
134  TLSIO_OPTIONS_RESULT result;
135  char* copied_value = NULL;
136 
137  if (options == NULL || optionName == NULL || value == NULL)
138  {
139  LogError("NULL parameter: options: %p, optionName: %p, value: %p",
140  options, optionName, value);
141  result = TLSIO_OPTIONS_RESULT_ERROR;
142  }
143  else if (!is_supported_string_option(optionName))
144  {
145  result = TLSIO_OPTIONS_RESULT_NOT_HANDLED;
146  }
147  else if(mallocAndStrcpy_s(&copied_value, value) != 0)
148  {
149  LogError("unable to mallocAndStrcpy_s option value");
150  result = TLSIO_OPTIONS_RESULT_ERROR;
151  }
152  else if (strcmp(OPTION_TRUSTED_CERT, optionName) == 0)
153  {
154  if ((options->supported_options & TLSIO_OPTION_BIT_TRUSTED_CERTS) == 0)
155  {
156  LogError("Trusted certs option not supported");
157  result = TLSIO_OPTIONS_RESULT_ERROR;
158  }
159  else if (options->trusted_certs != NULL)
160  {
161  LogError("unable to set trusted cert option more than once");
162  result = TLSIO_OPTIONS_RESULT_ERROR;
163  }
164  else
165  {
166  options->trusted_certs = copied_value;
167  result = TLSIO_OPTIONS_RESULT_SUCCESS;
168  }
169  }
170  else if (strcmp(SU_OPTION_X509_CERT, optionName) == 0 || strcmp(OPTION_X509_ECC_CERT, optionName) == 0)
171  {
172  TLSIO_OPTIONS_x509_TYPE this_type = (strcmp(SU_OPTION_X509_CERT, optionName) == 0) ? TLSIO_OPTIONS_x509_TYPE_RSA : TLSIO_OPTIONS_x509_TYPE_ECC;
173  if (options->x509_cert != NULL)
174  {
175  LogError("unable to set x509 cert more than once");
176  result = TLSIO_OPTIONS_RESULT_ERROR;
177  }
178  else if (set_and_validate_x509_type(options, this_type) != 0)
179  {
180  // Error logged by helper
181  result = TLSIO_OPTIONS_RESULT_ERROR;
182  }
183  else
184  {
185  options->x509_cert = copied_value;
186  result = TLSIO_OPTIONS_RESULT_SUCCESS;
187  }
188  }
189  else if (strcmp(SU_OPTION_X509_PRIVATE_KEY, optionName) == 0 || strcmp(OPTION_X509_ECC_KEY, optionName) == 0)
190  {
191  TLSIO_OPTIONS_x509_TYPE this_type = (strcmp(SU_OPTION_X509_PRIVATE_KEY, optionName) == 0) ? TLSIO_OPTIONS_x509_TYPE_RSA : TLSIO_OPTIONS_x509_TYPE_ECC;
192  if (options->x509_key != NULL)
193  {
194  LogError("unable to set x509 key more than once");
195  result = TLSIO_OPTIONS_RESULT_ERROR;
196  }
197  else if (set_and_validate_x509_type(options, this_type) != 0)
198  {
199  // Error logged by helper
200  result = TLSIO_OPTIONS_RESULT_ERROR;
201  }
202  else
203  {
204  options->x509_key = copied_value;
205  result = TLSIO_OPTIONS_RESULT_SUCCESS;
206  }
207  }
208  else
209  {
210  // This is logically impossible due to earlier tests, so just quiet the compiler
211  result = TLSIO_OPTIONS_RESULT_ERROR;
212  }
213 
214  if (result != TLSIO_OPTIONS_RESULT_SUCCESS)
215  {
216  free(copied_value);
217  }
218 
219  return result;
220 }

248 OPTIONHANDLER_HANDLE tlsio_options_retrieve_options_ex(TLSIO_OPTIONS* options,
249  pfCloneOption cloneOption, pfDestroyOption destroyOption, pfSetOption setOption)
250 {
251  OPTIONHANDLER_HANDLE result;
252  if (options == NULL || cloneOption == NULL || destroyOption == NULL || setOption == NULL)
253  {
254  LogError("Null parameter in options: %p, cloneOption: %p, destroyOption: %p, setOption: %p",
255  options, cloneOption, destroyOption, setOption);
256  result = NULL;
257  }
258  else
259  {
260  result = OptionHandler_Create(cloneOption, destroyOption, setOption);
261  if (result == NULL)
262  {
263  LogError("OptionHandler_Create failed");
264  /*return as is*/
265  }
266  else if (
267  (options->trusted_certs != NULL) &&
268  (OptionHandler_AddOption(result, OPTION_TRUSTED_CERT, options->trusted_certs) != OPTIONHANDLER_OK)
269  )
270  {
271  LogError("unable to save TrustedCerts option");
272  OptionHandler_Destroy(result);
273  result = NULL;
274  }
275  else if (options->x509_type != TLSIO_OPTIONS_x509_TYPE_UNSPECIFIED)
276  {
277  const char* x509_cert_option;
278  const char* x509_key_option;
279  if (options->x509_type == TLSIO_OPTIONS_x509_TYPE_ECC)
280  {
281  x509_cert_option = OPTION_X509_ECC_CERT;
282  x509_key_option = OPTION_X509_ECC_KEY;
283  }
284  else
285  {
286  x509_cert_option = SU_OPTION_X509_CERT;
287  x509_key_option = SU_OPTION_X509_PRIVATE_KEY;
288  }
289  if (
290  (options->x509_cert != NULL) &&
291  (OptionHandler_AddOption(result, x509_cert_option, options->x509_cert) != OPTIONHANDLER_OK)
292  )
293  {
294  LogError("unable to save x509 cert option");
295  OptionHandler_Destroy(result);
296  result = NULL;
297  }
298  else if (
299  (options->x509_key != NULL) &&
300  (OptionHandler_AddOption(result, x509_key_option, options->x509_key) != OPTIONHANDLER_OK)
301  )
302  {
303  LogError("unable to save x509 key option");
304  OptionHandler_Destroy(result);
305  result = NULL;
306  }
307  }
308  }
309 
310  return result;
311 }
