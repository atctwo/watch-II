75 bool dns_async_is_lookup_complete(DNS_ASYNC_HANDLE dns_in)
76 {
77  DNS_ASYNC_INSTANCE* dns = (DNS_ASYNC_INSTANCE*)dns_in;
78 
79  bool result;
80  if (dns == NULL)
81  {
82  /* Codes_SRS_DNS_ASYNC_30_020: [ If the dns parameter is NULL, dns_async_is_create_complete shall log an error and return false. ]*/
83  LogError("NULL dns");
84  result = false;
85  }
86  else
87  {
88  if (dns->is_complete)
89  {
90  /* Codes_SRS_DNS_ASYNC_30_024: [ If dns_async_is_create_complete has previously returned true, dns_async_is_create_complete shall do nothing and return true. ]*/
91  result = true;
92  }
93  else
94  {
95  struct addrinfo *addrInfo = NULL;
96  struct addrinfo *ptr = NULL;
97  struct addrinfo hints;
98  int getAddrResult;
99 
100  /* Codes_SRS_DNS_ASYNC_30_021: [ dns_async_is_create_complete shall perform the asynchronous work of DNS lookup and log any errors. ]*/
101  // Only make one attempt at lookup for this
102  // synchronous implementation
103  dns->is_complete = true;
104 
105  //--------------------------------
106  // Setup the hints address info structure
107  // which is passed to the getaddrinfo() function
108  memset(&hints, 0, sizeof(hints));
109  hints.ai_family = AF_INET;
110  hints.ai_socktype = SOCK_STREAM;
111  hints.ai_protocol = IPPROTO_TCP;
112 
113  //--------------------------------
114  // Call getaddrinfo(). If the call succeeds,
115  // the result variable will hold a linked list
116  // of addrinfo structures containing response
117  // information
118  getAddrResult = getaddrinfo(dns->hostname, NULL, &hints, &addrInfo);
119  if (getAddrResult == 0)
120  {
121  // If we find the AF_INET address, use it as the return value
122  for (ptr = addrInfo; ptr != NULL; ptr = ptr->ai_next)
123  {
124  switch (ptr->ai_family)
125  {
126  case AF_INET:
127  /* Codes_SRS_DNS_ASYNC_30_032: [ If dns_async_is_create_complete has returned true and the lookup process has succeeded, dns_async_get_ipv4 shall return the discovered IPv4 address. ]*/
128  dns->ip_v4 = EXTRACT_IPV4(ptr);
129  break;
130  }
131  }
132  /* Codes_SRS_DNS_ASYNC_30_033: [ If dns_async_is_create_complete has returned true and the lookup process has failed, dns_async_get_ipv4 shall return 0. ]*/
133  dns->is_failed = (dns->ip_v4 == 0);
134  freeaddrinfo(addrInfo);
135  }
136  else
137  {
138  /* Codes_SRS_DNS_ASYNC_30_033: [ If dns_async_is_create_complete has returned true and the lookup process has failed, dns_async_get_ipv4 shall return 0. ]*/
139  LogInfo("Failed DNS lookup for %s: %d", dns->hostname, getAddrResult);
140  dns->is_failed = true;
141  }
142  // This synchronous implementation is incapable of being incomplete, so SRS_DNS_ASYNC_30_023 does not ever happen
143  /* Codes_SRS_DNS_ASYNC_30_023: [ If the DNS lookup process is not yet complete, dns_async_is_create_complete shall return false. ]*/
144  /* Codes_SRS_DNS_ASYNC_30_022: [ If the DNS lookup process has completed, dns_async_is_create_complete shall return true. ]*/
145  result = true;
146  }
147  }
148 
149  return result;
