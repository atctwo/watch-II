113  // Algorithm that finds a subsequence of equal values in a sequence. Returns
114  // (iterator,true) if a full match was found, in which case the iterator
115  // points to the beginning of the match. Returns (iterator,false) if a
116  // partial match was found at the end of the first sequence, in which case
117  // the iterator points to the beginning of the partial match. Returns
118  // (last1,false) if no full or partial match was found.
119  template <typename Iterator1, typename Iterator2>
120  std::pair<Iterator1, bool> partial_search(
121  Iterator1 first1, Iterator1 last1, Iterator2 first2, Iterator2 last2)
122  {
123  for (Iterator1 iter1 = first1; iter1 != last1; ++iter1)
124  {
125  Iterator1 test_iter1 = iter1;
126  Iterator2 test_iter2 = first2;
127  for (;; ++test_iter1, ++test_iter2)
128  {
129  if (test_iter2 == last2)
130  return std::make_pair(iter1, true);
131  if (test_iter1 == last1)
132  {
133  if (test_iter2 != first2)
134  return std::make_pair(iter1, false);
135  else
136  break;
137  }
138  if (*test_iter1 != *test_iter2)
139  break;
140  }
141  }
142  return std::make_pair(last1, false);
143  }

467  void operator()(const asio::error_code& ec,
468  std::size_t bytes_transferred, int start = 0)
469  {
470  const std::size_t not_found = (std::numeric_limits<std::size_t>::max)();
471  std::size_t bytes_to_read;
472  switch (start_ = start)
473  {
474  case 1:
475  for (;;)
476  {
477  {
478  // Determine the range of the data to be searched.
479  typedef typename DynamicBuffer::const_buffers_type
480  buffers_type;
481  typedef buffers_iterator<buffers_type> iterator;
482  buffers_type data_buffers = buffers_.data();
483  iterator begin = iterator::begin(data_buffers);
484  iterator start_pos = begin + search_position_;
485  iterator end = iterator::end(data_buffers);
486 
487  // Look for a match.
488  iterator iter = std::find(start_pos, end, delim_);
489  if (iter != end)
490  {
491  // Found a match. We're done.
492  search_position_ = iter - begin + 1;
493  bytes_to_read = 0;
494  }
495 
496  // No match yet. Check if buffer is full.
497  else if (buffers_.size() == buffers_.max_size())
498  {
499  search_position_ = not_found;
500  bytes_to_read = 0;
501  }
502 
503  // Need to read some more data.
504  else
505  {
506  // Next search can start with the new data.
507  search_position_ = end - begin;
508  bytes_to_read = std::min<std::size_t>(
509  std::max<std::size_t>(512,
510  buffers_.capacity() - buffers_.size()),
511  std::min<std::size_t>(65536,
512  buffers_.max_size() - buffers_.size()));
513  }
514  }
515 
516  // Check if we're done.
517  if (!start && bytes_to_read == 0)
518  break;
519 
520  // Start a new asynchronous read operation to obtain more data.
521  stream_.async_read_some(buffers_.prepare(bytes_to_read),
522  ASIO_MOVE_CAST(read_until_delim_op)(*this));
523  return; default:
524  buffers_.commit(bytes_transferred);
525  if (ec || bytes_transferred == 0)
526  break;
527  }
528 
529  const asio::error_code result_ec =
530  (search_position_ == not_found)
531  ? error::not_found : ec;
532 
533  const std::size_t result_n =
534  (ec || search_position_ == not_found)
535  ? 0 : search_position_;
536 
537  handler_(result_ec, result_n);
538  }
539  }
