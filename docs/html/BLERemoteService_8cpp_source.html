161 void BLERemoteService::retrieveCharacteristics() {
162  log_v(">> getCharacteristics() for service: %s", getUUID().toString().c_str());
163 
164  removeCharacteristics(); // Forget any previous characteristics.
165 
166  uint16_t offset = 0;
167  esp_gattc_char_elem_t result;
168  while (true) {
169  uint16_t count = 1; // only room for 1 result allocated, so go one by one
170  esp_gatt_status_t status = ::esp_ble_gattc_get_all_char(
171  getClient()->getGattcIf(),
172  getClient()->getConnId(),
173  m_startHandle,
174  m_endHandle,
175  &result,
176  &count,
177  offset
178  );
179 
180  if (status == ESP_GATT_INVALID_OFFSET) { // We have reached the end of the entries.
181  break;
182  }
183 
184  if (status != ESP_GATT_OK) { // If we got an error, end.
185  log_e("esp_ble_gattc_get_all_char: %s", BLEUtils::gattStatusToString(status).c_str());
186  break;
187  }
188 
189  if (count == 0) { // If we failed to get any new records, end.
190  break;
191  }
192 
193  log_d("Found a characteristic: Handle: %d, UUID: %s", result.char_handle, BLEUUID(result.uuid).toString().c_str());
194 
195  // We now have a new characteristic ... let us add that to our set of known characteristics
196  BLERemoteCharacteristic *pNewRemoteCharacteristic = new BLERemoteCharacteristic(
197  result.char_handle,
198  BLEUUID(result.uuid),
199  result.properties,
200  this
201  );
202 
203  m_characteristicMap.insert(std::pair<std::string, BLERemoteCharacteristic*>(pNewRemoteCharacteristic->getUUID().toString(), pNewRemoteCharacteristic));
204  m_characteristicMapByHandle.insert(std::pair<uint16_t, BLERemoteCharacteristic*>(result.char_handle, pNewRemoteCharacteristic));
205  offset++; // Increment our count of number of descriptors found.
206  } // Loop forever (until we break inside the loop).
207 
208  m_haveCharacteristics = true; // Remember that we have received the characteristics.
209  log_v("<< getCharacteristics()");
210 } // getCharacteristics
