146 void BLERemoteCharacteristic::gattClientEventHandler(esp_gattc_cb_event_t event, esp_gatt_if_t gattc_if, esp_ble_gattc_cb_param_t* evtParam) {
147  switch(event) {
148  // ESP_GATTC_NOTIFY_EVT
149  //
150  // notify
151  // - uint16_t conn_id - The connection identifier of the server.
152  // - esp_bd_addr_t remote_bda - The device address of the BLE server.
153  // - uint16_t handle - The handle of the characteristic for which the event is being received.
154  // - uint16_t value_len - The length of the received data.
155  // - uint8_t* value - The received data.
156  // - bool is_notify - True if this is a notify, false if it is an indicate.
157  //
158  // We have received a notification event which means that the server wishes us to know about a notification
159  // piece of data. What we must now do is find the characteristic with the associated handle and then
160  // invoke its notification callback (if it has one).
161  case ESP_GATTC_NOTIFY_EVT: {
162  if (evtParam->notify.handle != getHandle()) break;
163  if (m_notifyCallback != nullptr) {
164  log_d("Invoking callback for notification on characteristic %s", toString().c_str());
165  m_notifyCallback(this, evtParam->notify.value, evtParam->notify.value_len, evtParam->notify.is_notify);
166  } // End we have a callback function ...
167  break;
168  } // ESP_GATTC_NOTIFY_EVT
169 
170  // ESP_GATTC_READ_CHAR_EVT
171  // This event indicates that the server has responded to the read request.
172  //
173  // read:
174  // - esp_gatt_status_t status
175  // - uint16_t conn_id
176  // - uint16_t handle
177  // - uint8_t* value
178  // - uint16_t value_len
179  case ESP_GATTC_READ_CHAR_EVT: {
180  // If this event is not for us, then nothing further to do.
181  if (evtParam->read.handle != getHandle()) break;
182 
183  // At this point, we have determined that the event is for us, so now we save the value
184  // and unlock the semaphore to ensure that the requestor of the data can continue.
185  if (evtParam->read.status == ESP_GATT_OK) {
186  m_value = std::string((char*) evtParam->read.value, evtParam->read.value_len);
187  if(m_rawData != nullptr) free(m_rawData);
188  m_rawData = (uint8_t*) calloc(evtParam->read.value_len, sizeof(uint8_t));
189  memcpy(m_rawData, evtParam->read.value, evtParam->read.value_len);
190  } else {
191  m_value = "";
192  }
193 
194  m_semaphoreReadCharEvt.give();
195  break;
196  } // ESP_GATTC_READ_CHAR_EVT
197 
198  // ESP_GATTC_REG_FOR_NOTIFY_EVT
199  //
200  // reg_for_notify:
201  // - esp_gatt_status_t status
202  // - uint16_t handle
203  case ESP_GATTC_REG_FOR_NOTIFY_EVT: {
204  // If the request is not for this BLERemoteCharacteristic then move on to the next.
205  if (evtParam->reg_for_notify.handle != getHandle()) break;
206 
207  // We have processed the notify registration and can unlock the semaphore.
208  m_semaphoreRegForNotifyEvt.give();
209  break;
210  } // ESP_GATTC_REG_FOR_NOTIFY_EVT
211 
212  // ESP_GATTC_UNREG_FOR_NOTIFY_EVT
213  //
214  // unreg_for_notify:
215  // - esp_gatt_status_t status
216  // - uint16_t handle
217  case ESP_GATTC_UNREG_FOR_NOTIFY_EVT: {
218  if (evtParam->unreg_for_notify.handle != getHandle()) break;
219  // We have processed the notify un-registration and can unlock the semaphore.
220  m_semaphoreRegForNotifyEvt.give();
221  break;
222  } // ESP_GATTC_UNREG_FOR_NOTIFY_EVT:
223 
224  // ESP_GATTC_WRITE_CHAR_EVT
225  //
226  // write:
227  // - esp_gatt_status_t status
228  // - uint16_t conn_id
229  // - uint16_t handle
230  case ESP_GATTC_WRITE_CHAR_EVT: {
231  // Determine if this event is for us and, if not, pass onwards.
232  if (evtParam->write.handle != getHandle()) break;
233 
234  // There is nothing further we need to do here. This is merely an indication
235  // that the write has completed and we can unlock the caller.
236  m_semaphoreWriteCharEvt.give();
237  break;
238  } // ESP_GATTC_WRITE_CHAR_EVT
239 
240 
241  default:
242  break;
243  } // End switch
244 }; // gattClientEventHandler

250 void BLERemoteCharacteristic::retrieveDescriptors() {
251  log_v(">> retrieveDescriptors() for characteristic: %s", getUUID().toString().c_str());
252 
253  removeDescriptors(); // Remove any existing descriptors.
254 
255  // Loop over each of the descriptors within the service associated with this characteristic.
256  // For each descriptor we find, create a BLERemoteDescriptor instance.
257  uint16_t offset = 0;
258  esp_gattc_descr_elem_t result;
259  while(true) {
260  uint16_t count = 10;
261  esp_gatt_status_t status = ::esp_ble_gattc_get_all_descr(
262  getRemoteService()->getClient()->getGattcIf(),
263  getRemoteService()->getClient()->getConnId(),
264  getHandle(),
265  &result,
266  &count,
267  offset
268  );
269 
270  if (status == ESP_GATT_INVALID_OFFSET) { // We have reached the end of the entries.
271  break;
272  }
273 
274  if (status != ESP_GATT_OK) {
275  log_e("esp_ble_gattc_get_all_descr: %s", BLEUtils::gattStatusToString(status).c_str());
276  break;
277  }
278 
279  if (count == 0) break;
280 
281  log_d("Found a descriptor: Handle: %d, UUID: %s", result.handle, BLEUUID(result.uuid).toString().c_str());
282 
283  // We now have a new characteristic ... let us add that to our set of known characteristics
284  BLERemoteDescriptor* pNewRemoteDescriptor = new BLERemoteDescriptor(
285  result.handle,
286  BLEUUID(result.uuid),
287  this
288  );
289 
290  m_descriptorMap.insert(std::pair<std::string, BLERemoteDescriptor*>(pNewRemoteDescriptor->getUUID().toString(), pNewRemoteDescriptor));
291 
292  offset++;
293  } // while true
294  //m_haveCharacteristics = true; // Remember that we have received the characteristics.
295  log_v("<< retrieveDescriptors(): Found %d descriptors.", offset);
296 } // getDescriptors

409 std::string BLERemoteCharacteristic::readValue() {
410  log_v(">> readValue(): uuid: %s, handle: %d 0x%.2x", getUUID().toString().c_str(), getHandle(), getHandle());
411 
412  // Check to see that we are connected.
413  if (!getRemoteService()->getClient()->isConnected()) {
414  log_e("Disconnected");
415  return std::string();
416  }
417 
418  m_semaphoreReadCharEvt.take("readValue");
419 
420  // Ask the BLE subsystem to retrieve the value for the remote hosted characteristic.
421  // This is an asynchronous request which means that we must block waiting for the response
422  // to become available.
423  esp_err_t errRc = ::esp_ble_gattc_read_char(
424  m_pRemoteService->getClient()->getGattcIf(),
425  m_pRemoteService->getClient()->getConnId(), // The connection ID to the BLE server
426  getHandle(), // The handle of this characteristic
427  m_auth); // Security
428 
429  if (errRc != ESP_OK) {
430  log_e("esp_ble_gattc_read_char: rc=%d %s", errRc, GeneralUtils::errorToString(errRc));
431  return "";
432  }
433 
434  // Block waiting for the event that indicates that the read has completed. When it has, the std::string found
435  // in m_value will contain our data.
436  m_semaphoreReadCharEvt.wait("readValue");
437 
438  log_v("<< readValue(): length: %d", m_value.length());
439  return m_value;
440 } // readValue

450  log_v(">> registerForNotify(): %s", toString().c_str());
451 
452  m_notifyCallback = notifyCallback; // Save the notification callback.
453

455 
456  if (notifyCallback != nullptr) { // If we have a callback function, then this is a registration.
457  esp_err_t errRc = ::esp_ble_gattc_register_for_notify(
458  m_pRemoteService->getClient()->getGattcIf(),
459  *m_pRemoteService->getClient()->getPeerAddress().getNative(),
460  getHandle()
461  );
462 
463  if (errRc != ESP_OK) {
464  log_e("esp_ble_gattc_register_for_notify: rc=%d %s", errRc, GeneralUtils::errorToString(errRc));
465  }
466 
467  uint8_t val[] = {0x01, 0x00};
468  if(!notifications) val[0] = 0x02;
469  BLERemoteDescriptor* desc = getDescriptor(BLEUUID((uint16_t)0x2902));
470  desc->writeValue(val, 2);
471  } // End Register
