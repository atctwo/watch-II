34 bool btStart(){
35  esp_bt_controller_config_t cfg = BT_CONTROLLER_INIT_CONFIG_DEFAULT();
36  if(esp_bt_controller_get_status() == ESP_BT_CONTROLLER_STATUS_ENABLED){
37  return true;
38  }
39  if(esp_bt_controller_get_status() == ESP_BT_CONTROLLER_STATUS_IDLE){
40  esp_bt_controller_init(&cfg);
41  while(esp_bt_controller_get_status() == ESP_BT_CONTROLLER_STATUS_IDLE){}
42  }
43  if(esp_bt_controller_get_status() == ESP_BT_CONTROLLER_STATUS_INITED){
44  if (esp_bt_controller_enable(BT_MODE)) {
45  log_e("BT Enable failed");
46  return false;
47  }
48  }

50  return true;
51  }
52  log_e("BT Start failed");
53  return false;
54 }

56 bool btStop(){
57  if(esp_bt_controller_get_status() == ESP_BT_CONTROLLER_STATUS_IDLE){
58  return true;
59  }
60  if(esp_bt_controller_get_status() == ESP_BT_CONTROLLER_STATUS_ENABLED){
61  if (esp_bt_controller_disable()) {
62  log_e("BT Disable failed");

64  }
65  while(esp_bt_controller_get_status() == ESP_BT_CONTROLLER_STATUS_ENABLED);
66  }
67  if(esp_bt_controller_get_status() == ESP_BT_CONTROLLER_STATUS_INITED){
68  if (esp_bt_controller_deinit()) {
69  log_e("BT deint failed");
70  return false;
71  }
72  vTaskDelay(1);

74  return false;
