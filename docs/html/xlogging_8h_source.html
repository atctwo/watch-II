82 /*
83 The ESP8266 compiler doesn't do a good job compiling this code; it doesn't understand that the 'format' is
84 a 'const char*' and moves it to RAM as a global variable, increasing the .bss size. So we create a
85 specific LogInfo that explicitly pins the 'format' on the PROGMEM (flash) using a _localFORMAT variable
86 with the macro PSTR.
87 #define ICACHE_FLASH_ATTR __attribute__((section(".irom0.text")))
88 #define PROGMEM ICACHE_RODATA_ATTR
89 #define PSTR(s) (__extension__({static const char __c[] PROGMEM = (s); &__c[0];}))
90 const char* __localFORMAT = PSTR(FORMAT);
91 On the other hand, vsprintf does not support the pinned 'format' and os_printf does not work with va_list,
92 so we compacted the log in the macro LogInfo.
93 */
94 #define LOG(log_category, log_options, FORMAT, ...) { \
