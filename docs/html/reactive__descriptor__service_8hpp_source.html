167  // Wait for the descriptor to become ready to read, ready to write, or to have
168  // pending error conditions.
169  asio::error_code wait(implementation_type& impl,
170  posix::descriptor_base::wait_type w, asio::error_code& ec)
171  {
172  switch (w)

174  case posix::descriptor_base::wait_read:
175  descriptor_ops::poll_read(impl.descriptor_, impl.state_, ec);
176  break;
177  case posix::descriptor_base::wait_write:

179  break;
180  case posix::descriptor_base::wait_error:
181  descriptor_ops::poll_error(impl.descriptor_, impl.state_, ec);
182  break;
183  default:
184  ec = asio::error::invalid_argument;
185  break;
186  }
187 
188  return ec;

191  // Asynchronously wait for the descriptor to become ready to read, ready to
192  // write, or to have pending error conditions.
193  template <typename Handler>
194  void async_wait(implementation_type& impl,
195  posix::descriptor_base::wait_type w, Handler& handler)
196  {
197  bool is_continuation =
198  asio_handler_cont_helpers::is_continuation(handler);
199 
200  // Allocate and construct an operation to wrap the handler.
201  typedef reactive_wait_op<Handler> op;
202  typename op::ptr p = { asio::detail::addressof(handler),
203  op::ptr::allocate(handler), 0 };
204  p.p = new (p.v) op(handler);
205 
206  ASIO_HANDLER_CREATION((reactor_.context(), *p.p, "descriptor",
207  &impl, impl.descriptor_, "async_wait"));
208 
209  int op_type;
210  switch (w)
211  {
212  case posix::descriptor_base::wait_read:
213  op_type = reactor::read_op;
214  break;
215  case posix::descriptor_base::wait_write:
216  op_type = reactor::write_op;
217  break;
218  case posix::descriptor_base::wait_error:
219  op_type = reactor::except_op;
220  break;
221  default:
222  p.p->ec_ = asio::error::invalid_argument;
223  reactor_.post_immediate_completion(p.p, is_continuation);
224  p.v = p.p = 0;
225  return;
226  }
227 
228  start_op(impl, op_type, p.p, is_continuation, false, false);
229  p.v = p.p = 0;
230  }
