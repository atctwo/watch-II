32 void IRAM_ATTR __touchISR(void * arg)
33 {
34  uint32_t pad_intr = READ_PERI_REG(SENS_SAR_TOUCH_CTRL2_REG) & 0x3ff;
35  uint32_t rtc_intr = READ_PERI_REG(RTC_CNTL_INT_ST_REG);
36  uint8_t i = 0;
37  //clear interrupt
38  WRITE_PERI_REG(RTC_CNTL_INT_CLR_REG, rtc_intr);
39  SET_PERI_REG_MASK(SENS_SAR_TOUCH_CTRL2_REG, SENS_TOUCH_MEAS_EN_CLR);
40 
41  if (rtc_intr & RTC_CNTL_TOUCH_INT_ST) {
42  for (i = 0; i < 10; ++i) {
43  if ((pad_intr >> i) & 0x01) {
44  if(__touchInterruptHandlers[i]){
45  __touchInterruptHandlers[i]();
46  }
47  }
48  }
49  }
50 }

52 void __touchSetCycles(uint16_t measure, uint16_t sleep)
53 {
54  __touchSleepCycles = sleep;
55  __touchMeasureCycles = measure;
56  //Touch pad SleepCycle Time
57  SET_PERI_REG_BITS(SENS_SAR_TOUCH_CTRL2_REG, SENS_TOUCH_SLEEP_CYCLES, __touchSleepCycles, SENS_TOUCH_SLEEP_CYCLES_S);
58  //Touch Pad Measure Time
59  SET_PERI_REG_BITS(SENS_SAR_TOUCH_CTRL1_REG, SENS_TOUCH_MEAS_DELAY, __touchMeasureCycles, SENS_TOUCH_MEAS_DELAY_S);
60 }

62 void __touchInit()
63 {
64  static bool initialized = false;
65  if(initialized){
66  return;
67  }
68  initialized = true;
69  SET_PERI_REG_BITS(RTC_IO_TOUCH_CFG_REG, RTC_IO_TOUCH_XPD_BIAS, 1, RTC_IO_TOUCH_XPD_BIAS_S);
70  SET_PERI_REG_MASK(SENS_SAR_TOUCH_CTRL2_REG, SENS_TOUCH_MEAS_EN_CLR);
71  //clear touch enable
72  WRITE_PERI_REG(SENS_SAR_TOUCH_ENABLE_REG, 0x0);
73  SET_PERI_REG_MASK(RTC_CNTL_STATE0_REG, RTC_CNTL_TOUCH_SLP_TIMER_EN);
74 
75  __touchSetCycles(__touchMeasureCycles, __touchSleepCycles);
76 
77  esp_intr_alloc(ETS_RTC_CORE_INTR_SOURCE, (int)ESP_INTR_FLAG_IRAM, __touchISR, NULL, &touch_intr_handle);
78 }

80 uint16_t __touchRead(uint8_t pin)
81 {
82  int8_t pad = digitalPinToTouchChannel(pin);
83  if(pad < 0){
84  return 0;
85  }
86 
87  pinMode(pin, ANALOG);
88 
89  __touchInit();
90 
91  uint32_t v0 = READ_PERI_REG(SENS_SAR_TOUCH_ENABLE_REG);
92  //Disable Intr & enable touch pad
93  WRITE_PERI_REG(SENS_SAR_TOUCH_ENABLE_REG,
94  (v0 & ~((1 << (pad + SENS_TOUCH_PAD_OUTEN2_S)) | (1 << (pad + SENS_TOUCH_PAD_OUTEN1_S))))
95  | (1 << (pad + SENS_TOUCH_PAD_WORKEN_S)));
96 
97  SET_PERI_REG_MASK(SENS_SAR_TOUCH_ENABLE_REG, (1 << (pad + SENS_TOUCH_PAD_WORKEN_S)));
98 
99  uint32_t rtc_tio_reg = RTC_IO_TOUCH_PAD0_REG + pad * 4;
100  WRITE_PERI_REG(rtc_tio_reg, (READ_PERI_REG(rtc_tio_reg)
101  & ~(RTC_IO_TOUCH_PAD0_DAC_M))
102  | (7 << RTC_IO_TOUCH_PAD0_DAC_S)//Touch Set Slope
103  | RTC_IO_TOUCH_PAD0_TIE_OPT_M //Enable Tie,Init Level
104  | RTC_IO_TOUCH_PAD0_START_M //Enable Touch Pad IO
105  | RTC_IO_TOUCH_PAD0_XPD_M); //Enable Touch Pad Power on
106 
107  //force oneTime test start
108  SET_PERI_REG_MASK(SENS_SAR_TOUCH_CTRL2_REG, SENS_TOUCH_START_EN_M|SENS_TOUCH_START_FORCE_M);
109 
110  SET_PERI_REG_BITS(SENS_SAR_TOUCH_CTRL1_REG, SENS_TOUCH_XPD_WAIT, 10, SENS_TOUCH_XPD_WAIT_S);
111 
112  while (GET_PERI_REG_MASK(SENS_SAR_TOUCH_CTRL2_REG, SENS_TOUCH_MEAS_DONE) == 0) {};
113 
114  uint16_t touch_value = READ_PERI_REG(SENS_SAR_TOUCH_OUT1_REG + (pad / 2) * 4) >> ((pad & 1) ? SENS_TOUCH_MEAS_OUT1_S : SENS_TOUCH_MEAS_OUT0_S);
115 
116  //clear touch force ,select the Touch mode is Timer
117  CLEAR_PERI_REG_MASK(SENS_SAR_TOUCH_CTRL2_REG, SENS_TOUCH_START_EN_M|SENS_TOUCH_START_FORCE_M);
118 
119  //restore previous value
120  WRITE_PERI_REG(SENS_SAR_TOUCH_ENABLE_REG, v0);
121  return touch_value;
122 }

124 void __touchAttachInterrupt(uint8_t pin, void (*userFunc)(void), uint16_t threshold)
125 {
126  int8_t pad = digitalPinToTouchChannel(pin);
127  if(pad < 0){
128  return;
129  }
130 
131  pinMode(pin, ANALOG);
132 
133  __touchInit();
134 
135  __touchInterruptHandlers[pad] = userFunc;
136 
137  //clear touch force ,select the Touch mode is Timer
138  CLEAR_PERI_REG_MASK(SENS_SAR_TOUCH_CTRL2_REG, SENS_TOUCH_START_EN_M|SENS_TOUCH_START_FORCE_M);
139 
140  //interrupt when touch value < threshold
141  CLEAR_PERI_REG_MASK(SENS_SAR_TOUCH_CTRL1_REG, SENS_TOUCH_OUT_SEL);
142  //Intr will give ,when SET0 < threshold
143  SET_PERI_REG_MASK(SENS_SAR_TOUCH_CTRL1_REG, SENS_TOUCH_OUT_1EN);
144  //Enable Rtc Touch Module Intr,the Interrupt need Rtc out Enable
145  SET_PERI_REG_MASK(RTC_CNTL_INT_ENA_REG, RTC_CNTL_TOUCH_INT_ENA);
146 
147  //set threshold
148  uint8_t shift = (pad & 1) ? SENS_TOUCH_OUT_TH1_S : SENS_TOUCH_OUT_TH0_S;
149  SET_PERI_REG_BITS((SENS_SAR_TOUCH_THRES1_REG + (pad / 2) * 4), SENS_TOUCH_OUT_TH0, threshold, shift);
150 
151  uint32_t rtc_tio_reg = RTC_IO_TOUCH_PAD0_REG + pad * 4;
152  WRITE_PERI_REG(rtc_tio_reg, (READ_PERI_REG(rtc_tio_reg)
153  & ~(RTC_IO_TOUCH_PAD0_DAC_M))
154  | (7 << RTC_IO_TOUCH_PAD0_DAC_S)//Touch Set Slope
155  | RTC_IO_TOUCH_PAD0_TIE_OPT_M //Enable Tie,Init Level
156  | RTC_IO_TOUCH_PAD0_START_M //Enable Touch Pad IO
157  | RTC_IO_TOUCH_PAD0_XPD_M); //Enable Touch Pad Power on
158 
159  //Enable Digital rtc control :work mode and out mode
160  SET_PERI_REG_MASK(SENS_SAR_TOUCH_ENABLE_REG,
161  (1 << (pad + SENS_TOUCH_PAD_WORKEN_S)) | \
162  (1 << (pad + SENS_TOUCH_PAD_OUTEN2_S)) | \
163  (1 << (pad + SENS_TOUCH_PAD_OUTEN1_S)));
164 }
