53 static void triggerApbChangeCallback(apb_change_ev_t ev_type, uint32_t old_apb, uint32_t new_apb){
54  initApbChangeCallback();
55  xSemaphoreTake(apb_change_lock, portMAX_DELAY);
56  apb_change_t * r = apb_change_callbacks;
57  if( r != NULL ){
58  if(ev_type == APB_BEFORE_CHANGE )
59  while(r != NULL){
60  r->cb(r->arg, ev_type, old_apb, new_apb);
61  r=r->next;
62  }
63  else { // run backwards through chain
64  while(r->next != NULL) r = r->next; // find first added
65  while( r != NULL){
66  r->cb(r->arg, ev_type, old_apb, new_apb);
67  r=r->prev;
68  }
69  }
70  }
71  xSemaphoreGive(apb_change_lock);
72 }

74 bool addApbChangeCallback(void * arg, apb_change_cb_t cb){
75  initApbChangeCallback();
76  apb_change_t * c = (apb_change_t*)malloc(sizeof(apb_change_t));
77  if(!c){
78  log_e("Callback Object Malloc Failed");
79  return false;
80  }
81  c->next = NULL;
82  c->prev = NULL;
83  c->arg = arg;
84  c->cb = cb;
85  xSemaphoreTake(apb_change_lock, portMAX_DELAY);
86  if(apb_change_callbacks == NULL){
87  apb_change_callbacks = c;
88  } else {
89  apb_change_t * r = apb_change_callbacks;
90  // look for duplicate callbacks
91  while( (r != NULL ) && !((r->cb == cb) && ( r->arg == arg))) r = r->next;
92  if (r) {
93  log_e("duplicate func=%08X arg=%08X",c->cb,c->arg);
94  free(c);
95  xSemaphoreGive(apb_change_lock);
96  return false;
97  }
98  else {
99  c->next = apb_change_callbacks;
100  apb_change_callbacks-> prev = c;
101  apb_change_callbacks = c;
102  }
103  }
104  xSemaphoreGive(apb_change_lock);
105  return true;
106 }

108 bool removeApbChangeCallback(void * arg, apb_change_cb_t cb){
109  initApbChangeCallback();
110  xSemaphoreTake(apb_change_lock, portMAX_DELAY);
111  apb_change_t * r = apb_change_callbacks;
112  // look for matching callback
113  while( (r != NULL ) && !((r->cb == cb) && ( r->arg == arg))) r = r->next;
114  if ( r == NULL ) {
115  log_e("not found func=%08X arg=%08X",cb,arg);
116  xSemaphoreGive(apb_change_lock);
117  return false;
118  }
119  else {
120  // patch links
121  if(r->prev) r->prev->next = r->next;
122  else { // this is first link
123  apb_change_callbacks = r->next;
124  }
125  if(r->next) r->next->prev = r->prev;
126  free(r);
127  }
128  xSemaphoreGive(apb_change_lock);
129  return true;
130 }

141 bool setCpuFrequencyMhz(uint32_t cpu_freq_mhz){
142  rtc_cpu_freq_config_t conf, cconf;
143  uint32_t capb, apb;
144  //Get XTAL Frequency and calculate min CPU MHz
145  rtc_xtal_freq_t xtal = rtc_clk_xtal_freq_get();
146  if(xtal > RTC_XTAL_FREQ_AUTO){
147  if(xtal < RTC_XTAL_FREQ_40M) {
148  if(cpu_freq_mhz <= xtal && cpu_freq_mhz != xtal && cpu_freq_mhz != (xtal/2)){
149  log_e("Bad frequency: %u MHz! Options are: 240, 160, 80, %u and %u MHz", cpu_freq_mhz, xtal, xtal/2);
150  return false;
151  }
152  } else if(cpu_freq_mhz <= xtal && cpu_freq_mhz != xtal && cpu_freq_mhz != (xtal/2) && cpu_freq_mhz != (xtal/4)){
153  log_e("Bad frequency: %u MHz! Options are: 240, 160, 80, %u, %u and %u MHz", cpu_freq_mhz, xtal, xtal/2, xtal/4);
154  return false;
155  }
156  }
157  if(cpu_freq_mhz > xtal && cpu_freq_mhz != 240 && cpu_freq_mhz != 160 && cpu_freq_mhz != 80){
158  if(xtal >= RTC_XTAL_FREQ_40M){
159  log_e("Bad frequency: %u MHz! Options are: 240, 160, 80, %u, %u and %u MHz", cpu_freq_mhz, xtal, xtal/2, xtal/4);
160  } else {
161  log_e("Bad frequency: %u MHz! Options are: 240, 160, 80, %u and %u MHz", cpu_freq_mhz, xtal, xtal/2);
162  }
163  return false;
164  }
165  //check if cpu supports the frequency
166  if(cpu_freq_mhz == 240){
167  //Check if ESP32 is rated for a CPU frequency of 160MHz only
168  if (REG_GET_BIT(EFUSE_BLK0_RDATA3_REG, EFUSE_RD_CHIP_CPU_FREQ_RATED) &&
169  REG_GET_BIT(EFUSE_BLK0_RDATA3_REG, EFUSE_RD_CHIP_CPU_FREQ_LOW)) {
170  log_e("Can not switch to 240 MHz! Chip CPU frequency rated for 160MHz.");
171  cpu_freq_mhz = 160;
172  }
173  }
174  //Get current CPU clock configuration
175  rtc_clk_cpu_freq_get_config(&cconf);
176  //return if frequency has not changed
177  if(cconf.freq_mhz == cpu_freq_mhz){
178  return true;
179  }
180  //Get configuration for the new CPU frequency
181  if(!rtc_clk_cpu_freq_mhz_to_config(cpu_freq_mhz, &conf)){
182  log_e("CPU clock could not be set to %u MHz", cpu_freq_mhz);
183  return false;
184  }
185  //Current APB
186  capb = calculateApb(&cconf);
187  //New APB
188  apb = calculateApb(&conf);
189  log_d("%s: %u / %u = %u Mhz, APB: %u Hz", (conf.source == RTC_CPU_FREQ_SRC_PLL)?"PLL":((conf.source == RTC_CPU_FREQ_SRC_APLL)?"APLL":((conf.source == RTC_CPU_FREQ_SRC_XTAL)?"XTAL":"8M")), conf.source_freq_mhz, conf.div, conf.freq_mhz, apb);
190  //Call peripheral functions before the APB change
191  if(apb_change_callbacks){
192  triggerApbChangeCallback(APB_BEFORE_CHANGE, capb, apb);
193  }
194  //Make the frequency change
195  rtc_clk_cpu_freq_set_config_fast(&conf);
196  if(capb != apb){
197  //Update REF_TICK (uncomment if REF_TICK is different than 1MHz)
198  //if(conf.freq_mhz < 80){
199  // ESP_REG(APB_CTRL_XTAL_TICK_CONF_REG) = conf.freq_mhz / (REF_CLK_FREQ / MHZ) - 1;
200  // }
201  //Update APB Freq REG
202  rtc_clk_apb_freq_update(apb);
203  //Update esp_timer divisor
204  esp_timer_impl_update_apb_freq(apb / MHZ);
205  }
206  //Update FreeRTOS Tick Divisor
207  uint32_t fcpu = (conf.freq_mhz >= 80)?(conf.freq_mhz * MHZ):(apb);
208  _xt_tick_divisor = fcpu / XT_TICK_PER_SEC;
209  //Call peripheral functions after the APB change
210  if(apb_change_callbacks){
211  triggerApbChangeCallback(APB_AFTER_CHANGE, capb, apb);
212  }
213  return true;
214 }
