36 class executor::function
37 {
38 public:
39  template <typename F, typename Alloc>
40  explicit function(F f, const Alloc& a)
41  {
42  // Allocate and construct an operation to wrap the function.
43  typedef detail::executor_op<F, Alloc> op;
44  typename op::ptr p = { detail::addressof(a), op::ptr::allocate(a), 0 };
45  op_ = new (p.v) op(ASIO_MOVE_CAST(F)(f), a);
46  p.v = 0;
47  }
48 
49  function(function&& other)
50  : op_(other.op_)
51  {
52  other.op_ = 0;
53  }
54
