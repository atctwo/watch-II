202 // An executor that adapts the system_executor to capture any exeption thrown
203 // by a submitted function object and save it into a promise.
204 template <typename T>
205 class promise_executor
206 {
207 public:
208  explicit promise_executor(const shared_ptr<std::promise<T> >& p)
209  : p_(p)
210  {
211  }
212 
213  execution_context& context() const ASIO_NOEXCEPT
214  {
215  return system_executor().context();
216  }
217 
218  void on_work_started() const ASIO_NOEXCEPT {}
219  void on_work_finished() const ASIO_NOEXCEPT {}
220 
221  template <typename F, typename A>
222  void dispatch(ASIO_MOVE_ARG(F) f, const A&) const
223  {
224  promise_invoker<T, F>(p_, ASIO_MOVE_CAST(F)(f))();
225  }
226 
227  template <typename F, typename A>
228  void post(ASIO_MOVE_ARG(F) f, const A& a) const
229  {
230  system_executor().post(
231  promise_invoker<T, F>(p_, ASIO_MOVE_CAST(F)(f)), a);
232  }
233 
234  template <typename F, typename A>
235  void defer(ASIO_MOVE_ARG(F) f, const A& a) const
236  {
237  system_executor().defer(
238  promise_invoker<T, F>(p_, ASIO_MOVE_CAST(F)(f)), a);
239  }
240 
241  friend bool operator==(const promise_executor& a,
242  const promise_executor& b) ASIO_NOEXCEPT
243  {
244  return a.p_ == b.p_;
245  }
246 
247  friend bool operator!=(const promise_executor& a,
248  const promise_executor& b) ASIO_NOEXCEPT
249  {
250  return a.p_ != b.p_;
251  }
252 
253 private:
254  shared_ptr<std::promise<T> > p_;
255 };

257 // The base class for all completion handlers that create promises.
258 template <typename T>
259 class promise_creator
260 {

265  {
266  return executor_type(p_);

271  future_type get_future()

278  void create_promise(const Allocator& a)
