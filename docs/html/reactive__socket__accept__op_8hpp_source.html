105  static void do_complete(void* owner, operation* base,
106  const asio::error_code& /*ec*/,
107  std::size_t /*bytes_transferred*/)
108  {
109  // Take ownership of the handler object.
110  reactive_socket_accept_op* o(static_cast<reactive_socket_accept_op*>(base));
111  ptr p = { asio::detail::addressof(o->handler_), o, o };
112  handler_work<Handler> w(o->handler_);
113 
114  // On success, assign new connection to peer socket object.
115  if (owner)
116  o->do_assign();
117 
118  ASIO_HANDLER_COMPLETION((*o));
119 
120  // Make a copy of the handler so that the memory can be deallocated before
121  // the upcall is made. Even if we're not about to make an upcall, a
122  // sub-object of the handler may be the true owner of the memory associated
123  // with the handler. Consequently, a local copy of the handler is required
124  // to ensure that any owning sub-object remains valid until after we have
125  // deallocated the memory here.
126  detail::binder1<Handler, asio::error_code>
127  handler(o->handler_, o->ec_);
128  p.h = asio::detail::addressof(handler.handler_);
129  p.reset();
130 
131  // Make the upcall if required.
132  if (owner)
133  {
134  fenced_block b(fenced_block::half);
135  ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_));
136  w.complete(handler, handler.handler_);
137  ASIO_HANDLER_INVOCATION_END;
138  }
139  }
140 
141 private:
142  Handler handler_;
143 };
