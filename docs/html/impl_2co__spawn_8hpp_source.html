690  executor_work_guard<Executor> work_guard, F f, Handler handler)
691 {

693

741 template <typename Executor, typename F, typename CompletionToken>
742 auto co_spawn(const Executor& ex, F&& f, CompletionToken&& token)
743 {
744  typedef typename result_of<F()>::type awaitable_type;
745  typedef typename awaitable_type::executor_type executor_type;
746  typedef typename awaitable_signature<awaitable_type>::type signature_type;
747 
748  async_completion<CompletionToken, signature_type> completion(token);
749 
750  executor_type ex2(ex);
751  auto work_guard = make_work_guard(completion.completion_handler, ex2);
752 
753  auto* a = (co_spawn_entry_point)(
754  static_cast<awaitable_type*>(nullptr), std::move(work_guard),
755  std::forward<F>(f), std::move(completion.completion_handler));
756 
757  a->set_executor(ex2);
758  (post)(co_spawn_handler<executor_type>(a));
