98 template <typename Executor, typename Function, typename Allocator>
99 void strand_executor_service::dispatch(const implementation_type& impl,
100  Executor& ex, ASIO_MOVE_ARG(Function) function, const Allocator& a)
101 {
102  typedef typename decay<Function>::type function_type;
103 
104  // If we are already in the strand then the function can run immediately.
105  if (call_stack<strand_impl>::contains(impl.get()))
106  {
107  // Make a local, non-const copy of the function.
108  function_type tmp(ASIO_MOVE_CAST(Function)(function));
109 
110  fenced_block b(fenced_block::full);
111  asio_handler_invoke_helpers::invoke(tmp, tmp);
112  return;
113  }
114 
115  // Allocate and construct an operation to wrap the function.
116  typedef executor_op<function_type, Allocator> op;
117  typename op::ptr p = { detail::addressof(a), op::ptr::allocate(a), 0 };
118  p.p = new (p.v) op(ASIO_MOVE_CAST(Function)(function), a);
119 
120  ASIO_HANDLER_CREATION((impl->service_->context(), *p.p,
121  "strand_executor", impl.get(), 0, "dispatch"));
122 
123  // Add the function to the strand and schedule the strand if required.
124  bool first = enqueue(impl, p.p);
125  p.v = p.p = 0;
126  if (first)
127  ex.dispatch(invoker<Executor>(impl, ex), a);
128 }
