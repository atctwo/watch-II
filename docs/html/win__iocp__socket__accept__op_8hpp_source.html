75  static void do_complete(void* owner, operation* base,
76  const asio::error_code& result_ec,
77  std::size_t /*bytes_transferred*/)
78  {
79  asio::error_code ec(result_ec);
80 
81  // Take ownership of the operation object.
82  win_iocp_socket_accept_op* o(static_cast<win_iocp_socket_accept_op*>(base));
83  ptr p = { asio::detail::addressof(o->handler_), o, o };
84  handler_work<Handler> w(o->handler_);
85 
86  if (owner)
87  {
88  typename Protocol::endpoint peer_endpoint;
89  std::size_t addr_len = peer_endpoint.capacity();
90  socket_ops::complete_iocp_accept(o->socket_,
91  o->output_buffer(), o->address_length(),
92  peer_endpoint.data(), &addr_len,
93  o->new_socket_.get(), ec);
94 
95  // Restart the accept operation if we got the connection_aborted error
96  // and the enable_connection_aborted socket option is not set.
97  if (ec == asio::error::connection_aborted
98  && !o->enable_connection_aborted_)
99  {
100  o->reset();
101  o->socket_service_.restart_accept_op(o->socket_,
102  o->new_socket_, o->protocol_.family(),
103  o->protocol_.type(), o->protocol_.protocol(),
104  o->output_buffer(), o->address_length(), o);
105  p.v = p.p = 0;
106  return;
107  }
108 
109  // If the socket was successfully accepted, transfer ownership of the
110  // socket to the peer object.
111  if (!ec)
112  {
113  o->peer_.assign(o->protocol_,
114  typename Socket::native_handle_type(
115  o->new_socket_.get(), peer_endpoint), ec);
116  if (!ec)
117  o->new_socket_.release();
118  }
119 
120  // Pass endpoint back to caller.
121  if (o->peer_endpoint_)
122  *o->peer_endpoint_ = peer_endpoint;
123  }
124 
125  ASIO_HANDLER_COMPLETION((*o));
126 
127  // Make a copy of the handler so that the memory can be deallocated before
128  // the upcall is made. Even if we're not about to make an upcall, a
129  // sub-object of the handler may be the true owner of the memory associated
130  // with the handler. Consequently, a local copy of the handler is required
131  // to ensure that any owning sub-object remains valid until after we have
132  // deallocated the memory here.
133  detail::binder1<Handler, asio::error_code>
134  handler(o->handler_, ec);
135  p.h = asio::detail::addressof(handler.handler_);
136  p.reset();
137 
138  // Make the upcall if required.
139  if (owner)
140  {
141  fenced_block b(fenced_block::half);
142  ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_));
143  w.complete(handler, handler.handler_);
144  ASIO_HANDLER_INVOCATION_END;
145  }
146  }
