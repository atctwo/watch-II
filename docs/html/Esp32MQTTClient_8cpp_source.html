51 static char *GetHostNameFromConnectionString(char *connectionString)
52 {
53  if (connectionString == NULL)
54  {
55  return NULL;
56  }
57  int start = 0;
58  int cur = 0;
59  bool find = false;
60  while (connectionString[cur] > 0)
61  {
62  if (connectionString[cur] == '=')
63  {
64  // Check the key
65  if (memcmp(&connectionString[start], "HostName", 8) == 0)
66  {
67  // This is the host name
68  find = true;
69  }
70  start = ++cur;
71  // Value
72  while (connectionString[cur] > 0)
73  {
74  if (connectionString[cur] == ';')
75  {
76  break;
77  }
78  cur++;
79  }
80  if (find && cur - start > 0)
81  {
82  char *hostname = (char *)malloc(cur - start + 1);
83  memcpy(hostname, &connectionString[start], cur - start);
84  hostname[cur - start] = 0;
85  return hostname;
86  }
87  start = cur + 1;
88  }
89  cur++;
90  }
91  return NULL;
92 }

177 static IOTHUBMESSAGE_DISPOSITION_RESULT ReceiveMessageCallback(IOTHUB_MESSAGE_HANDLE message, void *userContextCallback)

206

208  (*counter)++;

260 static bool SendEventOnce(EVENT_INSTANCE *event)
261 {
262  if (event == NULL)
263  {
264  return false;
265  }
266 
267  if (iotHubClientHandle == NULL)
268  {
269  FreeEventInstance(event);
270  return false;
271  }
272 
273  event->trackingId = trackingId++;
274  currentTrackingId = event->trackingId;
275 
276  uint64_t start_ms = millis();
277 
278  CheckConnection();
279 
280  if (event->type == MESSAGE)
281  {
282  if (IoTHubClient_LL_SendEventAsync(iotHubClientHandle, event->messageHandle, SendConfirmationCallback, event) != IOTHUB_CLIENT_OK)
283  {
284  LogError("IoTHubClient_LL_SendEventAsync..........FAILED!");
285  FreeEventInstance(event);
286  return false;
287  }
288  LogInfo(">>>IoTHubClient_LL_SendEventAsync accepted message for transmission to IoT Hub.");
289  }
290  else if (event->type == STATE)
291  {
292  if (IoTHubClient_LL_SendReportedState(iotHubClientHandle, (const unsigned char *)event->stateString, strlen(event->stateString), ReportConfirmationCallback, event) != IOTHUB_CLIENT_OK)
293  {
294  LogError("IoTHubClient_LL_SendReportedState..........FAILED!");
295  FreeEventInstance(event);
296  return false;
297  }
298  LogInfo(">>>IoTHubClient_LL_SendReportedState accepted state for transmission to IoT Hub.");
299  }
300 
301  while (true)
302  {
303  IoTHubClient_LL_DoWork(iotHubClientHandle);
304 
305  if (currentTrackingId == EVENT_CONFIRMED)
306  {
307  // IoT Hub got this event
308  return true;
309  }
310 
311  // Check timeout
312  int diff = (int)(millis() - start_ms);
313  if (diff >= EVENT_TIMEOUT_MS)
314  {
315  // Time out, reset the client
316  LogError("Waiting for send confirmation, time is up %d", diff);
317  resetClient = true;
318  }
319 
320  if (resetClient)
321  {
322  // resetClient also can be set as true in the IoTHubClient_LL_DoWork
323  // Disconnected, re-send the message
324  break;
325  }
326  else
327  {
328  // Sleep a while
329  ThreadAPI_Sleep(100);
330  }
331  }
332 
333  return false;
