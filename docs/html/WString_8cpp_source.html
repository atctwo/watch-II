155 unsigned char String::changeBuffer(unsigned int maxStrLen) {
156  // Can we use SSO here to avoid allocation?
157  if (maxStrLen < sizeof(sso.buff) - 1) {
158  if (isSSO() || !buffer()) {
159  // Already using SSO, nothing to do
160  uint16_t oldLen = len();
161  setSSO(true);
162  setLen(oldLen);
163  return 1;
164  } else { // if bufptr && !isSSO()
165  // Using bufptr, need to shrink into sso.buff
166  char temp[sizeof(sso.buff)];
167  memcpy(temp, buffer(), maxStrLen);
168  free(wbuffer());
169  uint16_t oldLen = len();
170  setSSO(true);
171  setLen(oldLen);
172  memcpy(wbuffer(), temp, maxStrLen);
173  return 1;
174  }
175  }
176  // Fallthrough to normal allocator
177  size_t newSize = (maxStrLen + 16) & (~0xf);
178  // Make sure we can fit newsize in the buffer
179  if (newSize > CAPACITY_MAX) {
180  return false;
181  }
182  uint16_t oldLen = len();
183  char *newbuffer = (char *) realloc(isSSO() ? nullptr : wbuffer(), newSize);
184  if (newbuffer) {
185  size_t oldSize = capacity() + 1; // include NULL.
186  if (isSSO()) {
187  // Copy the SSO buffer into allocated space
188  memmove(newbuffer, sso.buff, sizeof(sso.buff));
189  }
190  if (newSize > oldSize)
191  {
192  memset(newbuffer + oldSize, 0, newSize - oldSize);
193  }
194  setSSO(false);
195  setCapacity(newSize - 1);
196  setLen(oldLen); // Needed in case of SSO where len() never existed
197  setBuffer(newbuffer);
198  return 1;
199  }
200  return 0;
201 }

550 unsigned char String::equalsConstantTime(const String &s2) const {
551  // To avoid possible time-based attacks present function
552  // compares given strings in a constant time.
553  if(len() != s2.len())
554  return 0;
555  //at this point lengths are the same
556  if(len() == 0)
557  return 1;
558  //at this point lenghts are the same and non-zero

560  const char *p2 = s2.buffer();
561  unsigned int equalchars = 0;
562  unsigned int diffchars = 0;
563  while(*p1) {
564  if(*p1 == *p2)
565  ++equalchars;
566  else
567  ++diffchars;
568  ++p1;
569  ++p2;
570  }

572  unsigned char equalcond = (equalchars == len());
573  unsigned char diffcond = (diffchars == 0);
574  return (equalcond & diffcond); //bitwise AND

742  memmove(foundAt, replace.buffer(), replace.len());
