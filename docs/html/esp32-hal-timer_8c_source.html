28 typedef struct {
29  union {

34  uint32_t edge_int_en: 1; /*When set edge type interrupt will be generated during alarm*/
35  uint32_t divider: 16; /*Timer clock (T0/1_clk) pre-scale value.*/
36  uint32_t autoreload: 1; /*When set timer 0/1 auto-reload at alarming is enabled*/
37  uint32_t increase: 1; /*When set timer 0/1 time-base counter increment. When cleared timer 0 time-base counter decrement.*/
38  uint32_t enable: 1; /*When set timer 0/1 time-base counter is enabled*/
39  };
40  uint32_t val;
41  } config;
42  uint32_t cnt_low; /*Register to store timer 0/1 time-base counter current value lower 32 bits.*/
43  uint32_t cnt_high; /*Register to store timer 0 time-base counter current value higher 32 bits.*/
44  uint32_t update; /*Write any value will trigger a timer 0 time-base counter value update (timer 0 current value will be stored in registers above)*/
45  uint32_t alarm_low; /*Timer 0 time-base counter value lower 32 bits that will trigger the alarm*/
46  uint32_t alarm_high; /*Timer 0 time-base counter value higher 32 bits that will trigger the alarm*/
47  uint32_t load_low; /*Lower 32 bits of the value that will load into timer 0 time-base counter*/
48  uint32_t load_high; /*higher 32 bits of the value that will load into timer 0 time-base counter*/
49  uint32_t reload; /*Write any value will trigger timer 0 time-base counter reload*/

70 void IRAM_ATTR __timerISR(void * arg){
71  uint32_t s0 = TIMERG0.int_st_timers.val;
72  uint32_t s1 = TIMERG1.int_st_timers.val;
73  TIMERG0.int_clr_timers.val = s0;
74  TIMERG1.int_clr_timers.val = s1;
75  uint8_t status = (s1 & 3) << 2 | (s0 & 3);
76  uint8_t i = 4;
77  //restart the timers that should autoreload
78  while(i--){
79  hw_timer_reg_t * dev = hw_timer[i].dev;
80  if((status & (1 << i)) && dev->config.autoreload){
81  dev->config.alarm_en = 1;
82  }
83  }
84  i = 4;
85  //call callbacks
86  while(i--){
87  if(__timerInterruptHandlers[i] && (status & (1 << i))){
88  __timerInterruptHandlers[i]();
89  }
90  }
91 }

188 static void _on_apb_change(void * arg, apb_change_ev_t ev_type, uint32_t old_apb, uint32_t new_apb){
189  hw_timer_t * timer = (hw_timer_t *)arg;
190  if(ev_type == APB_BEFORE_CHANGE){
191  timer->dev->config.enable = 0;
192  } else {
193  old_apb /= 1000000;
194  new_apb /= 1000000;
195  timer->dev->config.divider = (new_apb * timer->dev->config.divider) / old_apb;
196  timer->dev->config.enable = 1;
197  }
198 }

200 hw_timer_t * timerBegin(uint8_t num, uint16_t divider, bool countUp){
201  if(num > 3){
202  return NULL;
203  }
204  hw_timer_t * timer = &hw_timer[num];
205  if(timer->group) {
206  DPORT_SET_PERI_REG_MASK(DPORT_PERIP_CLK_EN_REG, DPORT_TIMERGROUP1_CLK_EN);
207  DPORT_CLEAR_PERI_REG_MASK(DPORT_PERIP_RST_EN_REG, DPORT_TIMERGROUP1_RST);
208  TIMERG1.int_ena.val &= ~BIT(timer->timer);
209  } else {
210  DPORT_SET_PERI_REG_MASK(DPORT_PERIP_CLK_EN_REG, DPORT_TIMERGROUP_CLK_EN);
211  DPORT_CLEAR_PERI_REG_MASK(DPORT_PERIP_RST_EN_REG, DPORT_TIMERGROUP_RST);
212  TIMERG0.int_ena.val &= ~BIT(timer->timer);
213  }
214  timer->dev->config.enable = 0;
215  timerSetDivider(timer, divider);
216  timerSetCountUp(timer, countUp);
217  timerSetAutoReload(timer, false);
218  timerAttachInterrupt(timer, NULL, false);
219  timerWrite(timer, 0);
220  timer->dev->config.enable = 1;
221  addApbChangeCallback(timer, _on_apb_change);
222  return timer;
223 }

231 void timerAttachInterrupt(hw_timer_t *timer, void (*fn)(void), bool edge){
232  static bool initialized = false;
233  static intr_handle_t intr_handle = NULL;
234  if(intr_handle){
235  esp_intr_disable(intr_handle);
236  }
237  if(fn == NULL){
238  timer->dev->config.level_int_en = 0;
239  timer->dev->config.edge_int_en = 0;
240  timer->dev->config.alarm_en = 0;
241  if(timer->num & 2){
242  TIMERG1.int_ena.val &= ~BIT(timer->timer);
243  } else {
244  TIMERG0.int_ena.val &= ~BIT(timer->timer);
245  }
246  __timerInterruptHandlers[timer->num] = NULL;
247  } else {
248  __timerInterruptHandlers[timer->num] = fn;
249  timer->dev->config.level_int_en = edge?0:1;//When set, an alarm will generate a level type interrupt.
250  timer->dev->config.edge_int_en = edge?1:0;//When set, an alarm will generate an edge type interrupt.
251  int intr_source = 0;
252  if(!edge){
253  if(timer->group){
254  intr_source = ETS_TG1_T0_LEVEL_INTR_SOURCE + timer->timer;
255  } else {
256  intr_source = ETS_TG0_T0_LEVEL_INTR_SOURCE + timer->timer;
257  }
258  } else {
259  if(timer->group){
260  intr_source = ETS_TG1_T0_EDGE_INTR_SOURCE + timer->timer;
261  } else {
262  intr_source = ETS_TG0_T0_EDGE_INTR_SOURCE + timer->timer;
263  }
264  }
265  if(!initialized){
266  initialized = true;
267  esp_intr_alloc(intr_source, (int)(ESP_INTR_FLAG_IRAM|ESP_INTR_FLAG_LOWMED|ESP_INTR_FLAG_EDGE), __timerISR, NULL, &intr_handle);
268  } else {
269  intr_matrix_set(esp_intr_get_cpu(intr_handle), intr_source, esp_intr_get_intno(intr_handle));
270  }
271  if(timer->group){
272  TIMERG1.int_ena.val |= BIT(timer->timer);
273  } else {
274  TIMERG0.int_ena.val |= BIT(timer->timer);
275  }
276  }
277  if(intr_handle){
278  esp_intr_enable(intr_handle);
279  }
280 }
