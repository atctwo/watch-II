66 bool EEPROMClass::begin(size_t size) {
67  if (!size) {
68  return false;
69  }
70 
71  esp_err_t res = nvs_open(_name, NVS_READWRITE, &_handle);
72  if (res != ESP_OK) {
73  log_e("Unable to open NVS namespace: %d", res);
74  return false;
75  }
76 
77  size_t key_size = 0;
78  res = nvs_get_blob(_handle, _name, NULL, &key_size);
79  if(res != ESP_OK && res != ESP_ERR_NVS_NOT_FOUND) {
80  log_e("Unable to read NVS key: %d", res);
81  return false;
82  }
83  if (size < key_size) { // truncate
84  log_w("truncating EEPROM from %d to %d", key_size, size);
85  uint8_t* key_data = (uint8_t*) malloc(key_size);
86  if(!key_data) {
87  log_e("Not enough memory to truncate EEPROM!");
88  return false;
89  }
90  nvs_get_blob(_handle, _name, key_data, &key_size);
91  nvs_set_blob(_handle, _name, key_data, size);
92  nvs_commit(_handle);
93  free(key_data);
94  }
95  else if (size > key_size) { // expand or new
96  size_t expand_size = size - key_size;
97  uint8_t* expand_key = (uint8_t*) malloc(expand_size);
98  if(!expand_key) {
99  log_e("Not enough memory to expand EEPROM!");
100  return false;
101  }
102  // check for adequate free space
103  if(nvs_set_blob(_handle, "expand", expand_key, expand_size)) {
104  log_e("Not enough space to expand EEPROM from %d to %d", key_size, size);
105  free(expand_key);
106  return false;
107  }
108  free(expand_key);
109  nvs_erase_key(_handle, "expand");
110  uint8_t* key_data = (uint8_t*) malloc(size);
111  if(!key_data) {
112  log_e("Not enough memory to expand EEPROM!");
113  return false;
114  }
115  memset(key_data, 0xFF, size);
116  if(key_size) {
117  log_i("Expanding EEPROM from %d to %d", key_size, size);
118  // hold data while key is deleted
119  nvs_get_blob(_handle, _name, key_data, &key_size);
120  nvs_erase_key(_handle, _name);
121  } else {
122  log_i("New EEPROM of %d bytes", size);
123  }
124  nvs_commit(_handle);
125  nvs_set_blob(_handle, _name, key_data, size);
126  free(key_data);
127  nvs_commit(_handle);
128  }
129 
130  if (_data) {
131  delete[] _data;
132  }
133 
134  _data = (uint8_t*) malloc(size);
135  if(!_data) {
136  log_e("Not enough memory for %d bytes in EEPROM");
137  return false;
138  }
139  _size = size;
140  nvs_get_blob(_handle, _name, _data, &_size);
141  return true;
142 }

225 uint16_t EEPROMClass::convert (bool clear, const char* EEPROMname, const char* nvsname)
226 {
227  uint16_t result = 0;
228  const esp_partition_t* mypart = esp_partition_find_first(ESP_PARTITION_TYPE_DATA, ESP_PARTITION_SUBTYPE_ANY, EEPROMname);
229  if (mypart == NULL) {
230  log_i("EEPROM partition not found for conversion");
231  return result;
232  }
233 
234  size_t size = mypart->size;
235  uint8_t* data = (uint8_t*) malloc(size);
236  if (!data) {
237  log_e("Not enough memory to convert EEPROM!");
238  goto exit;
239  }
240 
241  if (esp_partition_read (mypart, 0, (void *) data, size) != ESP_OK) {
242  log_e("Unable to read EEPROM partition");
243  goto exit;
244  }
245 
246  bool empty;
247  empty = true;
248  for (int x=0; x<size; x++) {
249  if (data[x] != 0xFF) {
250  empty = false;
251  break;
252  }
253  }
254  if (empty) {
255  log_i("EEPROM partition is empty, will not convert");
256  goto exit;
257  }
258 
259  nvs_handle handle;
260  if (nvs_open(nvsname, NVS_READWRITE, &handle) != ESP_OK) {
261  log_e("Unable to open NVS");
262  goto exit;
263  }
264  esp_err_t err;
265  err = nvs_set_blob(handle, nvsname, data, size);
266  if (err != ESP_OK) {
267  log_e("Unable to add EEPROM data to NVS: %s", esp_err_to_name(err));
268  goto exit;
269  }
270  result = size;
271 
272  if (clear) {
273  if (esp_partition_erase_range (mypart, 0, size) != ESP_OK) {
274  log_w("Unable to clear EEPROM partition");
275  }
276  }
277 exit:
278  free(data);
279  return result;
280 }
