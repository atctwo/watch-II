83 bool __adcAttachPin(uint8_t pin){
84  int8_t channel = digitalPinToAnalogChannel(pin);
85  if(channel < 0){
86  log_e("Pin %u is not ADC pin!", pin);
87  return false;
88  }
89  int8_t pad = digitalPinToTouchChannel(pin);
90  if(pad >= 0){
91  uint32_t touch = READ_PERI_REG(SENS_SAR_TOUCH_ENABLE_REG);
92  if(touch & (1 << pad)){
93  touch &= ~((1 << (pad + SENS_TOUCH_PAD_OUTEN2_S))
94  | (1 << (pad + SENS_TOUCH_PAD_OUTEN1_S))
95  | (1 << (pad + SENS_TOUCH_PAD_WORKEN_S)));
96  WRITE_PERI_REG(SENS_SAR_TOUCH_ENABLE_REG, touch);
97  }
98  } else if(pin == 25){
99  CLEAR_PERI_REG_MASK(RTC_IO_PAD_DAC1_REG, RTC_IO_PDAC1_XPD_DAC | RTC_IO_PDAC1_DAC_XPD_FORCE);//stop dac1
100  } else if(pin == 26){
101  CLEAR_PERI_REG_MASK(RTC_IO_PAD_DAC2_REG, RTC_IO_PDAC2_XPD_DAC | RTC_IO_PDAC2_DAC_XPD_FORCE);//stop dac2
102  }
103 
104  pinMode(pin, ANALOG);
105  __analogSetPinAttenuation(pin, __analogAttenuation);
106  return true;
107 }

117 uint16_t __analogRead(uint8_t pin)
118 {
119  int8_t channel = digitalPinToAnalogChannel(pin);
120  int value = 0;
121  esp_err_t r = ESP_OK;
122  if(channel < 0){
123  log_e("Pin %u is not ADC pin!", pin);
124  return value;
125  }
126  __adcAttachPin(pin);
127  if(channel > 9){
128  channel -= 10;
129  r = adc2_get_raw( channel, __analogWidth, &value);
130  if ( r == ESP_OK ) {
131  return value;
132  } else if ( r == ESP_ERR_INVALID_STATE ) {
133  log_e("GPIO%u: %s: ADC2 not initialized yet.", pin, esp_err_to_name(r));
134  } else if ( r == ESP_ERR_TIMEOUT ) {
135  log_e("GPIO%u: %s: ADC2 is in use by Wi-Fi.", pin, esp_err_to_name(r));
136  } else {
137  log_e("GPIO%u: %s", pin, esp_err_to_name(r));
138  }
139  } else {
140  return adc1_get_raw(channel);
141  }
142  return value;
143 }

152 uint32_t __analogReadMilliVolts(uint8_t pin){
153  int8_t channel = digitalPinToAnalogChannel(pin);
154  if(channel < 0){
155  log_e("Pin %u is not ADC pin!", pin);
156  return 0;
157  }
158  if(!__analogVRef){
159  if (esp_adc_cal_check_efuse(ESP_ADC_CAL_VAL_EFUSE_TP) == ESP_OK) {
160  log_d("eFuse Two Point: Supported");
161  __analogVRef = DEFAULT_VREF;
162  }
163  if (esp_adc_cal_check_efuse(ESP_ADC_CAL_VAL_EFUSE_VREF) == ESP_OK) {
164  log_d("eFuse Vref: Supported");
165  __analogVRef = DEFAULT_VREF;
166  }
167  if(!__analogVRef){
168  __analogVRef = DEFAULT_VREF;
169  if(__analogVRefPin){
170  esp_adc_cal_characteristics_t chars;
171  if(adc2_vref_to_gpio(__analogVRefPin) == ESP_OK){
172  __analogVRef = __analogRead(__analogVRefPin);
173  esp_adc_cal_characterize(1, __analogAttenuation, __analogWidth, DEFAULT_VREF, &chars);
174  __analogVRef = esp_adc_cal_raw_to_voltage(__analogVRef, &chars);
175  log_d("Vref to GPIO%u: %u", __analogVRefPin, __analogVRef);
176  }
177  }
178  }
179  }
180  uint8_t unit = 1;
181  if(channel > 9){
182  unit = 2;
183  }
184  uint16_t adc_reading = __analogRead(pin);
185  if(__analogCharacteristics[unit - 1] == NULL){
186  __analogCharacteristics[unit - 1] = calloc(1, sizeof(esp_adc_cal_characteristics_t));
187  if(__analogCharacteristics[unit - 1] == NULL){
188  return 0;
189  }
190  esp_adc_cal_value_t val_type = esp_adc_cal_characterize(unit, __analogAttenuation, __analogWidth, __analogVRef, __analogCharacteristics[unit - 1]);
191  if (val_type == ESP_ADC_CAL_VAL_EFUSE_TP) {
192  log_i("ADC%u: Characterized using Two Point Value: %u\n", unit, __analogCharacteristics[unit - 1]->vref);
193  } else if (val_type == ESP_ADC_CAL_VAL_EFUSE_VREF) {
194  log_i("ADC%u: Characterized using eFuse Vref: %u\n", unit, __analogCharacteristics[unit - 1]->vref);
195  } else if(__analogVRef != DEFAULT_VREF){
196  log_i("ADC%u: Characterized using Vref to GPIO%u: %u\n", unit, __analogVRefPin, __analogCharacteristics[unit - 1]->vref);
197  } else {
198  log_i("ADC%u: Characterized using Default Vref: %u\n", unit, __analogCharacteristics[unit - 1]->vref);
199  }
200  }
201  return esp_adc_cal_raw_to_voltage(adc_reading, __analogCharacteristics[unit - 1]);
202 }

204 int __hallRead() //hall sensor without LNA
205 {
206  int Sens_Vp0;
207  int Sens_Vn0;
208  int Sens_Vp1;
209  int Sens_Vn1;
210 
211  pinMode(36, ANALOG);
212  pinMode(39, ANALOG);
213  SET_PERI_REG_MASK(SENS_SAR_TOUCH_CTRL1_REG, SENS_XPD_HALL_FORCE_M); // hall sens force enable
214  SET_PERI_REG_MASK(RTC_IO_HALL_SENS_REG, RTC_IO_XPD_HALL); // xpd hall
215  SET_PERI_REG_MASK(SENS_SAR_TOUCH_CTRL1_REG, SENS_HALL_PHASE_FORCE_M); // phase force
216  CLEAR_PERI_REG_MASK(RTC_IO_HALL_SENS_REG, RTC_IO_HALL_PHASE); // hall phase
217  Sens_Vp0 = __analogRead(36);
218  Sens_Vn0 = __analogRead(39);
219  SET_PERI_REG_MASK(RTC_IO_HALL_SENS_REG, RTC_IO_HALL_PHASE);
220  Sens_Vp1 = __analogRead(36);
221  Sens_Vn1 = __analogRead(39);
222  SET_PERI_REG_BITS(SENS_SAR_MEAS_WAIT2_REG, SENS_FORCE_XPD_SAR, 0, SENS_FORCE_XPD_SAR_S);
223  CLEAR_PERI_REG_MASK(SENS_SAR_TOUCH_CTRL1_REG, SENS_XPD_HALL_FORCE);
224  CLEAR_PERI_REG_MASK(SENS_SAR_TOUCH_CTRL1_REG, SENS_HALL_PHASE_FORCE);
225  return (Sens_Vp1 - Sens_Vp0) - (Sens_Vn1 - Sens_Vn0);
226 }
