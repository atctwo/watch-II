180 bool UpdateClass::_writeBuffer(){
181  //first bytes of new firmware
182  if(!_progress && _command == U_FLASH){
183  //check magic
184  if(_buffer[0] != ESP_IMAGE_HEADER_MAGIC){
185  _abort(UPDATE_ERROR_MAGIC_BYTE);
186  return false;
187  }
188  //remove magic byte from the firmware now and write it upon success
189  //this ensures that partially written firmware will not be bootable
190  _buffer[0] = 0xFF;
191  }
192  if (!_progress && _progress_callback) {
193  _progress_callback(0, _size);
194  }
195  if(!ESP.flashEraseSector((_partition->address + _progress)/SPI_FLASH_SEC_SIZE)){
196  _abort(UPDATE_ERROR_ERASE);
197  return false;
198  }
199  if (!ESP.flashWrite(_partition->address + _progress, (uint32_t*)_buffer, _bufferLen)) {
200  _abort(UPDATE_ERROR_WRITE);
201  return false;
202  }
203  //restore magic or md5 will fail
204  if(!_progress && _command == U_FLASH){
205  _buffer[0] = ESP_IMAGE_HEADER_MAGIC;
206  }
207  _md5.add(_buffer, _bufferLen);
208  _progress += _bufferLen;
