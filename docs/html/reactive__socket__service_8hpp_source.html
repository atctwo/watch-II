381  template <typename Socket>
382  asio::error_code accept(implementation_type& impl,
383  Socket& peer, endpoint_type* peer_endpoint, asio::error_code& ec)
384  {
385  // We cannot accept a socket that is already open.
386  if (peer.is_open())
387  {
388  ec = asio::error::already_open;
389  return ec;
390  }
391 
392  std::size_t addr_len = peer_endpoint ? peer_endpoint->capacity() : 0;
393  socket_holder new_socket(socket_ops::sync_accept(impl.socket_,
394  impl.state_, peer_endpoint ? peer_endpoint->data() : 0,
395  peer_endpoint ? &addr_len : 0, ec));
396

398  if (new_socket.get() != invalid_socket)

402  peer.assign(impl.protocol_, new_socket.get(), ec);
403  if (!ec)
404  new_socket.release();
405  }
