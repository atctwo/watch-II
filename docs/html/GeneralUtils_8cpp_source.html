38  a4[0] = (a3[0] & 0xfc) >> 2;
39  a4[1] = ((a3[0] & 0x03) << 4) + ((a3[1] & 0xf0) >> 4);
40  a4[2] = ((a3[1] & 0x0f) << 2) + ((a3[2] & 0xc0) >> 6);

349 std::vector<std::string> GeneralUtils::split(std::string source, char delimiter) {
350  // See also: https://stackoverflow.com/questions/5167625/splitting-a-c-stdstring-using-tokens-e-g
351  std::vector<std::string> strings;
352  std::size_t current, previous = 0;
353  current = source.find(delimiter);
354  while (current != std::string::npos) {
355  strings.push_back(trim(source.substr(previous, current - previous)));
356  previous = current + 1;
357  current = source.find(delimiter, previous);
358  }
359  strings.push_back(trim(source.substr(previous, current - previous)));
360  return strings;
