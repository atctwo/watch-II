132  std::size_t cancel(implementation_type& impl, asio::error_code& ec)
133  {

144  impl.might_have_pending_waits = false;

214  // Perform a blocking wait on the timer.
215  void wait(implementation_type& impl, asio::error_code& ec)
216  {
217  time_type now = Time_Traits::now();
218  ec = asio::error_code();
219  while (Time_Traits::less_than(now, impl.expiry) && !ec)
220  {
221  this->do_wait(Time_Traits::to_posix_duration(
222  Time_Traits::subtract(impl.expiry, now)), ec);
223  now = Time_Traits::now();
224  }
225  }

227  // Start an asynchronous wait on the timer.
228  template <typename Handler>
229  void async_wait(implementation_type& impl, Handler& handler)
230  {
231  // Allocate and construct an operation to wrap the handler.
232  typedef wait_handler<Handler> op;
233  typename op::ptr p = { asio::detail::addressof(handler),
234  op::ptr::allocate(handler), 0 };
235  p.p = new (p.v) op(handler);
236 
237  impl.might_have_pending_waits = true;
238 
239  ASIO_HANDLER_CREATION((scheduler_.context(),
240  *p.p, "deadline_timer", &impl, 0, "async_wait"));
241 
242  scheduler_.schedule_timer(timer_queue_, impl.expiry, impl.timer_data, p.p);
243  p.v = p.p = 0;
244  }

247  // Helper function to wait given a duration type. The duration type should
248  // either be of type boost::posix_time::time_duration, or implement the
249  // required subset of its interface.
250  template <typename Duration>
251  void do_wait(const Duration& timeout, asio::error_code& ec)
252  {
253 #if defined(ASIO_WINDOWS_RUNTIME)
254  std::this_thread::sleep_for(
255  std::chrono::seconds(timeout.total_seconds())
256  + std::chrono::microseconds(timeout.total_microseconds()));
257  ec = asio::error_code();
258 #else // defined(ASIO_WINDOWS_RUNTIME)
259  ::timeval tv;
260  tv.tv_sec = timeout.total_seconds();
261  tv.tv_usec = timeout.total_microseconds() % 1000000;
262  socket_ops::select(0, 0, 0, 0, &tv, ec);
263 #endif // defined(ASIO_WINDOWS_RUNTIME)
264  }
