56 static void _on_apb_change(void * arg, apb_change_ev_t ev_type, uint32_t old_apb, uint32_t new_apb){
57  if(ev_type == APB_AFTER_CHANGE && old_apb != new_apb){
58  uint16_t iarg = *(uint16_t*)arg;
59  uint8_t chan = 0;
60  old_apb /= 1000000;
61  new_apb /= 1000000;
62  while(iarg){ // run though all active channels, adjusting timing configurations
63  if(iarg & 1) {// this channel is active
64  uint8_t group=(chan/8), timer=((chan/2)%4);
65  if(LEDC_TIMER(group, timer).conf.tick_sel){
66  LEDC_MUTEX_LOCK();
67  uint32_t old_div = LEDC_TIMER(group, timer).conf.clock_divider;
68  uint32_t div_num = (new_apb * old_div) / old_apb;
69  if(div_num > LEDC_DIV_NUM_HSTIMER0_V){
70  div_num = ((REF_CLK_FREQ /1000000) * old_div) / old_apb;
71  if(div_num > LEDC_DIV_NUM_HSTIMER0_V) {
72  div_num = LEDC_DIV_NUM_HSTIMER0_V;//lowest clock possible
73  }
74  LEDC_TIMER(group, timer).conf.tick_sel = 0;
75  } else if(div_num < 256) {
76  div_num = 256;//highest clock possible
77  }
78  LEDC_TIMER(group, timer).conf.clock_divider = div_num;
79  LEDC_MUTEX_UNLOCK();
80  }
81  else {
82  log_d("using REF_CLK chan=%d",chan);
83  }
84  }
85  iarg = iarg >> 1;
86  chan++;
87  }
88  }
89 }

91 //uint32_t frequency = (80MHz or 1MHz)/((div_num / 256.0)*(1 << bit_num));
92 static void _ledcSetupTimer(uint8_t chan, uint32_t div_num, uint8_t bit_num, bool apb_clk)
93 {
94  uint8_t group=(chan/8), timer=((chan/2)%4);
95  static bool tHasStarted = false;
96  static uint16_t _activeChannels = 0;
97  if(!tHasStarted) {
98  tHasStarted = true;
99  DPORT_SET_PERI_REG_MASK(DPORT_PERIP_CLK_EN_REG, DPORT_LEDC_CLK_EN);
100  DPORT_CLEAR_PERI_REG_MASK(DPORT_PERIP_RST_EN_REG, DPORT_LEDC_RST);
101  LEDC.conf.apb_clk_sel = 1;//LS use apb clock
102  addApbChangeCallback((void*)&_activeChannels, _on_apb_change);
103 
104 #if !CONFIG_DISABLE_HAL_LOCKS
105  _ledc_sys_lock = xSemaphoreCreateMutex();
106 #endif
107  }
108  LEDC_MUTEX_LOCK();
109  LEDC_TIMER(group, timer).conf.clock_divider = div_num;//18 bit (10.8) This register is used to configure parameter for divider in timer the least significant eight bits represent the decimal part.
110  LEDC_TIMER(group, timer).conf.duty_resolution = bit_num;//5 bit This register controls the range of the counter in timer. the counter range is [0 2**bit_num] the max bit width for counter is 20.
111  LEDC_TIMER(group, timer).conf.tick_sel = apb_clk;//apb clock
112  if(group) {
113  LEDC_TIMER(group, timer).conf.low_speed_update = 1;//This bit is only useful for low speed timer channels, reserved for high speed timers
114  }
115  LEDC_TIMER(group, timer).conf.pause = 0;
116  LEDC_TIMER(group, timer).conf.rst = 1;//This bit is used to reset timer the counter will be 0 after reset.
117  LEDC_TIMER(group, timer).conf.rst = 0;
118  LEDC_MUTEX_UNLOCK();
119  _activeChannels |= (1 << chan); // mark as active for APB callback
120 }

123 //max bit_num 0x1F (31)
124 static double _ledcSetupTimerFreq(uint8_t chan, double freq, uint8_t bit_num)
125 {
126  uint64_t clk_freq = getApbFrequency();
127  clk_freq <<= 8;//div_num is 8 bit decimal
128  uint32_t div_num = (clk_freq >> bit_num) / freq;
129  bool apb_clk = true;
130  if(div_num > LEDC_DIV_NUM_HSTIMER0_V) {
131  clk_freq /= 80;
132  div_num = (clk_freq >> bit_num) / freq;
133  if(div_num > LEDC_DIV_NUM_HSTIMER0_V) {
134  div_num = LEDC_DIV_NUM_HSTIMER0_V;//lowest clock possible
135  }
136  apb_clk = false;
137  } else if(div_num < 256) {
138  div_num = 256;//highest clock possible
139  }
140  _ledcSetupTimer(chan, div_num, bit_num, apb_clk);
141  //log_i("Fin: %f, Fclk: %uMhz, bits: %u, DIV: %u, Fout: %f",
142  // freq, apb_clk?80:1, bit_num, div_num, (clk_freq >> bit_num) / (double)div_num);
143  return (clk_freq >> bit_num) / (double)div_num;
144 }

146 static double _ledcTimerRead(uint8_t chan)
147 {
148  uint32_t div_num;
149  uint8_t bit_num;
150  bool apb_clk;
151  uint8_t group=(chan/8), timer=((chan/2)%4);
152  LEDC_MUTEX_LOCK();
153  div_num = LEDC_TIMER(group, timer).conf.clock_divider;//18 bit (10.8) This register is used to configure parameter for divider in timer the least significant eight bits represent the decimal part.
154  bit_num = LEDC_TIMER(group, timer).conf.duty_resolution;//5 bit This register controls the range of the counter in timer. the counter range is [0 2**bit_num] the max bit width for counter is 20.
155  apb_clk = LEDC_TIMER(group, timer).conf.tick_sel;//apb clock
156  LEDC_MUTEX_UNLOCK();
157  uint64_t clk_freq = 1000000;
158  if(apb_clk) {
159  clk_freq = getApbFrequency();
160  }
161  clk_freq <<= 8;//div_num is 8 bit decimal
162  return (clk_freq >> bit_num) / (double)div_num;
163 }

165 static void _ledcSetupChannel(uint8_t chan, uint8_t idle_level)
166 {
167  uint8_t group=(chan/8), channel=(chan%8), timer=((chan/2)%4);
168  LEDC_MUTEX_LOCK();
169  LEDC_CHAN(group, channel).conf0.timer_sel = timer;//2 bit Selects the timer to attach 0-3
170  LEDC_CHAN(group, channel).conf0.idle_lv = idle_level;//1 bit This bit is used to control the output value when channel is off.
171  LEDC_CHAN(group, channel).hpoint.hpoint = 0;//20 bit The output value changes to high when timer selected by channel has reached hpoint
172  LEDC_CHAN(group, channel).conf1.duty_inc = 1;//1 bit This register is used to increase the duty of output signal or decrease the duty of output signal for high speed channel
173  LEDC_CHAN(group, channel).conf1.duty_num = 1;//10 bit This register is used to control the number of increased or decreased times for channel
174  LEDC_CHAN(group, channel).conf1.duty_cycle = 1;//10 bit This register is used to increase or decrease the duty every duty_cycle cycles for channel
175  LEDC_CHAN(group, channel).conf1.duty_scale = 0;//10 bit This register controls the increase or decrease step scale for channel.
176  LEDC_CHAN(group, channel).duty.duty = 0;
177  LEDC_CHAN(group, channel).conf0.sig_out_en = 0;//This is the output enable control bit for channel
178  LEDC_CHAN(group, channel).conf1.duty_start = 0;//When duty_num duty_cycle and duty_scale has been configured. these register won't take effect until set duty_start. this bit is automatically cleared by hardware.
179  if(group) {
180  LEDC_CHAN(group, channel).conf0.low_speed_update = 1;
181  } else {
182  LEDC_CHAN(group, channel).conf0.clk_en = 0;
183  }
184  LEDC_MUTEX_UNLOCK();
185 }

197 void ledcWrite(uint8_t chan, uint32_t duty)
198 {
199  if(chan > 15) {
200  return;
201  }
202  uint8_t group=(chan/8), channel=(chan%8);
203  LEDC_MUTEX_LOCK();
204  LEDC_CHAN(group, channel).duty.duty = duty << 4;//25 bit (21.4)
205  if(duty) {
206  LEDC_CHAN(group, channel).conf0.sig_out_en = 1;//This is the output enable control bit for channel
207  LEDC_CHAN(group, channel).conf1.duty_start = 1;//When duty_num duty_cycle and duty_scale has been configured. these register won't take effect until set duty_start. this bit is automatically cleared by hardware.
208  if(group) {
209  LEDC_CHAN(group, channel).conf0.low_speed_update = 1;
210  } else {
211  LEDC_CHAN(group, channel).conf0.clk_en = 1;
212  }
213  } else {
214  LEDC_CHAN(group, channel).conf0.sig_out_en = 0;//This is the output enable control bit for channel
215  LEDC_CHAN(group, channel).conf1.duty_start = 0;//When duty_num duty_cycle and duty_scale has been configured. these register won't take effect until set duty_start. this bit is automatically cleared by hardware.
216  if(group) {
217  LEDC_CHAN(group, channel).conf0.low_speed_update = 1;
218  } else {
219  LEDC_CHAN(group, channel).conf0.clk_en = 0;
220  }
221  }
222  LEDC_MUTEX_UNLOCK();
