82 bool F_Fat::format(bool full_wipe, char* partitionLabel)
83 {
84  esp_err_t result;
85  bool res = true;
86  if(_wl_handle != WL_INVALID_HANDLE){
87  log_w("Already Mounted!");
88  return false;
89  }
90  wl_handle_t temp_handle;
91 // Attempt to mount to see if there is already data
92  const esp_partition_t *ffat_partition = check_ffat_partition(partitionLabel);
93  if (!ffat_partition){
94  log_w("No partition!");
95  return false;
96  }
97  result = wl_mount(ffat_partition, &temp_handle);
98 
99  if (result == ESP_OK) {
100 // Wipe disk- quick just wipes the FAT. Full zeroes the whole disk
101  uint32_t wipe_size = full_wipe ? wl_size(temp_handle) : 16384;
102  wl_erase_range(temp_handle, 0, wipe_size);
103  wl_unmount(temp_handle);
104  } else {
105  res = false;
106  log_w("wl_mount failed!");
107  }
108 // Now do a mount with format_if_fail (which it will)
109  esp_vfs_fat_mount_config_t conf = {
110  .format_if_mount_failed = true,
111  .max_files = 1,
112  .allocation_unit_size = CONFIG_WL_SECTOR_SIZE
113  };
114  result = esp_vfs_fat_spiflash_mount("/format_ffat", partitionLabel, &conf, &temp_handle);
115  esp_vfs_fat_spiflash_unmount("/format_ffat", temp_handle);
116  if (result != ESP_OK){
117  res = false;
118  log_w("esp_vfs_fat_spiflash_mount failed!");
119  }
120  return res;
121 }
