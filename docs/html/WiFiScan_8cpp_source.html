57 int16_t WiFiScanClass::scanNetworks(bool async, bool show_hidden, bool passive, uint32_t max_ms_per_chan, uint8_t channel)
58 {
59  if(WiFiGenericClass::getStatusBits() & WIFI_SCANNING_BIT) {
60  return WIFI_SCAN_RUNNING;
61  }
62 
63  WiFiScanClass::_scanTimeout = max_ms_per_chan * 20;
64  WiFiScanClass::_scanAsync = async;
65 
66  WiFi.enableSTA(true);
67 
68  scanDelete();
69 
70  wifi_scan_config_t config;
71  config.ssid = 0;
72  config.bssid = 0;
73  config.channel = channel;
74  config.show_hidden = show_hidden;
75  if(passive){
76  config.scan_type = WIFI_SCAN_TYPE_PASSIVE;
77  config.scan_time.passive = max_ms_per_chan;
78  } else {
79  config.scan_type = WIFI_SCAN_TYPE_ACTIVE;
80  config.scan_time.active.min = 100;
81  config.scan_time.active.max = max_ms_per_chan;
82  }
83  if(esp_wifi_scan_start(&config, false) == ESP_OK) {
84  _scanStarted = millis();
85  if (!_scanStarted) { //Prevent 0 from millis overflow
86  ++_scanStarted;
87  }
88 
89  WiFiGenericClass::clearStatusBits(WIFI_SCAN_DONE_BIT);
90  WiFiGenericClass::setStatusBits(WIFI_SCANNING_BIT);
91 
92  if(WiFiScanClass::_scanAsync) {
93  return WIFI_SCAN_RUNNING;
94  }
95  if(WiFiGenericClass::waitStatusBits(WIFI_SCAN_DONE_BIT, 10000)){
96  return (int16_t) WiFiScanClass::_scanCount;
97  }
98  }
99  return WIFI_SCAN_FAILED;
100 }

109 void WiFiScanClass::_scanDone()
110 {
111  esp_wifi_scan_get_ap_num(&(WiFiScanClass::_scanCount));
112  if(WiFiScanClass::_scanCount) {
113  WiFiScanClass::_scanResult = new wifi_ap_record_t[WiFiScanClass::_scanCount];
114  if(!WiFiScanClass::_scanResult || esp_wifi_scan_get_ap_records(&(WiFiScanClass::_scanCount), (wifi_ap_record_t*)_scanResult) != ESP_OK) {
115  WiFiScanClass::_scanCount = 0;
116  }
117  }
118  WiFiScanClass::_scanStarted=0; //Reset after a scan is completed for normal behavior
119  WiFiGenericClass::setStatusBits(WIFI_SCAN_DONE_BIT);
120  WiFiGenericClass::clearStatusBits(WIFI_SCANNING_BIT);
121 }

142 int16_t WiFiScanClass::scanComplete()
143 {
144  if (WiFiScanClass::_scanStarted && (millis()-WiFiScanClass::_scanStarted) > WiFiScanClass::_scanTimeout) { //Check is scan was started and if the delay expired, return WIFI_SCAN_FAILED in this case
145  WiFiGenericClass::clearStatusBits(WIFI_SCANNING_BIT);
146  return WIFI_SCAN_FAILED;
147  }
148 
149  if(WiFiGenericClass::getStatusBits() & WIFI_SCAN_DONE_BIT) {
150  return WiFiScanClass::_scanCount;
151  }
152 
153  if(WiFiGenericClass::getStatusBits() & WIFI_SCANNING_BIT) {
154  return WIFI_SCAN_RUNNING;
155  }
156 
157  return WIFI_SCAN_FAILED;
158 }
