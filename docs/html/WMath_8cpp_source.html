38 long random(long howbig)
39 {
40  uint32_t x = esp_random();
41  uint64_t m = uint64_t(x) * uint64_t(howbig);
42  uint32_t l = uint32_t(m);
43  if (l < howbig) {
44  uint32_t t = -howbig;
45  if (t >= howbig) {
46  t -= howbig;
47  if (t >= howbig)
48  t %= howbig;
49  }
50  while (l < t) {
51  x = esp_random();
52  m = uint64_t(x) * uint64_t(howbig);
53  l = uint32_t(m);
54  }
55  }
56  return m >> 32;
57 }
