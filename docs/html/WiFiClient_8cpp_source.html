43  size_t r_available()
44  {
45  if(_fd < 0){
46  return 0;
47  }
48  int count;
49  int res = lwip_ioctl_r(_fd, FIONREAD, &count);
50  if(res < 0) {
51  _failed = true;
52  return 0;
53  }
54  return count;

57  size_t fillBuffer()
58  {
59  if(!_buffer){
60  _buffer = (uint8_t *)malloc(_size);
61  if(!_buffer) {
62  log_e("Not enough memory to allocate buffer");
63  _failed = true;
64  return 0;
65  }
66  }
67  if(_fill && _pos == _fill){
68  _fill = 0;
69  _pos = 0;
70  }
71  if(!_buffer || _size <= _fill || !r_available()) {
72  return 0;
73  }
74  int res = recv(_fd, _buffer + _fill, _size - _fill, MSG_DONTWAIT);
75  if(res < 0) {
76  if(errno != EWOULDBLOCK) {
77  _failed = true;
78  }
79  return 0;
80  }
81  _fill += res;
82  return res;
83  }

106  int read(uint8_t * dst, size_t len){
107  if(!dst || !len || (_pos == _fill && !fillBuffer())){
108  return -1;
109  }
110  size_t a = _fill - _pos;
111  if(len <= a || ((len - a) <= (_size - _fill) && fillBuffer() >= (len - a))){
112  if(len == 1){
113  *dst = _buffer[_pos];
114  } else {
115  memcpy(dst, _buffer + _pos, len);
116  }
117  _pos += len;
118  return len;
119  }
120  size_t left = len;
121  size_t toRead = a;
122  uint8_t * buf = dst;
123  memcpy(buf, _buffer + _pos, toRead);
124  _pos += toRead;
125  left -= toRead;
126  buf += toRead;
127  while(left){
128  if(!fillBuffer()){
129  return len - left;
130  }
131  a = _fill - _pos;
132  toRead = (a > left)?left:a;
133  memcpy(buf, _buffer + _pos, toRead);
134  _pos += toRead;
135  left -= toRead;
136  buf += toRead;
137  }
138  return len;
139  }
140 
141  int peek(){
142  if(_pos == _fill && !fillBuffer()){
143  return -1;
144  }
145  return _buffer[_pos];
146  }
147 
148  size_t available(){
149  return _fill - _pos + r_available();
150  }

208 int WiFiClient::connect(IPAddress ip, uint16_t port, int32_t timeout)
209 {
210  int sockfd = socket(AF_INET, SOCK_STREAM, 0);
211  if (sockfd < 0) {
212  log_e("socket: %d", errno);
213  return 0;
214  }
215  fcntl( sockfd, F_SETFL, fcntl( sockfd, F_GETFL, 0 ) | O_NONBLOCK );
216 
217  uint32_t ip_addr = ip;
218  struct sockaddr_in serveraddr;
219  bzero((char *) &serveraddr, sizeof(serveraddr));
220  serveraddr.sin_family = AF_INET;
221  bcopy((const void *)(&ip_addr), (void *)&serveraddr.sin_addr.s_addr, 4);
222  serveraddr.sin_port = htons(port);
223  fd_set fdset;
224  struct timeval tv;
225  FD_ZERO(&fdset);
226  FD_SET(sockfd, &fdset);
227  tv.tv_sec = 0;
228  tv.tv_usec = timeout * 1000;
229 
230  int res = lwip_connect_r(sockfd, (struct sockaddr*)&serveraddr, sizeof(serveraddr));
231  if (res < 0 && errno != EINPROGRESS) {
232  log_e("connect on fd %d, errno: %d, \"%s\"", sockfd, errno, strerror(errno));
233  close(sockfd);
234  return 0;
235  }
236 
237  res = select(sockfd + 1, nullptr, &fdset, nullptr, timeout<0 ? nullptr : &tv);
238  if (res < 0) {
239  log_e("select on fd %d, errno: %d, \"%s\"", sockfd, errno, strerror(errno));
240  close(sockfd);
241  return 0;
242  } else if (res == 0) {
243  log_i("select returned due to timeout %d ms for fd %d", timeout, sockfd);
244  close(sockfd);
245  return 0;
246  } else {
247  int sockerr;
248  socklen_t len = (socklen_t)sizeof(int);
249  res = getsockopt(sockfd, SOL_SOCKET, SO_ERROR, &sockerr, &len);
250 
251  if (res < 0) {
252  log_e("getsockopt on fd %d, errno: %d, \"%s\"", sockfd, errno, strerror(errno));
253  close(sockfd);
254  return 0;
255  }
256 
257  if (sockerr != 0) {
258  log_e("socket error on fd %d, errno: %d, \"%s\"", sockfd, sockerr, strerror(sockerr));
259  close(sockfd);
260  return 0;
261  }
262  }
263 
264  fcntl( sockfd, F_SETFL, fcntl( sockfd, F_GETFL, 0 ) & (~O_NONBLOCK) );
265  clientSocketHandle.reset(new WiFiClientSocketHandle(sockfd));
266  _rxBuffer.reset(new WiFiClientRxBuffer(sockfd));
267  _connected = true;
268  return 1;
269 }

352 size_t WiFiClient::write(const uint8_t *buf, size_t size)
353 {
354  int res =0;
355  int retry = WIFI_CLIENT_MAX_WRITE_RETRY;
356  int socketFileDescriptor = fd();
357  size_t totalBytesSent = 0;
358  size_t bytesRemaining = size;
359 
360  if(!_connected || (socketFileDescriptor < 0)) {
361  return 0;
362  }
363 
364  while(retry) {
365  //use select to make sure the socket is ready for writing
366  fd_set set;
367  struct timeval tv;
368  FD_ZERO(&set); // empties the set
369  FD_SET(socketFileDescriptor, &set); // adds FD to the set
370  tv.tv_sec = 0;
371  tv.tv_usec = WIFI_CLIENT_SELECT_TIMEOUT_US;
372  retry--;
373 
374  if(select(socketFileDescriptor + 1, NULL, &set, NULL, &tv) < 0) {
375  return 0;
376  }
377 
378  if(FD_ISSET(socketFileDescriptor, &set)) {
379  res = send(socketFileDescriptor, (void*) buf, bytesRemaining, MSG_DONTWAIT);
380  if(res > 0) {
381  totalBytesSent += res;
382  if (totalBytesSent >= size) {
383  //completed successfully
384  retry = 0;
385  } else {
386  buf += res;
387  bytesRemaining -= res;
388  retry = WIFI_CLIENT_MAX_WRITE_RETRY;
389  }
390  }
391  else if(res < 0) {
392  log_e("fail on fd %d, errno: %d, \"%s\"", fd(), errno, strerror(errno));
393  if(errno != EAGAIN) {
394  //if resource was busy, can try again, otherwise give up
395  stop();
396  res = 0;
397  retry = 0;
398  }
399  }
400  else {
401  // Try again
402  }
403  }
404  }
405  return totalBytesSent;
406 }

492 {
493  if (_connected) {
494  uint8_t dummy;
495  int res = recv(fd(), &dummy, 0, MSG_DONTWAIT);
496  // avoid unused var warning by gcc
497  (void)res;
498  // recv only sets errno if res is <= 0
499  if (res <= 0){
500  switch (errno) {
501  case EWOULDBLOCK:
502  case ENOENT: //caused by vfs
503  _connected = true;
504  break;
505  case ENOTCONN:

507  case ECONNRESET:
508  case ECONNREFUSED:

513  default:
514  log_i("Unexpected: RES: %d, ERR: %d", res, errno);
515  _connected = true;
516  break;
517  }
518  } else {
519  _connected = true;
520  }
521  }
