41  template <typename SyncRandomAccessReadDevice, typename MutableBufferSequence,
42  typename MutableBufferIterator, typename CompletionCondition>
43  std::size_t read_at_buffer_sequence(SyncRandomAccessReadDevice& d,
44  uint64_t offset, const MutableBufferSequence& buffers,
45  const MutableBufferIterator&, CompletionCondition completion_condition,
46  asio::error_code& ec)
47  {
48  ec = asio::error_code();
49  asio::detail::consuming_buffers<mutable_buffer,
50  MutableBufferSequence, MutableBufferIterator> tmp(buffers);
51  while (!tmp.empty())
52  {

54  completion_condition(ec, tmp.total_consumed())))

216  {

218  switch (start_ = start)

220  case 1:
