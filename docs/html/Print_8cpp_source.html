252 size_t Print::printNumber(unsigned long n, uint8_t base)
253 {
254  char buf[8 * sizeof(long) + 1]; // Assumes 8-bit chars plus zero byte.
255  char *str = &buf[sizeof(buf) - 1];
256 
257  *str = '\0';
258 
259  // prevent crash if called with base == 1
260  if(base < 2) {
261  base = 10;
262  }
263 
264  do {
265  unsigned long m = n;
266  n /= base;
267  char c = m - base * n;

269  } while(n);
270 
271  return write(str);

274 size_t Print::printFloat(double number, uint8_t digits)
275 {
276  size_t n = 0;
277 
278  if(isnan(number)) {
279  return print("nan");
280  }
281  if(isinf(number)) {
282  return print("inf");
283  }
284  if(number > 4294967040.0) {
285  return print("ovf"); // constant determined empirically
286  }
287  if(number < -4294967040.0) {
288  return print("ovf"); // constant determined empirically
289  }
290 
291  // Handle negative numbers
292  if(number < 0.0) {
293  n += print('-');
294  number = -number;
295  }
296 
297  // Round correctly so that print(1.999, 2) prints as "2.00"
298  double rounding = 0.5;
299  for(uint8_t i = 0; i < digits; ++i) {
300  rounding /= 10.0;
301  }
302 
303  number += rounding;
304 
305  // Extract the integer part of the number and print it
306  unsigned long int_part = (unsigned long) number;
307  double remainder = number - (double) int_part;
308  n += print(int_part);
309 
310  // Print the decimal point, but only if there are digits beyond
311  if(digits > 0) {
312  n += print(".");
313  }
314 
315  // Extract digits from the remainder one at a time
316  while(digits-- > 0) {
317  remainder *= 10.0;
318  int toPrint = int(remainder);
319  n += print(toPrint);
320  remainder -= toPrint;
321  }
322 
323  return n;
324 }
