83 extern void IRAM_ATTR __pinMode(uint8_t pin, uint8_t mode)
84 {
85 
86  if(!digitalPinIsValid(pin)) {
87  return;
88  }
89 
90  uint32_t rtc_reg = rtc_gpio_desc[pin].reg;
91  if(mode == ANALOG) {
92  if(!rtc_reg) {
93  return;//not rtc pin
94  }
95  //lock rtc
96  uint32_t reg_val = ESP_REG(rtc_reg);
97  if(reg_val & rtc_gpio_desc[pin].mux){
98  return;//already in adc mode
99  }
100  reg_val &= ~(
101  (RTC_IO_TOUCH_PAD1_FUN_SEL_V << rtc_gpio_desc[pin].func)
102  |rtc_gpio_desc[pin].ie
103  |rtc_gpio_desc[pin].pullup
104  |rtc_gpio_desc[pin].pulldown);
105  ESP_REG(RTC_GPIO_ENABLE_W1TC_REG) = (1 << (rtc_gpio_desc[pin].rtc_num + RTC_GPIO_ENABLE_W1TC_S));
106  ESP_REG(rtc_reg) = reg_val | rtc_gpio_desc[pin].mux;
107  //unlock rtc
108  ESP_REG(DR_REG_IO_MUX_BASE + esp32_gpioMux[pin].reg) = ((uint32_t)2 << MCU_SEL_S) | ((uint32_t)2 << FUN_DRV_S) | FUN_IE;
109  return;
110  }
111 
112  //RTC pins PULL settings
113  if(rtc_reg) {
114  //lock rtc
115  ESP_REG(rtc_reg) = ESP_REG(rtc_reg) & ~(rtc_gpio_desc[pin].mux);
116  if(mode & PULLUP) {
117  ESP_REG(rtc_reg) = (ESP_REG(rtc_reg) | rtc_gpio_desc[pin].pullup) & ~(rtc_gpio_desc[pin].pulldown);
118  } else if(mode & PULLDOWN) {
119  ESP_REG(rtc_reg) = (ESP_REG(rtc_reg) | rtc_gpio_desc[pin].pulldown) & ~(rtc_gpio_desc[pin].pullup);
120  } else {
121  ESP_REG(rtc_reg) = ESP_REG(rtc_reg) & ~(rtc_gpio_desc[pin].pullup | rtc_gpio_desc[pin].pulldown);
122  }
123  //unlock rtc
124  }
125 
126  uint32_t pinFunction = 0, pinControl = 0;
127 
128  //lock gpio
129  if(mode & INPUT) {
130  if(pin < 32) {
131  GPIO.enable_w1tc = ((uint32_t)1 << pin);
132  } else {
133  GPIO.enable1_w1tc.val = ((uint32_t)1 << (pin - 32));
134  }
135  } else if(mode & OUTPUT) {
136  if(pin > 33){
137  //unlock gpio
138  return;//pins above 33 can be only inputs
139  } else if(pin < 32) {
140  GPIO.enable_w1ts = ((uint32_t)1 << pin);
141  } else {
142  GPIO.enable1_w1ts.val = ((uint32_t)1 << (pin - 32));
143  }
144  }
145 
146  if(mode & PULLUP) {
147  pinFunction |= FUN_PU;
148  } else if(mode & PULLDOWN) {
149  pinFunction |= FUN_PD;
150  }
151 
152  pinFunction |= ((uint32_t)2 << FUN_DRV_S);//what are the drivers?
153  pinFunction |= FUN_IE;//input enable but required for output as well?
154 
155  if(mode & (INPUT | OUTPUT)) {
156  pinFunction |= ((uint32_t)2 << MCU_SEL_S);
157  } else if(mode == SPECIAL) {
158  pinFunction |= ((uint32_t)(((pin)==1||(pin)==3)?0:1) << MCU_SEL_S);
159  } else {
160  pinFunction |= ((uint32_t)(mode >> 5) << MCU_SEL_S);
161  }
162 
163  ESP_REG(DR_REG_IO_MUX_BASE + esp32_gpioMux[pin].reg) = pinFunction;
164 
165  if(mode & OPEN_DRAIN) {
166  pinControl = (1 << GPIO_PIN0_PAD_DRIVER_S);
167  }
168 
169  GPIO.pin[pin].val = pinControl;
170  //unlock gpio
171 }

202 static void IRAM_ATTR __onPinInterrupt()
203 {
204  uint32_t gpio_intr_status_l=0;
205  uint32_t gpio_intr_status_h=0;
206 
207  gpio_intr_status_l = GPIO.status;
208  gpio_intr_status_h = GPIO.status1.val;
209  GPIO.status_w1tc = gpio_intr_status_l;//Clear intr for gpio0-gpio31
210  GPIO.status1_w1tc.val = gpio_intr_status_h;//Clear intr for gpio32-39
211 
212  uint8_t pin=0;
213  if(gpio_intr_status_l) {
214  do {
215  if(gpio_intr_status_l & ((uint32_t)1 << pin)) {
216  if(__pinInterruptHandlers[pin].fn) {
217  if(__pinInterruptHandlers[pin].arg){
218  ((voidFuncPtrArg)__pinInterruptHandlers[pin].fn)(__pinInterruptHandlers[pin].arg);
219  } else {
220  __pinInterruptHandlers[pin].fn();
221  }
222  }
223  }
224  } while(++pin<32);
225  }
226  if(gpio_intr_status_h) {
227  pin=32;
228  do {
229  if(gpio_intr_status_h & ((uint32_t)1 << (pin - 32))) {
230  if(__pinInterruptHandlers[pin].fn) {
231  if(__pinInterruptHandlers[pin].arg){
232  ((voidFuncPtrArg)__pinInterruptHandlers[pin].fn)(__pinInterruptHandlers[pin].arg);
233  } else {
234  __pinInterruptHandlers[pin].fn();
235  }
236  }
237  }
238  } while(++pin<GPIO_PIN_COUNT);
239  }
240 }

244 extern void __attachInterruptFunctionalArg(uint8_t pin, voidFuncPtrArg userFunc, void * arg, int intr_type, bool functional)
245 {
246  static bool interrupt_initialized = false;
247 
248  if(!interrupt_initialized) {
249  interrupt_initialized = true;
250  esp_intr_alloc(ETS_GPIO_INTR_SOURCE, (int)ESP_INTR_FLAG_IRAM, __onPinInterrupt, NULL, &gpio_intr_handle);
251  }
252 
253  // if new attach without detach remove old info
254  if (__pinInterruptHandlers[pin].functional && __pinInterruptHandlers[pin].arg)
255  {
256  cleanupFunctional(__pinInterruptHandlers[pin].arg);
257  }
258  __pinInterruptHandlers[pin].fn = (voidFuncPtr)userFunc;
259  __pinInterruptHandlers[pin].arg = arg;
260  __pinInterruptHandlers[pin].functional = functional;
261 
262  esp_intr_disable(gpio_intr_handle);
263  if(esp_intr_get_cpu(gpio_intr_handle)) { //APP_CPU
264  GPIO.pin[pin].int_ena = 1;
265  } else { //PRO_CPU
266  GPIO.pin[pin].int_ena = 4;
267  }
268  GPIO.pin[pin].int_type = intr_type;
269  esp_intr_enable(gpio_intr_handle);
270 }
