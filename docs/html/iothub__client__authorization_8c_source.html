293 char* IoTHubClient_Auth_Get_SasToken(IOTHUB_AUTHORIZATION_HANDLE handle, const char* scope, size_t expire_time)
294 {
295  char* result;

297  if (handle == NULL)

299  LogError("Invalid Parameter handle: %p", handle);
300  result = NULL;
301  }
302  else
303  {
304  if (handle->cred_type == IOTHUB_CREDENTIAL_TYPE_DEVICE_AUTH)
305  {
306 #ifdef USE_DPS_MODULE
307  DEVICE_AUTH_CREDENTIAL_INFO dev_auth_cred;
308  size_t sec_since_epoch;
309

311  {
312  LogError("failure getting seconds from epoch");
