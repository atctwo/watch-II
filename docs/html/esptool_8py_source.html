325  """ Send a request and read the response """
326  def command(self, op=None, data=b"", chk=0, wait_response=True, timeout=DEFAULT_TIMEOUT):
327  saved_timeout = self._port.timeout
328  new_timeout = min(timeout, MAX_TIMEOUT)
329  if new_timeout != saved_timeout:
330  self._port.timeout = new_timeout
331 
332  try:
333  if op is not None:
334  self.trace("command op=0x%02x data len=%s wait_response=%d timeout=%.3f data=%s",
335  op, len(data), 1 if wait_response else 0, timeout, HexFormatter(data))
336  pkt = struct.pack(b'<BBHI', 0x00, op, len(data), chk) + data
337  self.write(pkt)
338 
339  if not wait_response:
340  return
341 
342  # tries to get a response until that response has the
343  # same operation as the request or a retries limit has
344  # exceeded. This is needed for some esp8266s that
345  # reply with more sync responses than expected.
346  for retry in range(100):
347  p = self.read()
348  if len(p) < 8:
349  continue
350  (resp, op_ret, len_ret, val) = struct.unpack('<BBHI', p[:8])
351  if resp != 1:
352  continue
353  data = p[8:]
354  if op is None or op_ret == op:
355  return val, data
356  finally:
357  if new_timeout != saved_timeout:
358  self._port.timeout = saved_timeout
359 
360  raise FatalError("Response doesn't match request")

362  def check_command(self, op_description, op=None, data=b'', chk=0, timeout=DEFAULT_TIMEOUT):
363  """
364  Execute a command with 'command', check the result code and throw an appropriate
365  FatalError if it fails.
366 
367  Returns the "result" of a successful command.
368  """
369  val, data = self.command(op, data, chk, timeout=timeout)
370 
371  # things are a bit weird here, bear with us
372 
373  # the status bytes are the last 2/4 bytes in the data (depending on chip)
374  if len(data) < self.STATUS_BYTES_LENGTH:
375  raise FatalError("Failed to %s. Only got %d byte status response." % (op_description, len(data)))
376  status_bytes = data[-self.STATUS_BYTES_LENGTH:]
377  # we only care if the first one is non-zero. If it is, the second byte is a reason.
378  if byte(status_bytes, 0) != 0:
379  raise FatalError.WithResult('Failed to %s' % op_description, status_bytes)
380 
381  # if we had more data than just the status bytes, return it as the result
382  # (this is used by the md5sum command, maybe other commands?)
383  if len(data) > self.STATUS_BYTES_LENGTH:
384  return data[:-self.STATUS_BYTES_LENGTH]
385  else: # otherwise, just return the 'val' field which comes from the reply header (this is used by read_reg)
386  return val

408  def _connect_attempt(self, mode='default_reset', esp32r0_delay=False):
409  """ A single connection attempt, with esp32r0 workaround options """
410  # esp32r0_delay is a workaround for bugs with the most common auto reset
411  # circuit and Windows, if the EN pin on the dev board does not have
412  # enough capacitance.
413  #
414  # Newer dev boards shouldn't have this problem (higher value capacitor
415  # on the EN pin), and ESP32 revision 1 can't use this workaround as it
416  # relies on a silicon bug.
417  #
418  # Details: https://github.com/espressif/esptool/issues/136
419  last_error = None
420 
421  # If we're doing no_sync, we're likely communicating as a pass through
422  # with an intermediate device to the ESP32
423  if mode == "no_reset_no_sync":
424  return last_error
425 
426  # issue reset-to-bootloader:
427  # RTS = either CH_PD/EN or nRESET (both active low = chip in reset
428  # DTR = GPIO0 (active low = boot to flasher)
429  #
430  # DTR & RTS are active low signals,
431  # ie True = pin @ 0V, False = pin @ VCC.
432  if mode != 'no_reset':
433  self._setDTR(False) # IO0=HIGH
434  self._setRTS(True) # EN=LOW, chip in reset
435  time.sleep(0.1)
436  if esp32r0_delay:
437  # Some chips are more likely to trigger the esp32r0
438  # watchdog reset silicon bug if they're held with EN=LOW
439  # for a longer period
440  time.sleep(1.2)
441  self._setDTR(True) # IO0=LOW
442  self._setRTS(False) # EN=HIGH, chip out of reset
443  if esp32r0_delay:
444  # Sleep longer after reset.
445  # This workaround only works on revision 0 ESP32 chips,
446  # it exploits a silicon bug spurious watchdog reset.
447  time.sleep(0.4) # allow watchdog reset to occur
448  time.sleep(0.05)
449  self._setDTR(False) # IO0=HIGH, done

453  self.flush_input()
454  self._port.flushOutput()
455  self.sync()

794  def run_spiflash_command(self, spiflash_command, data=b"", read_bits=0):
795  """Run an arbitrary SPI flash command.
796 
797  This function uses the "USR_COMMAND" functionality in the ESP
798  SPI hardware, rather than the precanned commands supported by
799  hardware. So the value of spiflash_command is an actual command
800  byte, sent over the wire.
801 
802  After writing command byte, writes 'data' to MOSI and then
803  reads back 'read_bits' of reply on MISO. Result is a number.
804  """
805 
806  # SPI_USR register flags
807  SPI_USR_COMMAND = (1 << 31)
808  SPI_USR_MISO = (1 << 28)
809  SPI_USR_MOSI = (1 << 27)
810 
811  # SPI registers, base address differs ESP32 vs 8266
812  base = self.SPI_REG_BASE
813  SPI_CMD_REG = base + 0x00
814  SPI_USR_REG = base + 0x1C
815  SPI_USR1_REG = base + 0x20
816  SPI_USR2_REG = base + 0x24
817  SPI_W0_REG = base + self.SPI_W0_OFFS
818 
819  # following two registers are ESP32 only
820  if self.SPI_HAS_MOSI_DLEN_REG:
821  # ESP32 has a more sophisticated wayto set up "user" commands
822  def set_data_lengths(mosi_bits, miso_bits):
823  SPI_MOSI_DLEN_REG = base + 0x28
824  SPI_MISO_DLEN_REG = base + 0x2C
825  if mosi_bits > 0:
826  self.write_reg(SPI_MOSI_DLEN_REG, mosi_bits - 1)
827  if miso_bits > 0:
828  self.write_reg(SPI_MISO_DLEN_REG, miso_bits - 1)
829  else:
830 
831  def set_data_lengths(mosi_bits, miso_bits):
832  SPI_DATA_LEN_REG = SPI_USR1_REG
833  SPI_MOSI_BITLEN_S = 17
834  SPI_MISO_BITLEN_S = 8
835  mosi_mask = 0 if (mosi_bits == 0) else (mosi_bits - 1)
836  miso_mask = 0 if (miso_bits == 0) else (miso_bits - 1)
837  self.write_reg(SPI_DATA_LEN_REG,
838  (miso_mask << SPI_MISO_BITLEN_S) | (
839  mosi_mask << SPI_MOSI_BITLEN_S))
840 
841  # SPI peripheral "command" bitmasks for SPI_CMD_REG
842  SPI_CMD_USR = (1 << 18)
843 
844  # shift values
845  SPI_USR2_DLEN_SHIFT = 28
846 
847  if read_bits > 32:
848  raise FatalError("Reading more than 32 bits back from a SPI flash operation is unsupported")
849  if len(data) > 64:
850  raise FatalError("Writing more than 64 bytes of data with one SPI command is unsupported")
851 
852  data_bits = len(data) * 8
853  old_spi_usr = self.read_reg(SPI_USR_REG)
854  old_spi_usr2 = self.read_reg(SPI_USR2_REG)
855  flags = SPI_USR_COMMAND
856  if read_bits > 0:
857  flags |= SPI_USR_MISO
858  if data_bits > 0:
859  flags |= SPI_USR_MOSI
860  set_data_lengths(data_bits, read_bits)
861  self.write_reg(SPI_USR_REG, flags)
862  self.write_reg(SPI_USR2_REG,
863  (7 << SPI_USR2_DLEN_SHIFT) | spiflash_command)
864  if data_bits == 0:
865  self.write_reg(SPI_W0_REG, 0) # clear data register before we read it
866  else:
867  data = pad_to(data, 4, b'\00') # pad to 32-bit multiple
868  words = struct.unpack("I" * (len(data) // 4), data)
869  next_reg = SPI_W0_REG
870  for word in words:
871  self.write_reg(next_reg, word)
872  next_reg += 4
873  self.write_reg(SPI_CMD_REG, SPI_CMD_USR)
874 
875  def wait_done():
876  for _ in range(10):
877  if (self.read_reg(SPI_CMD_REG) & SPI_CMD_USR) == 0:
878  return
879  raise FatalError("SPI command did not complete in time")
880  wait_done()
881 
882  status = self.read_reg(SPI_W0_REG)
883  # restore some SPI controller registers
884  self.write_reg(SPI_USR_REG, old_spi_usr)
885  self.write_reg(SPI_USR2_REG, old_spi_usr2)
886  return status

943  def get_crystal_freq(self):
944  # Figure out the crystal frequency from the UART clock divider
945  # Returns a normalized value in integer MHz (40 or 26 are the only supported values)
946  #
947  # The logic here is:
948  # - We know that our baud rate and the ESP UART baud rate are roughly the same, or we couldn't communicate
949  # - We can read the UART clock divider register to know how the ESP derives this from the APB bus frequency
950  # - Multiplying these two together gives us the bus frequency which is either the crystal frequency (ESP32)
951  # or double the crystal frequency (ESP8266). See the self.XTAL_CLK_DIVIDER parameter for this factor.
952  uart_div = self.read_reg(self.UART_CLKDIV_REG) & self.UART_CLKDIV_MASK
953  est_xtal = (self._port.baudrate * uart_div) / 1e6 / self.XTAL_CLK_DIVIDER
954  norm_xtal = 40 if est_xtal > 33 else 26
955  if abs(norm_xtal - est_xtal) > 1:
956  print("WARNING: Detected crystal freq %.2fMHz is quite different to normalized freq %dMHz. Unsupported crystal in use?" % (est_xtal, norm_xtal))
957  return norm_xtal

1478  def maybe_patch_segment_data(self, f, segment_data):
1479  """If SHA256 digest of the ELF file needs to be inserted into this segment, do so. Returns segment data."""
1480  segment_len = len(segment_data)
