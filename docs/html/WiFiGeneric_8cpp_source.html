64 static void _network_event_task(void * arg){
65  system_prov_event_t *data;
66  for (;;) {
67  if(xQueueReceive(_network_event_queue, &data, portMAX_DELAY) == pdTRUE){
68  if(data->prov_event != NULL){
69  WiFiGenericClass::_eventCallback(arg, data->sys_event, data->prov_event);
70  free(data->sys_event);
71  free(data->prov_event);
72  } else {
73  WiFiGenericClass::_eventCallback(arg, data->sys_event, NULL);
74  }
75  free(data);
76  }
77  }
78  vTaskDelete(NULL);
79  _network_event_task_handle = NULL;

368 esp_err_t WiFiGenericClass::_eventCallback(void *arg, system_event_t *event, wifi_prov_event_t *prov_event)
369 {
370  if(WiFi.isProvEnabled()) {
371  wifi_prov_mgr_event_handler(arg,event);
372  }
373  if(event->event_id < 26) {
374  log_d("Event: %d - %s", event->event_id, system_event_names[event->event_id]);
375  }
376  if(event->event_id == SYSTEM_EVENT_SCAN_DONE) {
377  WiFiScanClass::_scanDone();
378 
379  } else if(event->event_id == SYSTEM_EVENT_STA_START) {
380  WiFiSTAClass::_setStatus(WL_DISCONNECTED);
381  setStatusBits(STA_STARTED_BIT);
382  } else if(event->event_id == SYSTEM_EVENT_STA_STOP) {
383  WiFiSTAClass::_setStatus(WL_NO_SHIELD);
384  clearStatusBits(STA_STARTED_BIT | STA_CONNECTED_BIT | STA_HAS_IP_BIT | STA_HAS_IP6_BIT);
385  } else if(event->event_id == SYSTEM_EVENT_STA_CONNECTED) {
386  WiFiSTAClass::_setStatus(WL_IDLE_STATUS);
387  setStatusBits(STA_CONNECTED_BIT);
388  } else if(event->event_id == SYSTEM_EVENT_STA_DISCONNECTED) {
389  uint8_t reason = event->event_info.disconnected.reason;
390  log_w("Reason: %u - %s", reason, reason2str(reason));
391  if(reason == WIFI_REASON_NO_AP_FOUND) {
392  WiFiSTAClass::_setStatus(WL_NO_SSID_AVAIL);
393  } else if(reason == WIFI_REASON_AUTH_FAIL || reason == WIFI_REASON_ASSOC_FAIL) {
394  WiFiSTAClass::_setStatus(WL_CONNECT_FAILED);
395  } else if(reason == WIFI_REASON_BEACON_TIMEOUT || reason == WIFI_REASON_HANDSHAKE_TIMEOUT) {
396  WiFiSTAClass::_setStatus(WL_CONNECTION_LOST);
397  } else if(reason == WIFI_REASON_AUTH_EXPIRE) {
398 
399  } else {
400  WiFiSTAClass::_setStatus(WL_DISCONNECTED);
401  }
402  clearStatusBits(STA_CONNECTED_BIT | STA_HAS_IP_BIT | STA_HAS_IP6_BIT);
403  if(((reason == WIFI_REASON_AUTH_EXPIRE) ||
404  (reason >= WIFI_REASON_BEACON_TIMEOUT && reason != WIFI_REASON_AUTH_FAIL)) &&
405  WiFi.getAutoReconnect())
406  {
407  WiFi.disconnect();
408  WiFi.begin();
409  }
410  } else if(event->event_id == SYSTEM_EVENT_STA_GOT_IP) {
411 #if ARDUHAL_LOG_LEVEL >= ARDUHAL_LOG_LEVEL_DEBUG
412  uint8_t * ip = (uint8_t *)&(event->event_info.got_ip.ip_info.ip.addr);
413  uint8_t * mask = (uint8_t *)&(event->event_info.got_ip.ip_info.netmask.addr);
414  uint8_t * gw = (uint8_t *)&(event->event_info.got_ip.ip_info.gw.addr);
415  log_d("STA IP: %u.%u.%u.%u, MASK: %u.%u.%u.%u, GW: %u.%u.%u.%u",
416  ip[0], ip[1], ip[2], ip[3],
417  mask[0], mask[1], mask[2], mask[3],
418  gw[0], gw[1], gw[2], gw[3]);
419 #endif
420  WiFiSTAClass::_setStatus(WL_CONNECTED);
421  setStatusBits(STA_HAS_IP_BIT | STA_CONNECTED_BIT);
422  } else if(event->event_id == SYSTEM_EVENT_STA_LOST_IP) {
423  WiFiSTAClass::_setStatus(WL_IDLE_STATUS);
424  clearStatusBits(STA_HAS_IP_BIT);
425 
426  } else if(event->event_id == SYSTEM_EVENT_AP_START) {
427  setStatusBits(AP_STARTED_BIT);
428  } else if(event->event_id == SYSTEM_EVENT_AP_STOP) {
429  clearStatusBits(AP_STARTED_BIT | AP_HAS_CLIENT_BIT);
430  } else if(event->event_id == SYSTEM_EVENT_AP_STACONNECTED) {
431  setStatusBits(AP_HAS_CLIENT_BIT);
432  } else if(event->event_id == SYSTEM_EVENT_AP_STADISCONNECTED) {
433  wifi_sta_list_t clients;
434  if(esp_wifi_ap_get_sta_list(&clients) != ESP_OK || !clients.num){
435  clearStatusBits(AP_HAS_CLIENT_BIT);
436  }
437 
438  } else if(event->event_id == SYSTEM_EVENT_ETH_START) {
439  setStatusBits(ETH_STARTED_BIT);
440  } else if(event->event_id == SYSTEM_EVENT_ETH_STOP) {
441  clearStatusBits(ETH_STARTED_BIT | ETH_CONNECTED_BIT | ETH_HAS_IP_BIT | ETH_HAS_IP6_BIT);
442  } else if(event->event_id == SYSTEM_EVENT_ETH_CONNECTED) {
443  setStatusBits(ETH_CONNECTED_BIT);
444  } else if(event->event_id == SYSTEM_EVENT_ETH_DISCONNECTED) {
445  clearStatusBits(ETH_CONNECTED_BIT | ETH_HAS_IP_BIT | ETH_HAS_IP6_BIT);
446  } else if(event->event_id == SYSTEM_EVENT_ETH_GOT_IP) {
447 #if ARDUHAL_LOG_LEVEL >= ARDUHAL_LOG_LEVEL_DEBUG
448  uint8_t * ip = (uint8_t *)&(event->event_info.got_ip.ip_info.ip.addr);
449  uint8_t * mask = (uint8_t *)&(event->event_info.got_ip.ip_info.netmask.addr);
450  uint8_t * gw = (uint8_t *)&(event->event_info.got_ip.ip_info.gw.addr);
451  log_d("ETH IP: %u.%u.%u.%u, MASK: %u.%u.%u.%u, GW: %u.%u.%u.%u",
452  ip[0], ip[1], ip[2], ip[3],
453  mask[0], mask[1], mask[2], mask[3],
454  gw[0], gw[1], gw[2], gw[3]);
455 #endif
456  setStatusBits(ETH_CONNECTED_BIT | ETH_HAS_IP_BIT);
457 
458  } else if(event->event_id == SYSTEM_EVENT_GOT_IP6) {
459  if(event->event_info.got_ip6.if_index == TCPIP_ADAPTER_IF_AP){
460  setStatusBits(AP_HAS_IP6_BIT);
461  } else if(event->event_info.got_ip6.if_index == TCPIP_ADAPTER_IF_STA){
462  setStatusBits(STA_CONNECTED_BIT | STA_HAS_IP6_BIT);
463  } else if(event->event_info.got_ip6.if_index == TCPIP_ADAPTER_IF_ETH){
464  setStatusBits(ETH_CONNECTED_BIT | ETH_HAS_IP6_BIT);
465  }
466  }
467 
468  for(uint32_t i = 0; i < cbEventList.size(); i++) {
469  WiFiEventCbList_t entry = cbEventList[i];
470  if(entry.cb || entry.fcb || entry.scb) {
471  if(entry.event == (system_event_id_t) event->event_id || entry.event == SYSTEM_EVENT_MAX) {
472  if(entry.cb) {
473  entry.cb((system_event_id_t) event->event_id);
474  } else if(entry.fcb) {
475  entry.fcb((system_event_id_t) event->event_id, (system_event_info_t) event->event_info);
476  } else {
477  entry.scb(event);
478  }
479  }
480  }
481 
482  if(entry.provcb) {
483  entry.provcb(event,prov_event);
484  }
485  }
486  return ESP_OK;
487 }
