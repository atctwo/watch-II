627

760

764 
765  if(!connected()) {
766  return returnError(HTTPC_ERROR_NOT_CONNECTED);

787

793

796  size += len;

798 
799  // data left?

801  int r = writeToStreamDataBlock(stream, len);
802  if(r < 0) {

808

811  _size = size;
812  }
813

816  return returnError(HTTPC_ERROR_STREAM_WRITE);
817  }

819  }

826  }
827 
828  delay(0);
829  }
