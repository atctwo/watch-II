252 bool rmtWrite(rmt_obj_t* rmt, rmt_data_t* data, size_t size)
253 {
254  if (!rmt) {
255  return false;
256  }
257 
258  int channel = rmt->channel;
259  int allocated_size = MAX_DATA_PER_CHANNEL * rmt->buffers;
260 
261  if (size > allocated_size) {
262 
263  int half_tx_nr = MAX_DATA_PER_ITTERATION/2;
264  RMT_MUTEX_LOCK(channel);
265  // setup interrupt handler if not yet installed for half and full tx
266  if (!intr_handle) {
267  esp_intr_alloc(ETS_RMT_INTR_SOURCE, (int)ESP_INTR_FLAG_IRAM, _rmt_isr, NULL, &intr_handle);
268  }
269 
270  rmt->data_size = size - MAX_DATA_PER_ITTERATION;
271  rmt->data_ptr = ((uint32_t*)data) + MAX_DATA_PER_ITTERATION;
272  rmt->intr_mode = E_TX_INTR | E_TXTHR_INTR;
273  rmt->tx_state = E_SET_CONTI | E_FIRST_HALF;
274 
275  // init the tx limit for interruption
276  RMT.tx_lim_ch[channel].limit = half_tx_nr+2;
277  // reset memory pointer
278  RMT.conf_ch[channel].conf1.apb_mem_rst = 1;
279  RMT.conf_ch[channel].conf1.apb_mem_rst = 0;
280  RMT.conf_ch[channel].conf1.mem_rd_rst = 1;
281  RMT.conf_ch[channel].conf1.mem_rd_rst = 0;
282  RMT.conf_ch[channel].conf1.mem_wr_rst = 1;
283  RMT.conf_ch[channel].conf1.mem_wr_rst = 0;
284 
285  // set the tx end mark
286  RMTMEM.chan[channel].data32[MAX_DATA_PER_ITTERATION].val = 0;
287 
288  // clear and enable both Tx completed and half tx event
289  RMT.int_clr.val = _INT_TX_END(channel);
290  RMT.int_clr.val = _INT_THR_EVNT(channel);
291  RMT.int_clr.val = _INT_ERROR(channel);
292 
293  RMT.int_ena.val |= _INT_TX_END(channel);
294  RMT.int_ena.val |= _INT_THR_EVNT(channel);
295  RMT.int_ena.val |= _INT_ERROR(channel);
296 
297  RMT_MUTEX_UNLOCK(channel);
298 
299  // start the transation
300  return _rmtSendOnce(rmt, data, MAX_DATA_PER_ITTERATION, false);
301  } else {
302  // use one-go mode if data fits one buffer
303  return _rmtSendOnce(rmt, data, size, false);
304  }
305 }

394 bool rmtReadAsync(rmt_obj_t* rmt, rmt_data_t* data, size_t size, void* eventFlag, bool waitForData, uint32_t timeout)
395 {
396  if (!rmt) {
397  return false;
398  }
399  int channel = rmt->channel;
400 
401  if (g_rmt_objects[channel].buffers < size/MAX_DATA_PER_CHANNEL) {
402  return false;
403  }
404 
405  if (eventFlag) {
406  xEventGroupClearBits(eventFlag, RMT_FLAGS_ALL);
407  rmt->events = eventFlag;
408  }
409 
410  if (data && size>0) {
411  rmt->data_ptr = (uint32_t*)data;
412  rmt->data_size = size;
413  }
414 
415  RMT_MUTEX_LOCK(channel);
416  rmt->intr_mode = E_RX_INTR;
417 
418  RMT.conf_ch[channel].conf1.mem_owner = 1;
419 
420  RMT.int_clr.val = _INT_RX_END(channel);
421  RMT.int_clr.val = _INT_ERROR(channel);
422 
423  RMT.int_ena.val |= _INT_RX_END(channel);
424  RMT.int_ena.val |= _INT_ERROR(channel);
425 
426  RMT.conf_ch[channel].conf1.mem_wr_rst = 1;
427 
428  RMT.conf_ch[channel].conf1.rx_en = 1;
429  RMT_MUTEX_UNLOCK(channel);
430 
431  // wait for data if requested so
432  if (waitForData && eventFlag) {
433  uint32_t flags = xEventGroupWaitBits(eventFlag, RMT_FLAGS_ALL,
434  pdTRUE /* clear on exit */, pdFALSE /* wait for all bits */, timeout);
435  if (flags & RMT_FLAG_ERROR) {
436  return false;
437  }
438  }
439 
440  return true;
441 }

443 float rmtSetTick(rmt_obj_t* rmt, float tick)
444 {
445  if (!rmt) {
446  return false;
447  }
448  /*
449  divider field span from 1 (smallest), 2, 3, ... , 0xFF, 0x00 (highest)
450  * rmt tick from 1/80M -> 12.5ns (1x) div_cnt = 0x01
451  3.2 us (256x) div_cnt = 0x00
452  * rmt tick for 1 MHz -> 1us (1x) div_cnt = 0x01
453  256us (256x) div_cnt = 0x00
454  */
455  int apb_div = _LIMIT(tick/12.5, 256);
456  int ref_div = _LIMIT(tick/1000, 256);
457 
458  float apb_tick = 12.5 * apb_div;
459  float ref_tick = 1000.0 * ref_div;
460 
461  size_t channel = rmt->channel;
462 
463  if (_ABS(apb_tick - tick) < _ABS(ref_tick - tick)) {
464  RMT.conf_ch[channel].conf0.div_cnt = apb_div & 0xFF;
465  RMT.conf_ch[channel].conf1.ref_always_on = 1;
466  return apb_tick;
467  } else {
468  RMT.conf_ch[channel].conf0.div_cnt = ref_div & 0xFF;
469  RMT.conf_ch[channel].conf1.ref_always_on = 0;
470  return ref_tick;
471  }
472 }

474 rmt_obj_t* rmtInit(int pin, bool tx_not_rx, rmt_reserve_memsize_t memsize)
475 {
476  int buffers = memsize;
477  rmt_obj_t* rmt;
478  size_t i;
479  size_t j;
480 
481  // create common block mutex for protecting allocs from multiple threads
482  if (!g_rmt_block_lock) {
483  g_rmt_block_lock = xSemaphoreCreateMutex();
484  }
485  // lock
486  while (xSemaphoreTake(g_rmt_block_lock, portMAX_DELAY) != pdPASS) {}
487 
488  for (i=0; i<MAX_CHANNELS; i++) {
489  for (j=0; j<buffers && i+j < MAX_CHANNELS; j++) {
490  // if the space is ocupied break and continue on other channel
491  if (g_rmt_objects[i+j].allocated) {
492  i += j; // continue searching from latter channel
493  break;
494  }
495  }
496  if (j == buffers) {
497  // found a space in channel descriptors
498  break;
499  }
500  }
501  if (i == MAX_CHANNELS || i+j > MAX_CHANNELS || j != buffers) {
502  xSemaphoreGive(g_rmt_block_lock);
503  return NULL;
504  }
505  rmt = _rmtAllocate(pin, i, buffers);
506 
507  xSemaphoreGive(g_rmt_block_lock);
508 
509  size_t channel = i;
510 
511 #if !CONFIG_DISABLE_HAL_LOCKS
512  if(g_rmt_objlocks[channel] == NULL) {
513  g_rmt_objlocks[channel] = xSemaphoreCreateMutex();
514  if(g_rmt_objlocks[channel] == NULL) {
515  return NULL;
516  }
517  }
518 #endif
519 
520  RMT_MUTEX_LOCK(channel);
521 
522  rmt->pin = pin;
523  rmt->tx_not_rx = tx_not_rx;
524  rmt->buffers =buffers;
525  rmt->channel = channel;
526  _initPin(pin, channel, tx_not_rx);
527 
528  // Initialize the registers in default mode:
529  // - no carrier, filter
530  // - timebase tick of 1us
531  // - idle threshold set to 0x8000 (max pulse width + 1)
532  RMT.conf_ch[channel].conf0.div_cnt = 1;
533  RMT.conf_ch[channel].conf0.mem_size = buffers;
534  RMT.conf_ch[channel].conf0.carrier_en = 0;
535  RMT.conf_ch[channel].conf0.carrier_out_lv = 0;
536  RMT.conf_ch[channel].conf0.mem_pd = 0;
537 
538  RMT.conf_ch[channel].conf0.idle_thres = 0x80;
539  RMT.conf_ch[channel].conf1.rx_en = 0;
540  RMT.conf_ch[channel].conf1.tx_conti_mode = 0;
541  RMT.conf_ch[channel].conf1.ref_cnt_rst = 0;
542  RMT.conf_ch[channel].conf1.rx_filter_en = 0;
543  RMT.conf_ch[channel].conf1.rx_filter_thres = 0;
544  RMT.conf_ch[channel].conf1.idle_out_lv = 0; // signal level for idle
545  RMT.conf_ch[channel].conf1.idle_out_en = 1; // enable idle
546  RMT.conf_ch[channel].conf1.ref_always_on = 0; // base clock
547  RMT.apb_conf.fifo_mask = 1;
548 
549  if (tx_not_rx) {
550  // RMT.conf_ch[channel].conf1.rx_en = 0;
551  RMT.conf_ch[channel].conf1.mem_owner = 0;
552  RMT.conf_ch[channel].conf1.mem_rd_rst = 1;
553  } else {
554  // RMT.conf_ch[channel].conf1.rx_en = 1;
555  RMT.conf_ch[channel].conf1.mem_owner = 1;
556  RMT.conf_ch[channel].conf1.mem_wr_rst = 1;
557  }
558 
559  // install interrupt if at least one channel is active
560  if (!intr_handle) {
561  esp_intr_alloc(ETS_RMT_INTR_SOURCE, (int)ESP_INTR_FLAG_IRAM, _rmt_isr, NULL, &intr_handle);
562  }
563  RMT_MUTEX_UNLOCK(channel);
564 
565  return rmt;
566 }

571 bool _rmtSendOnce(rmt_obj_t* rmt, rmt_data_t* data, size_t size, bool continuous)
572 {
573  if (!rmt) {
574  return false;
575  }
576  int channel = rmt->channel;
577  RMT.apb_conf.fifo_mask = 1;
578  if (data && size>0) {
579  size_t i;
580  volatile uint32_t* rmt_mem_ptr = &(RMTMEM.chan[channel].data32[0].val);
581  for (i = 0; i < size; i++) {
582  *rmt_mem_ptr++ = data[i].val;
583  }
584  // tx end mark
585  RMTMEM.chan[channel].data32[size].val = 0;
586  }
587 
588  RMT_MUTEX_LOCK(channel);
589  RMT.conf_ch[channel].conf1.tx_conti_mode = continuous;
590  RMT.conf_ch[channel].conf1.mem_rd_rst = 1;
591  RMT.conf_ch[channel].conf1.tx_start = 1;
592  RMT_MUTEX_UNLOCK(channel);
593 
594  return true;
595 }

629 static void IRAM_ATTR _rmt_isr(void* arg)
630 {
631  int intr_val = RMT.int_st.val;
632  size_t ch;
633  for (ch = 0; ch < MAX_CHANNELS; ch++) {
634 
635  if (intr_val & _INT_RX_END(ch)) {
636  // clear the flag
637  RMT.int_clr.val = _INT_RX_END(ch);
638  RMT.int_ena.val &= ~_INT_RX_END(ch);
639

641  if (g_rmt_objects[ch].events) {
642  xEventGroupSetBits(g_rmt_objects[ch].events, RMT_FLAG_RX_DONE);
643  }
644  if (g_rmt_objects[ch].data_ptr && g_rmt_objects[ch].data_size > 0) {
645  size_t i;
646  uint32_t * data = g_rmt_objects[ch].data_ptr;
647  // in case of callback, provide switching between memories
648  if (g_rmt_objects[ch].cb) {
649  if (g_rmt_objects[ch].tx_state & E_FIRST_HALF) {
650  g_rmt_objects[ch].tx_state &= ~E_FIRST_HALF;
651  } else {
652  g_rmt_objects[ch].tx_state |= E_FIRST_HALF;
653  data += MAX_DATA_PER_CHANNEL*(g_rmt_objects[ch].buffers);
654  }
655  }
656  uint32_t *data_received = data;
657  for (i = 0; i < g_rmt_objects[ch].data_size; i++ ) {
658  *data++ = RMTMEM.chan[ch].data32[i].val;
659  }
660  if (g_rmt_objects[ch].cb) {
661  // actually received data ptr
662  (g_rmt_objects[ch].cb)(data_received, _rmt_get_mem_len(ch));
663

665  RMT.conf_ch[ch].conf1.mem_owner = 1;
666  RMT.conf_ch[ch].conf1.mem_wr_rst = 1;
667  RMT.conf_ch[ch].conf1.rx_en = 1;
668  RMT.int_ena.val |= _INT_RX_END(ch);
669  } else {
670  // if not callback provide, expect only one Rx
671  g_rmt_objects[ch].intr_mode &= ~E_RX_INTR;
672  }
673  }
674  } else {
675  // Report error and disable Rx interrupt
676  log_e("Unexpected Rx interrupt!\n"); // TODO: eplace messages with log_X
677  RMT.int_ena.val &= ~_INT_RX_END(ch);
678  }
679 
680 
681  }
682 
683  if (intr_val & _INT_ERROR(ch)) {
684  // clear the flag
685  RMT.int_clr.val = _INT_ERROR(ch);
686  RMT.int_ena.val &= ~_INT_ERROR(ch);
687  // report error
688  log_e("RMT Error %d!\n", ch);
689  if (g_rmt_objects[ch].events) {
690  xEventGroupSetBits(g_rmt_objects[ch].events, RMT_FLAG_ERROR);
691  }
692  // reset memory
693  RMT.conf_ch[ch].conf1.mem_rd_rst = 1;
694  RMT.conf_ch[ch].conf1.mem_rd_rst = 0;
695  RMT.conf_ch[ch].conf1.mem_wr_rst = 1;
696  RMT.conf_ch[ch].conf1.mem_wr_rst = 0;
697  }
698 
699  if (intr_val & _INT_TX_END(ch)) {
700 
701  RMT.int_clr.val = _INT_TX_END(ch);
702  _rmt_tx_mem_second(ch);
703  }
704 
705  if (intr_val & _INT_THR_EVNT(ch)) {
706  // clear the flag
707  RMT.int_clr.val = _INT_THR_EVNT(ch);
708 
709  // initial setup of continuous mode
710  if (g_rmt_objects[ch].tx_state & E_SET_CONTI) {
711  RMT.conf_ch[ch].conf1.tx_conti_mode = 1;
712  g_rmt_objects[ch].intr_mode &= ~E_SET_CONTI;
713  }
714  _rmt_tx_mem_first(ch);
715  }
716  }

719 static void IRAM_ATTR _rmt_tx_mem_second(uint8_t ch)
720 {
721  DEBUG_INTERRUPT_START(4)
722  uint32_t* data = g_rmt_objects[ch].data_ptr;
723  int half_tx_nr = MAX_DATA_PER_ITTERATION/2;
724  int i;
725 
726  RMT.tx_lim_ch[ch].limit = half_tx_nr+2;
727  RMT.int_clr.val = _INT_THR_EVNT(ch);
728  RMT.int_ena.val |= _INT_THR_EVNT(ch);
729 
730  g_rmt_objects[ch].tx_state |= E_FIRST_HALF;
731 
732  if (data) {
733  int remaining_size = g_rmt_objects[ch].data_size;
734  // will the remaining data occupy the entire halfbuffer
735  if (remaining_size > half_tx_nr) {
736  for (i = 0; i < half_tx_nr; i++) {
737  RMTMEM.chan[ch].data32[half_tx_nr+i].val = data[i];
738  }
739  g_rmt_objects[ch].data_size -= half_tx_nr;
740  g_rmt_objects[ch].data_ptr += half_tx_nr;
741  } else {
742  for (i = 0; i < half_tx_nr; i++) {
743  if (i < remaining_size) {
744  RMTMEM.chan[ch].data32[half_tx_nr+i].val = data[i];
745  } else {
746  RMTMEM.chan[ch].data32[half_tx_nr+i].val = 0x000F000F;
747  }
748  }
749  g_rmt_objects[ch].data_ptr = NULL;
750 
751  }
752  } else if ((!(g_rmt_objects[ch].tx_state & E_LAST_DATA)) &&
753  (!(g_rmt_objects[ch].tx_state & E_END_TRANS))) {
754  for (i = 0; i < half_tx_nr; i++) {
755  RMTMEM.chan[ch].data32[half_tx_nr+i].val = 0x000F000F;
756  }
757  RMTMEM.chan[ch].data32[half_tx_nr+i].val = 0;
758  g_rmt_objects[ch].tx_state |= E_LAST_DATA;
759  RMT.conf_ch[ch].conf1.tx_conti_mode = 0;
760  } else {
761  log_d("RMT Tx finished %d!\n", ch);
762  RMT.conf_ch[ch].conf1.tx_conti_mode = 0;
763  RMT.int_ena.val &= ~_INT_TX_END(ch);
764  RMT.int_ena.val &= ~_INT_THR_EVNT(ch);
765  g_rmt_objects[ch].intr_mode = E_NO_INTR;
766  g_rmt_objects[ch].tx_state = E_INACTIVE;
767  }
768  DEBUG_INTERRUPT_END(4);
769 }
