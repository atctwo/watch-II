196 int WiFiClientSecure::read(uint8_t *buf, size_t size)
197 {
198  int peeked = 0;
199  int avail = available();
200  if ((!buf && size) || avail <= 0) {
201  return -1;
202  }
203  if(!size){
204  return 0;
205  }
206  if(_peek >= 0){
207  buf[0] = _peek;
208  _peek = -1;
209  size--;
210  avail--;
211  if(!size || !avail){
212  return 1;
213  }
214  buf++;
215  peeked = 1;
216  }
217 
218  int res = get_ssl_receive(sslclient, buf, size);
219  if (res < 0) {
220  stop();
221  return peeked?peeked:res;
222  }
223  return res + peeked;
224 }
