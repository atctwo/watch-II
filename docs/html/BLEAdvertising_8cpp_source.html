2  * BLEAdvertising.cpp
3  *
4  * This class encapsulates advertising a BLE Server.
5  * Created on: Jun 21, 2017
6  * Author: kolban
7  *
8  * The ESP-IDF provides a framework for BLE advertising. It has determined that there are a common set
9  * of properties that are advertised and has built a data structure that can be populated by the programmer.
10  * This means that the programmer doesn't have to "mess with" the low level construction of a low level
11  * BLE advertising frame. Many of the fields are determined for us while others we can set before starting
12  * to advertise.
13  *
14  * Should we wish to construct our own payload, we can use the BLEAdvertisementData class and call the setters
15  * upon it. Once it is populated, we can then associate it with the advertising and what ever the programmer
16  * set in the data will be advertised.
17  *
18  */

180 void BLEAdvertising::start() {
181  log_v(">> start: customAdvData: %d, customScanResponseData: %d", m_customAdvData, m_customScanResponseData);
182 
183  // We have a vector of service UUIDs that we wish to advertise. In order to use the
184  // ESP-IDF framework, these must be supplied in a contiguous array of their 128bit (16 byte)
185  // representations. If we have 1 or more services to advertise then we allocate enough
186  // storage to host them and then copy them in one at a time into the contiguous storage.
187  int numServices = m_serviceUUIDs.size();
188  if (numServices > 0) {
189  m_advData.service_uuid_len = 16 * numServices;
190  m_advData.p_service_uuid = new uint8_t[m_advData.service_uuid_len];
191  uint8_t* p = m_advData.p_service_uuid;
192  for (int i = 0; i < numServices; i++) {
193  log_d("- advertising service: %s", m_serviceUUIDs[i].toString().c_str());
194  BLEUUID serviceUUID128 = m_serviceUUIDs[i].to128();
195  memcpy(p, serviceUUID128.getNative()->uuid.uuid128, 16);
196  p += 16;
197  }
198  } else {
199  m_advData.service_uuid_len = 0;
200  log_d("- no services advertised");
201  }
202 
203  esp_err_t errRc;
204 
205  if (!m_customAdvData) {
206  // Set the configuration for advertising.
207  m_advData.set_scan_rsp = false;
208  m_advData.include_name = !m_scanResp;
209  m_advData.include_txpower = !m_scanResp;
210  errRc = ::esp_ble_gap_config_adv_data(&m_advData);
211  if (errRc != ESP_OK) {
212  log_e("<< esp_ble_gap_config_adv_data: rc=%d %s", errRc, GeneralUtils::errorToString(errRc));
213  return;
214  }
215  }
216 
217  if (!m_customScanResponseData && m_scanResp) {
218  m_advData.set_scan_rsp = true;
219  m_advData.include_name = m_scanResp;
220  m_advData.include_txpower = m_scanResp;
221  errRc = ::esp_ble_gap_config_adv_data(&m_advData);
222  if (errRc != ESP_OK) {
223  log_e("<< esp_ble_gap_config_adv_data (Scan response): rc=%d %s", errRc, GeneralUtils::errorToString(errRc));
224  return;
225  }
226  }
227 
228  // If we had services to advertise then we previously allocated some storage for them.
229  // Here we release that storage.
230  if (m_advData.service_uuid_len > 0) {
231  delete[] m_advData.p_service_uuid;
232  m_advData.p_service_uuid = nullptr;
233  }
234 
235  // Start advertising.
236  errRc = ::esp_ble_gap_start_advertising(&m_advParams);
237  if (errRc != ESP_OK) {
238  log_e("<< esp_ble_gap_start_advertising: rc=%d %s", errRc, GeneralUtils::errorToString(errRc));
239  return;
240  }
241  log_v("<< start");
242 } // start
