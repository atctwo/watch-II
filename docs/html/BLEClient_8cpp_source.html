23 /*
24  * Design
25  * ------
26  * When we perform a searchService() requests, we are asking the BLE server to return each of the services
27  * that it exposes. For each service, we received an ESP_GATTC_SEARCH_RES_EVT event which contains details
28  * of the exposed service including its UUID.
29  *
30  * The objects we will invent for a BLEClient will be as follows:
31  * * BLERemoteService - A model of a remote service.
32  * * BLERemoteCharacteristic - A model of a remote characteristic
33  * * BLERemoteDescriptor - A model of a remote descriptor.
34  *
35  * Since there is a hierarchical relationship here, we will have the idea that from a BLERemoteService will own
36  * zero or more remote characteristics and a BLERemoteCharacteristic will own zero or more remote BLEDescriptors.
37  *
38  * We will assume that a BLERemoteService contains a map that maps BLEUUIDs to the set of owned characteristics
39  * and that a BLECharacteristic contains a map that maps BLEUUIDs to the set of owned descriptors.
40  *
41  *
42  */

156 
157  log_d("gattClientEventHandler [esp_gatt_if: %d] ... %s",
158  gattc_if, BLEUtils::gattClientEventTypeToString(event).c_str());
159 
160  // Execute handler code based on the type of event received.
161  switch(event) {
162 
163  case ESP_GATTC_SRVC_CHG_EVT:
164  log_i("SERVICE CHANGED");
165  break;
166 
167  case ESP_GATTC_CLOSE_EVT: {
168  // esp_ble_gattc_app_unregister(m_appId);

173  //
174  // ESP_GATTC_DISCONNECT_EVT
175  //
176  // disconnect:
177  // - esp_gatt_status_t status
178  // - uint16_t conn_id
179  // - esp_bd_addr_t remote_bda
180  case ESP_GATTC_DISCONNECT_EVT: {
181  // If we receive a disconnect event, set the class flag that indicates that we are
182  // no longer connected.
183  m_isConnected = false;
184  if (m_pClientCallbacks != nullptr) {
185  m_pClientCallbacks->onDisconnect(this);
186  }
187  esp_ble_gattc_app_unregister(m_gattc_if);
188  m_semaphoreOpenEvt.give(ESP_GATT_IF_NONE);
189  m_semaphoreRssiCmplEvt.give();
190  m_semaphoreSearchCmplEvt.give(1);
191  BLEDevice::removePeerDevice(m_appId, true);
192  break;
193  } // ESP_GATTC_DISCONNECT_EVT
194 
195  //
196  // ESP_GATTC_OPEN_EVT
197  //
198  // open:
199  // - esp_gatt_status_t status
200  // - uint16_t conn_id
201  // - esp_bd_addr_t remote_bda
202  //
203  case ESP_GATTC_OPEN_EVT: {
204  m_conn_id = evtParam->open.conn_id;
205  if (m_pClientCallbacks != nullptr) {
206  m_pClientCallbacks->onConnect(this);
207  }
208  if (evtParam->open.status == ESP_GATT_OK) {
209  m_isConnected = true; // Flag us as connected.
210  }
211  m_semaphoreOpenEvt.give(evtParam->open.status);
212  break;
213  } // ESP_GATTC_OPEN_EVT
214 
215 
216  //
217  // ESP_GATTC_REG_EVT
218  //
219  // reg:
220  // esp_gatt_status_t status
221  // uint16_t app_id
222  //
223  case ESP_GATTC_REG_EVT: {
224  m_gattc_if = gattc_if;
225  m_semaphoreRegEvt.give();
226  break;
227  } // ESP_GATTC_REG_EVT
228 
229  case ESP_GATTC_CFG_MTU_EVT:
230  if(evtParam->cfg_mtu.status != ESP_GATT_OK) {
231  log_e("Config mtu failed");
232  }
233  m_mtu = evtParam->cfg_mtu.mtu;
234  break;
235 
236  case ESP_GATTC_CONNECT_EVT: {
237  BLEDevice::updatePeerDevice(this, true, m_gattc_if);
238  esp_err_t errRc = esp_ble_gattc_send_mtu_req(gattc_if, evtParam->connect.conn_id);

248  } // ESP_GATTC_CONNECT_EVT

270  // log_i("unknown service source");

376 BLERemoteService* BLEClient::getService(BLEUUID uuid) {
377  log_v(">> getService: uuid: %s", uuid.toString().c_str());
378 // Design
379 // ------
380 // We wish to retrieve the service given its UUID. It is possible that we have not yet asked the
381 // device what services it has in which case we have nothing to match against. If we have not
382 // asked the device about its services, then we do that now. Once we get the results we can then
383 // examine the services map to see if it has the service we are looking for.
384  if (!m_haveServices) {
385  getServices();
386  }
387  std::string uuidStr = uuid.toString();
388  for (auto &myPair : m_servicesMap) {
389  if (myPair.first == uuidStr) {
390  log_v("<< getService: found the service with uuid: %s", uuid.toString().c_str());
391  return myPair.second;
392  }
393  } // End of each of the services.
394  log_v("<< getService: not found");
395  return nullptr;
396 } // getService
