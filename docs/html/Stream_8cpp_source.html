108 // reads data from the stream until the target string of the given length is found
109 // search terminated if the terminator string is found
110 // returns true if target string is found, false if terminated or timed out
111 bool Stream::findUntil(const char *target, size_t targetLen, const char *terminator, size_t termLen)
112 {
113  size_t index = 0; // maximum target string length is 64k bytes!
114  size_t termIndex = 0;
115  int c;
116 
117  if(*target == 0) {
118  return true; // return true if target is a null string
119  }
120  while((c = timedRead()) > 0) {
121 
122  if(c != target[index]) {
123  index = 0; // reset index if any char does not match
124  }
125 
126  if(c == target[index]) {
128  if(++index >= targetLen) { // return true if all chars in the target match
129  return true;
130  }
131  }
132 
133  if(termLen > 0 && c == terminator[termIndex]) {
134  if(++termIndex >= termLen) {
135  return false; // return false if terminate string found before target string
136  }
137  } else {
138  termIndex = 0;
139  }
140  }
141  return false;
142 }

154 long Stream::parseInt(char skipChar)
