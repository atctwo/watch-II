59 static bool softap_config_equal(const wifi_config_t& lhs, const wifi_config_t& rhs)
60 {
61  if(strcmp(reinterpret_cast<const char*>(lhs.ap.ssid), reinterpret_cast<const char*>(rhs.ap.ssid)) != 0) {
62  return false;
63  }
64  if(strcmp(reinterpret_cast<const char*>(lhs.ap.password), reinterpret_cast<const char*>(rhs.ap.password)) != 0) {
65  return false;
66  }
67  if(lhs.ap.channel != rhs.ap.channel) {
68  return false;
69  }
70  if(lhs.ap.ssid_hidden != rhs.ap.ssid_hidden) {
71  return false;
72  }
73  if(lhs.ap.max_connection != rhs.ap.max_connection) {
74  return false;
75  }
76  return true;
77 }

92 bool WiFiAPClass::softAP(const char* ssid, const char* passphrase, int channel, int ssid_hidden, int max_connection)
93 {
94 
95  if(!WiFi.enableAP(true)) {
96  // enable AP failed
97  log_e("enable AP first!");
98  return false;
99  }
100 
101  if(!ssid || *ssid == 0) {
102  // fail SSID missing
103  log_e("SSID missing!");
104  return false;
105  }
106 
107  if(passphrase && (strlen(passphrase) > 0 && strlen(passphrase) < 8)) {
108  // fail passphrase too short
109  log_e("passphrase too short!");
110  return false;
111  }
112 
113  esp_wifi_start();
114 
115  wifi_config_t conf;
116  strlcpy(reinterpret_cast<char*>(conf.ap.ssid), ssid, sizeof(conf.ap.ssid));
117  conf.ap.channel = channel;
118  conf.ap.ssid_len = strlen(reinterpret_cast<char *>(conf.ap.ssid));
119  conf.ap.ssid_hidden = ssid_hidden;
120  conf.ap.max_connection = max_connection;
121  conf.ap.beacon_interval = 100;
122 
123  if(!passphrase || strlen(passphrase) == 0) {
124  conf.ap.authmode = WIFI_AUTH_OPEN;
125  *conf.ap.password = 0;
126  } else {
127  conf.ap.authmode = WIFI_AUTH_WPA2_PSK;
128  strlcpy(reinterpret_cast<char*>(conf.ap.password), passphrase, sizeof(conf.ap.password));
129  }
130 
131  wifi_config_t conf_current;
132  esp_wifi_get_config(WIFI_IF_AP, &conf_current);
133  if(!softap_config_equal(conf, conf_current) && esp_wifi_set_config(WIFI_IF_AP, &conf) != ESP_OK) {
134  return false;
135  }
136 
137  return true;
138 }

147 bool WiFiAPClass::softAPConfig(IPAddress local_ip, IPAddress gateway, IPAddress subnet)
148 {
149 
150  if(!WiFi.enableAP(true)) {
151  // enable AP failed
152  return false;
153  }
154 
155  esp_wifi_start();
156 
157  tcpip_adapter_ip_info_t info;
158  info.ip.addr = static_cast<uint32_t>(local_ip);
159  info.gw.addr = static_cast<uint32_t>(gateway);
160  info.netmask.addr = static_cast<uint32_t>(subnet);
161  tcpip_adapter_dhcps_stop(TCPIP_ADAPTER_IF_AP);
162  if(tcpip_adapter_set_ip_info(TCPIP_ADAPTER_IF_AP, &info) == ESP_OK) {
163  dhcps_lease_t lease;
164  lease.enable = true;
165  lease.start_ip.addr = static_cast<uint32_t>(local_ip) + (1 << 24);
166  lease.end_ip.addr = static_cast<uint32_t>(local_ip) + (11 << 24);
167 
168  tcpip_adapter_dhcps_option(
169  (tcpip_adapter_option_mode_t)TCPIP_ADAPTER_OP_SET,
170  (tcpip_adapter_option_id_t)REQUESTED_IP_ADDRESS,
171  (void*)&lease, sizeof(dhcps_lease_t)
172  );
173 
174  return tcpip_adapter_dhcps_start(TCPIP_ADAPTER_IF_AP) == ESP_OK;
175  }
176  return false;
177 }

186 bool WiFiAPClass::softAPdisconnect(bool wifioff)
187 {
188  bool ret;
189  wifi_config_t conf;
190 
191  if(WiFiGenericClass::getMode() == WIFI_MODE_NULL){
192  return ESP_ERR_INVALID_STATE;
193  }
194 
195  *conf.ap.ssid = 0;
196  *conf.ap.password = 0;
197  conf.ap.authmode = WIFI_AUTH_OPEN; // auth must be open if pass=0
198  ret = esp_wifi_set_config(WIFI_IF_AP, &conf) == ESP_OK;
199 
200  if(wifioff) {
201  ret = WiFi.enableAP(false) == ESP_OK;
202  }
203 
204  return ret;
205 }
