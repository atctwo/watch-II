61 static bool _init_gap(const char * name){
62  if(!btStarted() && !btStart()){
63  log_e("btStart failed");
64  return false;
65  }
66  esp_bluedroid_status_t bt_state = esp_bluedroid_get_status();
67  if(bt_state == ESP_BLUEDROID_STATUS_UNINITIALIZED){
68  if (esp_bluedroid_init()) {
69  log_e("esp_bluedroid_init failed");
70  return false;
71  }
72  }
73  if(bt_state != ESP_BLUEDROID_STATUS_ENABLED){
74  if (esp_bluedroid_enable()) {
75  log_e("esp_bluedroid_enable failed");
76  return false;
77  }
78  }
79  if(esp_ble_gap_set_device_name(name)){
80  log_e("gap_set_device_name failed");
81  return false;
82  }
83  if(esp_ble_gap_config_adv_data(&_adv_config)){
84  log_e("gap_config_adv_data failed");
85  return false;
86  }
87  if(esp_ble_gap_register_callback(_on_gap)){
88  log_e("gap_register_callback failed");
89  return false;
90  }
91  return true;
92 }
