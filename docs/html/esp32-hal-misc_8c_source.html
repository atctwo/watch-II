180 void initArduino()
181 {
182 #ifdef CONFIG_APP_ROLLBACK_ENABLE
183  const esp_partition_t *running = esp_ota_get_running_partition();
184  esp_ota_img_states_t ota_state;
185  if (esp_ota_get_state_partition(running, &ota_state) == ESP_OK) {
186  if (ota_state == ESP_OTA_IMG_PENDING_VERIFY) {
187  if (verifyOta()) {
188  esp_ota_mark_app_valid_cancel_rollback();
189  } else {
190  log_e("OTA verification failed! Start rollback to the previous version ...");
191  esp_ota_mark_app_invalid_rollback_and_reboot();
192  }
193  }
194  }
195 #endif
196  //init proper ref tick value for PLL (uncomment if REF_TICK is different than 1MHz)
197  //ESP_REG(APB_CTRL_PLL_TICK_CONF_REG) = APB_CLK_FREQ / REF_CLK_FREQ - 1;
198 #ifdef F_CPU
199  setCpuFrequencyMhz(F_CPU/1000000);
200 #endif
201 #if CONFIG_SPIRAM_SUPPORT
202  psramInit();
203 #endif
204  esp_log_level_set("*", CONFIG_LOG_DEFAULT_LEVEL);
205  esp_err_t err = nvs_flash_init();
206  if(err == ESP_ERR_NVS_NO_FREE_PAGES){
207  const esp_partition_t* partition = esp_partition_find_first(ESP_PARTITION_TYPE_DATA, ESP_PARTITION_SUBTYPE_DATA_NVS, NULL);
208  if (partition != NULL) {
209  err = esp_partition_erase_range(partition, 0, partition->size);
210  if(!err){
211  err = nvs_flash_init();
212  } else {
213  log_e("Failed to format the broken NVS partition!");
214  }
215  }
216  }
217  if(err) {
218  log_e("Failed to initialize NVS! Error: %u", err);
219  }
220 #ifdef CONFIG_BT_ENABLED
221  if(!btInUse()){
222  esp_bt_controller_mem_release(ESP_BT_MODE_BTDM);
223  }
224 #endif
225  init();
226  initVariant();
227 }
