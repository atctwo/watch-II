59 bool ETHClass::begin(uint8_t phy_addr, int power, int mdc, int mdio, eth_phy_type_t type, eth_clock_mode_t clock_mode)
60 {
61  esp_err_t err;
62  if(initialized){
63  err = esp_eth_enable();
64  if(err){
65  log_e("esp_eth_enable error: %d", err);
66  return false;
67  }
68  started = true;
69  return true;
70  }
71  _eth_phy_mdc_pin = mdc;
72  _eth_phy_mdio_pin = mdio;
73  _eth_phy_power_pin = power;
74 
75  if(type == ETH_PHY_LAN8720){
76  eth_config_t config = phy_lan8720_default_ethernet_config;
77  memcpy(&eth_config, &config, sizeof(eth_config_t));
78  } else if(type == ETH_PHY_TLK110){
79  eth_config_t config = phy_tlk110_default_ethernet_config;
80  memcpy(&eth_config, &config, sizeof(eth_config_t));
81  } else {
82  log_e("Bad ETH_PHY type: %u", (uint8_t)type);
83  return false;
84  }
85 
86  eth_config.phy_addr = (eth_phy_base_t)phy_addr;
87  eth_config.clock_mode = clock_mode;
88  eth_config.gpio_config = _eth_phy_config_gpio;
89  eth_config.tcpip_input = tcpip_adapter_eth_input;
90  if(_eth_phy_power_pin >= 0){
91  _eth_phy_power_enable_orig = eth_config.phy_power_enable;
92  eth_config.phy_power_enable = _eth_phy_power_enable;
93  }
94 
95  tcpipInit();
96  err = esp_eth_init(&eth_config);
97  if(!err){
98  initialized = true;
99  err = esp_eth_enable();
100  if(err){
101  log_e("esp_eth_enable error: %d", err);
102  } else {
103  started = true;
104  return true;
105  }
106  } else {
107  log_e("esp_eth_init error: %d", err);
108  }
109  return false;
110 }

112 bool ETHClass::config(IPAddress local_ip, IPAddress gateway, IPAddress subnet, IPAddress dns1, IPAddress dns2)
113 {
114  esp_err_t err = ESP_OK;
115  tcpip_adapter_ip_info_t info;
116 
117  if(local_ip != (uint32_t)0x00000000){
118  info.ip.addr = static_cast<uint32_t>(local_ip);
119  info.gw.addr = static_cast<uint32_t>(gateway);
120  info.netmask.addr = static_cast<uint32_t>(subnet);
121  } else {
122  info.ip.addr = 0;
123  info.gw.addr = 0;
124  info.netmask.addr = 0;
125  }
126 
127  err = tcpip_adapter_dhcpc_stop(TCPIP_ADAPTER_IF_ETH);
128  if(err != ESP_OK && err != ESP_ERR_TCPIP_ADAPTER_DHCP_ALREADY_STOPPED){
129  log_e("DHCP could not be stopped! Error: %d", err);
130  return false;
131  }
132 
133  err = tcpip_adapter_set_ip_info(TCPIP_ADAPTER_IF_ETH, &info);
134  if(err != ERR_OK){
135  log_e("STA IP could not be configured! Error: %d", err);
136  return false;
137 }
138  if(info.ip.addr){
139  staticIP = true;
140  } else {
141  err = tcpip_adapter_dhcpc_start(TCPIP_ADAPTER_IF_ETH);
142  if(err != ESP_OK && err != ESP_ERR_TCPIP_ADAPTER_DHCP_ALREADY_STARTED){
143  log_w("DHCP could not be started! Error: %d", err);
144  return false;
145  }
146  staticIP = false;
147  }
148 
149  ip_addr_t d;
150  d.type = IPADDR_TYPE_V4;
151 
152  if(dns1 != (uint32_t)0x00000000) {
153  // Set DNS1-Server
154  d.u_addr.ip4.addr = static_cast<uint32_t>(dns1);
155  dns_setserver(0, &d);
156  }
157 
158  if(dns2 != (uint32_t)0x00000000) {
159  // Set DNS2-Server
160  d.u_addr.ip4.addr = static_cast<uint32_t>(dns2);
161  dns_setserver(1, &d);
162  }
163 
164  return true;
165 }
