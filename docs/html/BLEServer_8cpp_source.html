142 void BLEServer::handleGATTServerEvent(esp_gatts_cb_event_t event, esp_gatt_if_t gatts_if, esp_ble_gatts_cb_param_t* param) {
143  log_v(">> handleGATTServerEvent: %s",
144  BLEUtils::gattServerEventTypeToString(event).c_str());
145 
146  switch(event) {
147  // ESP_GATTS_ADD_CHAR_EVT - Indicate that a characteristic was added to the service.
148  // add_char:
149  // - esp_gatt_status_t status
150  // - uint16_t attr_handle
151  // - uint16_t service_handle
152  // - esp_bt_uuid_t char_uuid
153  //
154  case ESP_GATTS_ADD_CHAR_EVT: {
155  break;
156  } // ESP_GATTS_ADD_CHAR_EVT
157 
158  case ESP_GATTS_MTU_EVT:
159  updatePeerMTU(param->mtu.conn_id, param->mtu.mtu);
160  break;
161 
162  // ESP_GATTS_CONNECT_EVT
163  // connect:
164  // - uint16_t conn_id
165  // - esp_bd_addr_t remote_bda
166  //
167  case ESP_GATTS_CONNECT_EVT: {
168  m_connId = param->connect.conn_id;
169  addPeerDevice((void*)this, false, m_connId);
170  if (m_pServerCallbacks != nullptr) {
171  m_pServerCallbacks->onConnect(this);
172  m_pServerCallbacks->onConnect(this, param);
173  }
174  m_connectedCount++; // Increment the number of connected devices count.
175  break;
176  } // ESP_GATTS_CONNECT_EVT
177 
178 
179  // ESP_GATTS_CREATE_EVT
180  // Called when a new service is registered as having been created.
181  //
182  // create:
183  // * esp_gatt_status_t status
184  // * uint16_t service_handle
185  // * esp_gatt_srvc_id_t service_id
186  //
187  case ESP_GATTS_CREATE_EVT: {
188  BLEService* pService = m_serviceMap.getByUUID(param->create.service_id.id.uuid, param->create.service_id.id.inst_id); // <--- very big bug for multi services with the same uuid
189  m_serviceMap.setByHandle(param->create.service_handle, pService);
190  m_semaphoreCreateEvt.give();
191  break;
192  } // ESP_GATTS_CREATE_EVT
193 
194 
195  // ESP_GATTS_DISCONNECT_EVT
196  //
197  // disconnect
198  // - uint16_t conn_id
199  // - esp_bd_addr_t remote_bda
200  // - esp_gatt_conn_reason_t reason
201  //
202  // If we receive a disconnect event then invoke the callback for disconnects (if one is present).
203  // we also want to start advertising again.
204  case ESP_GATTS_DISCONNECT_EVT: {
205  m_connectedCount--; // Decrement the number of connected devices count.
206  if (m_pServerCallbacks != nullptr) { // If we have callbacks, call now.
207  m_pServerCallbacks->onDisconnect(this);
208  }
209  startAdvertising(); //- do this with some delay from the loop()
210  removePeerDevice(param->disconnect.conn_id, false);
211  break;
212  } // ESP_GATTS_DISCONNECT_EVT
213 
214 
215  // ESP_GATTS_READ_EVT - A request to read the value of a characteristic has arrived.
216  //
217  // read:
218  // - uint16_t conn_id
219  // - uint32_t trans_id
220  // - esp_bd_addr_t bda
221  // - uint16_t handle
222  // - uint16_t offset
223  // - bool is_long
224  // - bool need_rsp
225  //
226  case ESP_GATTS_READ_EVT: {
227  break;
228  } // ESP_GATTS_READ_EVT
229 
230 
231  // ESP_GATTS_REG_EVT
232  // reg:
233  // - esp_gatt_status_t status
234  // - uint16_t app_id
235  //
236  case ESP_GATTS_REG_EVT: {
237  m_gatts_if = gatts_if;
238  m_semaphoreRegisterAppEvt.give(); // Unlock the mutex waiting for the registration of the app.
239  break;
240  } // ESP_GATTS_REG_EVT
241 
242 
243  // ESP_GATTS_WRITE_EVT - A request to write the value of a characteristic has arrived.
244  //
245  // write:
246  // - uint16_t conn_id
247  // - uint16_t trans_id
248  // - esp_bd_addr_t bda
249  // - uint16_t handle
250  // - uint16_t offset
251  // - bool need_rsp
252  // - bool is_prep
253  // - uint16_t len
254  // - uint8_t* value
255  //
256  case ESP_GATTS_WRITE_EVT: {
257  break;
258  }
259 
260  case ESP_GATTS_OPEN_EVT:
261  m_semaphoreOpenEvt.give(param->open.status);
262  break;
263 
264  default:
265  break;
266  }
267 
268  // Invoke the handler for every Service we have.
269  m_serviceMap.handleGATTServerEvent(event, gatts_if, param);
270 
271  log_v("<< handleGATTServerEvent");
272 } // handleGATTServerEvent

327 bool BLEServer::connect(BLEAddress address) {
328  esp_bd_addr_t addr;
329  memcpy(&addr, address.getNative(), 6);
330  // Perform the open connection request against the target BLE Server.
331  m_semaphoreOpenEvt.take("connect");
332  esp_err_t errRc = ::esp_ble_gatts_open(
333  getGattsIf(),
334  addr, // address
335  1 // direct connection
336  );
337  if (errRc != ESP_OK) {
338  log_e("esp_ble_gattc_open: rc=%d %s", errRc, GeneralUtils::errorToString(errRc));
339  return false;
340  }
341 
342  uint32_t rc = m_semaphoreOpenEvt.wait("connect"); // Wait for the connection to complete.
343  log_v("<< connect(), rc=%d", rc==ESP_GATT_OK);
344  return rc == ESP_GATT_OK;
345 } // connect
