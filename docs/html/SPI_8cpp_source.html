37 void SPIClass::begin(int8_t sck, int8_t miso, int8_t mosi, int8_t ss)
38 {
39  if(_spi) {
40  return;
41  }
42 
43  if(!_div) {
44  _div = spiFrequencyToClockDiv(_freq);
45  }
46 
47  _spi = spiStartBus(_spi_num, _div, SPI_MODE0, SPI_MSBFIRST);
48  if(!_spi) {
49  return;
50  }
51 
52  if(sck == -1 && miso == -1 && mosi == -1 && ss == -1) {
53  _sck = (_spi_num == VSPI) ? SCK : 14;
54  _miso = (_spi_num == VSPI) ? MISO : 12;
55  _mosi = (_spi_num == VSPI) ? MOSI : 13;
56  _ss = (_spi_num == VSPI) ? SS : 15;
57  } else {
58  _sck = sck;
59  _miso = miso;
60  _mosi = mosi;
61  _ss = ss;
62  }
63 
64  spiAttachSCK(_spi, _sck);
65  spiAttachMISO(_spi, _miso);
66  spiAttachMOSI(_spi, _mosi);
67

254 void SPIClass::writePattern(const uint8_t * data, uint8_t size, uint32_t repeat)
255 {
256  if(size > 64) {
257  return; //max Hardware FIFO
258  }
259 
260  uint32_t byte = (size * repeat);
261  uint8_t r = (64 / size);
262  const uint8_t max_bytes_FIFO = r * size; // Max number of whole patterns (in bytes) that can fit into the hardware FIFO
263 
264  while(byte) {
265  if(byte > max_bytes_FIFO) {
266  writePattern_(data, size, r);
267  byte -= max_bytes_FIFO;
268  } else {
269  writePattern_(data, size, (byte / size));
270  byte = 0;
271  }
272  }
273 }

275 void SPIClass::writePattern_(const uint8_t * data, uint8_t size, uint8_t repeat)
276 {
277  uint8_t bytes = (size * repeat);
278  uint8_t buffer[64];
279  uint8_t * bufferPtr = &buffer[0];
280  const uint8_t * dataPtr;
281  uint8_t dataSize = bytes;
282  for(uint8_t i = 0; i < repeat; i++) {
283  dataSize = size;
284  dataPtr = data;
285  while(dataSize--) {
286  *bufferPtr = *dataPtr;
287  dataPtr++;
288  bufferPtr++;
289  }
290  }
291 
292  writeBytes(&buffer[0], bytes);
293 }
