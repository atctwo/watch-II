413 spi_t * spiStartBus(uint8_t spi_num, uint32_t clockDiv, uint8_t dataMode, uint8_t bitOrder)
414 {
415  if(spi_num > 3){
416  return NULL;
417  }
418 
419  spi_t * spi = &_spi_bus_array[spi_num];
420 
421 #if !CONFIG_DISABLE_HAL_LOCKS
422  if(spi->lock == NULL){
423  spi->lock = xSemaphoreCreateMutex();
424  if(spi->lock == NULL) {
425  return NULL;
426  }
427  }
428 #endif
429 
430  if(spi_num == HSPI) {
431  DPORT_SET_PERI_REG_MASK(DPORT_PERIP_CLK_EN_REG, DPORT_SPI_CLK_EN);
432  DPORT_CLEAR_PERI_REG_MASK(DPORT_PERIP_RST_EN_REG, DPORT_SPI_RST);
433  } else if(spi_num == VSPI) {
434  DPORT_SET_PERI_REG_MASK(DPORT_PERIP_CLK_EN_REG, DPORT_SPI_CLK_EN_2);
435  DPORT_CLEAR_PERI_REG_MASK(DPORT_PERIP_RST_EN_REG, DPORT_SPI_RST_2);
436  } else {
437  DPORT_SET_PERI_REG_MASK(DPORT_PERIP_CLK_EN_REG, DPORT_SPI_CLK_EN_1);
438  DPORT_CLEAR_PERI_REG_MASK(DPORT_PERIP_RST_EN_REG, DPORT_SPI_RST_1);
439  }
440 
441  SPI_MUTEX_LOCK();
442  spiInitBus(spi);
443  spi->dev->user.usr_mosi = 1;
444  spi->dev->user.usr_miso = 1;
445  spi->dev->user.doutdin = 1;
446 
447  int i;
448  for(i=0; i<16; i++) {
449  spi->dev->data_buf[i] = 0x00000000;
450  }
451  SPI_MUTEX_UNLOCK();
452 
453  spiSetDataMode(spi, dataMode);
454  spiSetBitOrder(spi, bitOrder);
455  spiSetClockDiv(spi, clockDiv);
456 
457  addApbChangeCallback(spi, _on_apb_change);
458  return spi;
459 }
