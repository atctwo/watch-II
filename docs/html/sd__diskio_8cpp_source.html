97 char sdCommand(uint8_t pdrv, char cmd, unsigned int arg, unsigned int* resp)
98 {
99  char token;
100  ardu_sdcard_t * card = s_cards[pdrv];
101 
102  for (int f = 0; f < 3; f++) {
103  if (cmd == SEND_NUM_WR_BLOCKS || cmd == SET_WR_BLK_ERASE_COUNT || cmd == APP_OP_COND || cmd == APP_CLR_CARD_DETECT) {
104  token = sdCommand(pdrv, APP_CMD, 0, NULL);
105  sdDeselectCard(pdrv);
106  if (token > 1) {
107  return token;
108  }
109  if(!sdSelectCard(pdrv)) {
110  return 0xFF;
111  }
112  }
113 
114  char cmdPacket[7];
115  cmdPacket[0] = cmd | 0x40;
116  cmdPacket[1] = arg >> 24;
117  cmdPacket[2] = arg >> 16;
118  cmdPacket[3] = arg >> 8;
119  cmdPacket[4] = arg;
120  if(card->supports_crc || cmd == GO_IDLE_STATE || cmd == SEND_IF_COND) {
121  cmdPacket[5] = (CRC7(cmdPacket, 5) << 1) | 0x01;
122  } else {
123  cmdPacket[5] = 0x01;
124  }
125  cmdPacket[6] = 0xFF;
126 
127  card->spi->writeBytes((uint8_t*)cmdPacket, (cmd == STOP_TRANSMISSION)?7:6);
128 
129  for (int i = 0; i < 9; i++) {
130  token = card->spi->transfer(0xFF);
131  if (!(token & 0x80)) {
132  break;
133  }
134  }
135 
136  if (token == 0xFF) {
137  log_w("no token received");
138  sdDeselectCard(pdrv);
139  delay(100);
140  sdSelectCard(pdrv);
141  continue;
142  } else if (token & 0x08) {
143  log_w("crc error");
144  sdDeselectCard(pdrv);
145  delay(100);
146  sdSelectCard(pdrv);
147  continue;
148  } else if (token > 1) {
149  log_w("token error [%u] 0x%x", cmd, token);
150  break;
151  }
152 
153  if (cmd == SEND_STATUS && resp) {
154  *resp = card->spi->transfer(0xFF);
155  } else if ((cmd == SEND_IF_COND || cmd == READ_OCR) && resp) {
156  *resp = card->spi->transfer32(0xFFFFFFFF);
157  }
158 
159  break;
160  }
161 
162  return token;
163 }

165 bool sdReadBytes(uint8_t pdrv, char* buffer, int length)
166 {
167  char token;
168  unsigned short crc;
169  ardu_sdcard_t * card = s_cards[pdrv];
170 
171  uint32_t start = millis();
172  do {
173  token = card->spi->transfer(0xFF);
174  } while (token == 0xFF && (millis() - start) < 500);
175 
176  if (token != 0xFE) {
177  return false;
178  }
179 
180  card->spi->transferBytes(NULL, (uint8_t*)buffer, length);
181  crc = card->spi->transfer16(0xFFFF);
182  return (!card->supports_crc || crc == CRC16(buffer, length));
183 }

185 char sdWriteBytes(uint8_t pdrv, const char* buffer, char token)
186 {
187  ardu_sdcard_t * card = s_cards[pdrv];
188  unsigned short crc = (card->supports_crc)?CRC16(buffer, 512):0xFFFF;
189  if (!sdWait(pdrv, 500)) {
190  return false;
191  }
192 
193  card->spi->write(token);
194  card->spi->writeBytes((uint8_t*)buffer, 512);
195  card->spi->write16(crc);
196  return (card->spi->transfer(0xFF) & 0x1F);

233 bool sdReadSectors(uint8_t pdrv, char* buffer, unsigned long long sector, int count)
234 {
235  for (int f = 0; f < 3;) {
236  if(!sdSelectCard(pdrv)) {
237  break;
238  }
239 
240  if (!sdCommand(pdrv, READ_BLOCK_MULTIPLE, (s_cards[pdrv]->type == CARD_SDHC) ? sector : sector << 9, NULL)) {
241  do {
242  if (!sdReadBytes(pdrv, buffer, 512)) {
243  f++;
244  break;
245  }
246 
247  sector++;
248  buffer += 512;
249  f = 0;
250  } while (--count);
251 
252  if (sdCommand(pdrv, STOP_TRANSMISSION, 0, NULL)) {
253  log_e("command failed");
254  break;
255  }
256 
257  sdDeselectCard(pdrv);
258  if (count == 0) {
259  return true;
260  }
261  } else {
262  break;
263  }
264  }
265  sdDeselectCard(pdrv);
266  return false;

269 bool sdWriteSector(uint8_t pdrv, const char* buffer, unsigned long long sector)
270 {
271  for (int f = 0; f < 3; f++) {
272  if(!sdSelectCard(pdrv)) {
273  break;
274  }
275  if (!sdCommand(pdrv, WRITE_BLOCK_SINGLE, (s_cards[pdrv]->type == CARD_SDHC) ? sector : sector << 9, NULL)) {
276  char token = sdWriteBytes(pdrv, buffer, 0xFE);
277  sdDeselectCard(pdrv);
278 
279  if (token == 0x0A) {
280  continue;
281  } else if (token == 0x0C) {
282  return false;
283  }
284 
285  unsigned int resp;
286  if (sdTransaction(pdrv, SEND_STATUS, 0, &resp) || resp) {
287  return false;
288  }
289  return true;
290  } else {
291  break;
292  }
293  }
294  sdDeselectCard(pdrv);
295  return false;

298 bool sdWriteSectors(uint8_t pdrv, const char* buffer, unsigned long long sector, int count)
299 {
300  char token;
301  const char* currentBuffer = buffer;
302  unsigned long long currentSector = sector;
303  int currentCount = count;
304  ardu_sdcard_t * card = s_cards[pdrv];
305 
306  for (int f = 0; f < 3;) {
307  if (card->type != CARD_MMC) {
308  if (sdTransaction(pdrv, SET_WR_BLK_ERASE_COUNT, currentCount, NULL)) {
309  break;
310  }
311  }
312 
313  if(!sdSelectCard(pdrv)) {
314  break;
315  }
316 
317  if (!sdCommand(pdrv, WRITE_BLOCK_MULTIPLE, (card->type == CARD_SDHC) ? currentSector : currentSector << 9, NULL)) {
318  do {
319  token = sdWriteBytes(pdrv, currentBuffer, 0xFC);
320  if (token != 0x05) {
321  f++;
322  break;
323  }
324  currentBuffer += 512;
325  f = 0;
326  } while (--currentCount);
327 
328  if (!sdWait(pdrv, 500)) {
329  break;
330  }
331 
332  if (currentCount == 0) {
333  sdStop(pdrv);
334  sdDeselectCard(pdrv);
335 
336  unsigned int resp;
337  if (sdTransaction(pdrv, SEND_STATUS, 0, &resp) || resp) {
338  return false;
339  }
340  return true;
341  } else {
342  if (sdCommand(pdrv, STOP_TRANSMISSION, 0, NULL)) {
343  break;
344  }
345 
346  sdDeselectCard(pdrv);
347 
348  if (token == 0x0A) {
349  unsigned int writtenBlocks = 0;
350  if (card->type != CARD_MMC && sdSelectCard(pdrv)) {
351  if (!sdCommand(pdrv, SEND_NUM_WR_BLOCKS, 0, NULL)) {
352  char acmdData[4];
353  if (sdReadBytes(pdrv, acmdData, 4)) {
354  writtenBlocks = acmdData[0] << 24;
355  writtenBlocks |= acmdData[1] << 16;
356  writtenBlocks |= acmdData[2] << 8;
357  writtenBlocks |= acmdData[3];
358  }
359  }
360  sdDeselectCard(pdrv);
361  }
362  currentBuffer = buffer + (writtenBlocks << 9);
363  currentSector = sector + writtenBlocks;
364  currentCount = count - writtenBlocks;
365  continue;
366  } else {
367  return false;
368  }
369  }
370  } else {
371  break;
372  }
373  }
374  sdDeselectCard(pdrv);
375  return false;
376 }

378 unsigned long sdGetSectorsCount(uint8_t pdrv)
379 {
380  for (int f = 0; f < 3; f++) {
381  if(!sdSelectCard(pdrv)) {
382  break;
383  }
384 
385  if (!sdCommand(pdrv, SEND_CSD, 0, NULL)) {
386  char csd[16];
387  bool success = sdReadBytes(pdrv, csd, 16);
388  sdDeselectCard(pdrv);
389  if (success) {
390  if ((csd[0] >> 6) == 0x01) {
391  unsigned long size = (
392  ((unsigned long)(csd[7] & 0x3F) << 16)
393  | ((unsigned long)csd[8] << 8)
394  | csd[9]
395  ) + 1;
396  return size << 10;
397  }
398  unsigned long size = (
399  ((unsigned long)(csd[6] & 0x03) << 10)
400  | ((unsigned long)csd[7] << 2)
401  | ((csd[8] & 0xC0) >> 6)
402  ) + 1;
403  size <<= ((
404  ((csd[9] & 0x03) << 1)
405  | ((csd[10] & 0x80) >> 7)
406  ) + 2);
407  size <<= (csd[5] & 0x0F);
408  return size >> 9;
409  }
410  } else {
411  break;
412  }
413  }
414 
415  sdDeselectCard(pdrv);
416  return 0;
417 }

452 DSTATUS ff_sd_initialize(uint8_t pdrv)
453 {
454  char token;
455  unsigned int resp;
456  unsigned int start;
457  ardu_sdcard_t * card = s_cards[pdrv];
458 
459  if (!(card->status & STA_NOINIT)) {
460  return card->status;
461  }
462 
463  AcquireSPI card_locked(card, 400000);
464 
465  digitalWrite(card->ssPin, HIGH);
466  for (uint8_t i = 0; i < 20; i++) {
467  card->spi->transfer(0XFF);
468  }
469 
470  if (sdTransaction(pdrv, GO_IDLE_STATE, 0, NULL) != 1) {
471  log_w("GO_IDLE_STATE failed");
472  goto unknown_card;
473  }
474 
475  token = sdTransaction(pdrv, CRC_ON_OFF, 1, NULL);
476  if (token == 0x5) {
477  //old card maybe
478  card->supports_crc = false;
479  } else if (token != 1) {
480  log_w("CRC_ON_OFF failed: %u", token);
481  goto unknown_card;
482  }
483 
484  if (sdTransaction(pdrv, SEND_IF_COND, 0x1AA, &resp) == 1) {
485  if ((resp & 0xFFF) != 0x1AA) {
486  log_w("SEND_IF_COND failed: %03X", resp & 0xFFF);
487  goto unknown_card;
488  }
489 
490  if (sdTransaction(pdrv, READ_OCR, 0, &resp) != 1 || !(resp & (1 << 20))) {
491  log_w("READ_OCR failed: %X", resp);
492  goto unknown_card;
493  }
494 
495  start = millis();
496  do {
497  token = sdTransaction(pdrv, APP_OP_COND, 0x40100000, NULL);
498  } while (token == 1 && (millis() - start) < 1000);
499 
500  if (token) {
501  log_w("APP_OP_COND failed: %u", token);
502  goto unknown_card;
503  }
504 
505  if (!sdTransaction(pdrv, READ_OCR, 0, &resp)) {
506  if (resp & (1 << 30)) {
507  card->type = CARD_SDHC;
508  } else {
509  card->type = CARD_SD;
510  }
511  } else {
512  log_w("READ_OCR failed: %X", resp);
513  goto unknown_card;
514  }
515  } else {
516  if (sdTransaction(pdrv, READ_OCR, 0, &resp) != 1 || !(resp & (1 << 20))) {
517  log_w("READ_OCR failed: %X", resp);
518  goto unknown_card;
519  }
520 
521  start = millis();
522  do {
523  token = sdTransaction(pdrv, APP_OP_COND, 0x100000, NULL);
524  } while (token == 0x01 && (millis() - start) < 1000);
525 
526  if (!token) {
527  card->type = CARD_SD;
528  } else {
529  start = millis();
530  do {
531  token = sdTransaction(pdrv, SEND_OP_COND, 0x100000, NULL);
532  } while (token != 0x00 && (millis() - start) < 1000);
533 
534  if (token == 0x00) {
535  card->type = CARD_MMC;
536  } else {
537  log_w("SEND_OP_COND failed: %u", token);
538  goto unknown_card;
539  }
540  }
541  }
542 
543  if (card->type != CARD_MMC) {
544  if (sdTransaction(pdrv, APP_CLR_CARD_DETECT, 0, NULL)) {
545  log_w("APP_CLR_CARD_DETECT failed");
546  goto unknown_card;
547  }
548  }
549 
550  if (card->type != CARD_SDHC) {
551  if (sdTransaction(pdrv, SET_BLOCKLEN, 512, NULL) != 0x00) {
552  log_w("SET_BLOCKLEN failed");
553  goto unknown_card;
554  }
555  }
556 
557  card->sectors = sdGetSectorsCount(pdrv);
558 
559  if (card->frequency > 25000000) {
560  card->frequency = 25000000;
561  }
562 
563  card->status &= ~STA_NOINIT;
564  return card->status;
565 
566 unknown_card:
567  card->type = CARD_UNKNOWN;
568  return card->status;
569 }

584  AcquireSPI lock(card);

606  AcquireSPI lock(card);
607

615 DRESULT ff_sd_ioctl(uint8_t pdrv, uint8_t cmd, void* buff)
616 {
617  switch(cmd) {
618  case CTRL_SYNC:
619  {
620  AcquireSPI lock(s_cards[pdrv]);
621  if (sdSelectCard(pdrv)) {
622  sdDeselectCard(pdrv);
623  return RES_OK;
624  }
625  }
626  return RES_ERROR;
627  case GET_SECTOR_COUNT:
628  *((unsigned long*) buff) = s_cards[pdrv]->sectors;
629  return RES_OK;
630  case GET_SECTOR_SIZE:
631  *((WORD*) buff) = 512;
632  return RES_OK;
633  case GET_BLOCK_SIZE:
634  *((uint32_t*)buff) = 1;
635  return RES_OK;
636  }
637  return RES_PARERR;
