95 template <typename Clock, typename Duration>
96 std::size_t io_context::run_one_until(
97  const chrono::time_point<Clock, Duration>& abs_time)
98 {
99  typename Clock::time_point now = Clock::now();
100  while (now < abs_time)
101  {
102  typename Clock::duration rel_time = abs_time - now;
103  if (rel_time > chrono::seconds(1))
104  rel_time = chrono::seconds(1);
105 
106  asio::error_code ec;
107  std::size_t s = impl_.wait_one(
108  static_cast<long>(chrono::duration_cast<
109  chrono::microseconds>(rel_time).count()), ec);
110  asio::detail::throw_error(ec);
111 
112  if (s || impl_.stopped())
113  return s;
114 
115  now = Clock::now();
116  }
117 
118  return 0;
119 }

229  ASIO_MOVE_ARG(Function) f, const Allocator& a) const
230 {

232 
233  // Invoke immediately if we are already inside the thread pool.
234  if (io_context_.impl_.can_dispatch())
235  {
236  // Make a local, non-const copy of the function.
237  function_type tmp(ASIO_MOVE_CAST(Function)(f));
238 
239  detail::fenced_block b(detail::fenced_block::full);
240  asio_handler_invoke_helpers::invoke(tmp, tmp);
241  return;
242  }

246  typename op::ptr p = { detail::addressof(a), op::ptr::allocate(a), 0 };
247  p.p = new (p.v) op(ASIO_MOVE_CAST(Function)(f), a);
