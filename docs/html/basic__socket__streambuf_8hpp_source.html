431  {

433  ec_ = asio::error::operation_not_supported;
434  return traits_type::eof();
435 #else // defined(ASIO_WINDOWS_RUNTIME)

437  return traits_type::eof();

439  for (;;)

441  // Check if we are past the expiry time.
442  if (traits_helper::less_than(expiry_time_, traits_helper::now()))
443  {

449  if (!socket().native_non_blocking())

484  int_type overflow(int_type c)
485  {
486 #if defined(ASIO_WINDOWS_RUNTIME)
487  ec_ = asio::error::operation_not_supported;
488  return traits_type::eof();
489 #else // defined(ASIO_WINDOWS_RUNTIME)
490  char_type ch = traits_type::to_char_type(c);
491 
492  // Determine what needs to be sent.
493  const_buffer output_buffer;
494  if (put_buffer_.empty())
495  {
496  if (traits_type::eq_int_type(c, traits_type::eof()))
497  return traits_type::not_eof(c); // Nothing to do.
498  output_buffer = asio::buffer(&ch, sizeof(char_type));
499  }
500  else
501  {
502  output_buffer = asio::buffer(pbase(),
503  (pptr() - pbase()) * sizeof(char_type));
504  }
505 
506  while (output_buffer.size() > 0)
507  {
508  // Check if we are past the expiry time.
509  if (traits_helper::less_than(expiry_time_, traits_helper::now()))
510  {
511  ec_ = asio::error::timed_out;
512  return traits_type::eof();
513  }
514 
515  // Try to complete the operation without blocking.
516  if (!socket().native_non_blocking())
517  socket().native_non_blocking(true, ec_);
518  detail::buffer_sequence_adapter<
519  const_buffer, const_buffer> bufs(output_buffer);
520  detail::signed_size_type bytes = detail::socket_ops::send(
521  socket().native_handle(), bufs.buffers(), bufs.count(), 0, ec_);
522 
523  // Check if operation succeeded.
524  if (bytes > 0)
525  {
526  output_buffer += static_cast<std::size_t>(bytes);
527  continue;
528  }
529 
530  // Operation failed.
531  if (ec_ != asio::error::would_block
532  && ec_ != asio::error::try_again)
533  return traits_type::eof();
534 
535  // Wait for socket to become ready.
536  if (detail::socket_ops::poll_write(
537  socket().native_handle(), 0, timeout(), ec_) < 0)
538  return traits_type::eof();
539  }
540 
541  if (!put_buffer_.empty())
542  {
543  setp(&put_buffer_[0], &put_buffer_[0] + put_buffer_.size());
544 
545  // If the new character is eof then our work here is done.
546  if (traits_type::eq_int_type(c, traits_type::eof()))
547  return traits_type::not_eof(c);
548 
549  // Add the new character to the output buffer.
550  *pptr() = ch;
551  pbump(1);
552  }
553 
554  return c;
555 #endif // defined(ASIO_WINDOWS_RUNTIME)

612  template <typename EndpointIterator>
613  void connect_to_endpoints(EndpointIterator begin, EndpointIterator end)
614  {
615 #if defined(ASIO_WINDOWS_RUNTIME)
616  ec_ = asio::error::operation_not_supported;
617 #else // defined(ASIO_WINDOWS_RUNTIME)
618  if (ec_)
619  return;
620 
621  ec_ = asio::error::not_found;
622  for (EndpointIterator i = begin; i != end; ++i)
623  {
624  // Check if we are past the expiry time.
625  if (traits_helper::less_than(expiry_time_, traits_helper::now()))
626  {
627  ec_ = asio::error::timed_out;
628  return;
629  }
630 
631  // Close and reopen the socket.
632  typename Protocol::endpoint ep(*i);
633  socket().close(ec_);
634  socket().open(ep.protocol(), ec_);
635  if (ec_)
636  continue;
637 
638  // Try to complete the operation without blocking.
639  if (!socket().native_non_blocking())
640  socket().native_non_blocking(true, ec_);
641  detail::socket_ops::connect(socket().native_handle(),
642  ep.data(), ep.size(), ec_);
643 
644  // Check if operation succeeded.
645  if (!ec_)
646  return;
647 
648  // Operation failed.
649  if (ec_ != asio::error::in_progress
650  && ec_ != asio::error::would_block)
651  continue;
652 
653  // Wait for socket to become ready.
654  if (detail::socket_ops::poll_connect(
655  socket().native_handle(), timeout(), ec_) < 0)
656  continue;
657 
658  // Get the error code from the connect operation.
659  int connect_error = 0;
660  size_t connect_error_len = sizeof(connect_error);
661  if (detail::socket_ops::getsockopt(socket().native_handle(), 0,
662  SOL_SOCKET, SO_ERROR, &connect_error, &connect_error_len, ec_)
663  == detail::socket_error_retval)
664  return;
665 
666  // Check the result of the connect operation.
667  ec_ = asio::error_code(connect_error,
668  asio::error::get_system_category());
669  if (!ec_)
670  return;
671  }
672 #endif // defined(ASIO_WINDOWS_RUNTIME)
