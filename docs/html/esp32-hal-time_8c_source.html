19 static void setTimeZone(long offset, int daylight)
20 {
21  char cst[17] = {0};
22  char cdt[17] = "DST";
23  char tz[33] = {0};
24 
25  if(offset % 3600){
26  sprintf(cst, "UTC%ld:%02u:%02u", offset / 3600, abs((offset % 3600) / 60), abs(offset % 60));
27  } else {
28  sprintf(cst, "UTC%ld", offset / 3600);
29  }
30  if(daylight != 3600){
31  long tz_dst = offset - daylight;
32  if(tz_dst % 3600){
33  sprintf(cdt, "DST%ld:%02u:%02u", tz_dst / 3600, abs((tz_dst % 3600) / 60), abs(tz_dst % 60));
34  } else {
35  sprintf(cdt, "DST%ld", tz_dst / 3600);
36  }
37  }
38  sprintf(tz, "%s%s", cst, cdt);
39  setenv("TZ", tz, 1);
40  tzset();
41 }
