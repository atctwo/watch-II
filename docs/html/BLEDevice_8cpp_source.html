598  log_d("update conn_id: %d, GATT role: %s", conn_id, _client? "client":"server");
599  std::map<uint16_t, conn_status_t>::iterator it = m_connectedClientsMap.find(ESP_GATT_IF_NONE);
600  if (it != m_connectedClientsMap.end()) {
601  std::swap(m_connectedClientsMap[conn_id], it->second);
