39  template <typename SyncRandomAccessWriteDevice, typename ConstBufferSequence,
40  typename ConstBufferIterator, typename CompletionCondition>
41  std::size_t write_at_buffer_sequence(SyncRandomAccessWriteDevice& d,
42  uint64_t offset, const ConstBufferSequence& buffers,
43  const ConstBufferIterator&, CompletionCondition completion_condition,
44  asio::error_code& ec)
45  {
46  ec = asio::error_code();
47  asio::detail::consuming_buffers<const_buffer,
48  ConstBufferSequence, ConstBufferIterator> tmp(buffers);
49  while (!tmp.empty())
50  {
51  if (std::size_t max_size = detail::adapt_completion_condition_result(
52  completion_condition(ec, tmp.total_consumed())))
53  {
54  tmp.consume(d.write_some_at(offset + tmp.total_consumed(),
55  tmp.prepare(max_size), ec));
56  }
57  else
58  break;
59  }
60  return tmp.total_consumed();;
61  }

199  void operator()(const asio::error_code& ec,
200  std::size_t bytes_transferred, int start = 0)
201  {
202  std::size_t max_size;
203  switch (start_ = start)
204  {
205  case 1:
206  max_size = this->check_for_completion(ec, buffers_.total_consumed());
207  do
208  {
209  device_.async_write_some_at(
210  offset_ + buffers_.total_consumed(), buffers_.prepare(max_size),
211  ASIO_MOVE_CAST(write_at_op)(*this));
212  return; default:
213  buffers_.consume(bytes_transferred);
214  if ((!ec && bytes_transferred == 0) || buffers_.empty())
215  break;
216  max_size = this->check_for_completion(ec, buffers_.total_consumed());
217  } while (max_size > 0);
218 
219  handler_(ec, buffers_.total_consumed());
220  }
221  }
