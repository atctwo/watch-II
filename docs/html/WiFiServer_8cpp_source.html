65 void WiFiServer::begin(uint16_t port){
66  if(_listening)
67  return;
68  if(port){
69  _port = port;
70  }

73  if (sockfd < 0)
74  return;
75  server.sin_family = AF_INET;
76  server.sin_addr.s_addr = INADDR_ANY;
77  server.sin_port = htons(_port);
78  if(bind(sockfd, (struct sockaddr *)&server, sizeof(server)) < 0)
79  return;
80  if(listen(sockfd , _max_clients) < 0)
81  return;
