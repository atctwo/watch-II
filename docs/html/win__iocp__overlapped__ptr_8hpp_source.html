74  // Reset to contain the specified handler, freeing any current OVERLAPPED
75  // object.
76  template <typename Handler>
77  void reset(asio::io_context& io_context, Handler handler)
78  {
79  typedef win_iocp_overlapped_op<Handler> op;
80  typename op::ptr p = { asio::detail::addressof(handler),
81  op::ptr::allocate(handler), 0 };
82  p.p = new (p.v) op(handler);
83 
84  ASIO_HANDLER_CREATION((io_context, *p.p,
85  "io_context", &io_context.impl_, 0, "overlapped"));
86 
87  io_context.impl_.work_started();
88  reset();
89  ptr_ = p.p;
90  p.v = p.p = 0;
91  iocp_service_ = &io_context.impl_;
92  }
